function wrap( t, l )
    -- change i=0 to move left and i=1 to right
    for i = 1, l do
        table.insert( t, 1, table.remove( t, #t ) )
    end
end

function shift( t, direction )

for i=1,#t do
    if(direction == "left") then
            if(i == 1) then
                tempElement = t[#t][6] 
            else
                tempElement = t[i-1][6]
            end
    else
            if(i == 7) then
                tempElement = t[1][6] 
            else
                tempElement = t[i+1][6]
            end
    end    

table.insert( t[i], 6, tempElement )
table.remove( t[i], 12)

end

end
