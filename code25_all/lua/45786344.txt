-- we include the module written in C. If it was written in Lua we could use 
-- "package.path" instead
package.cpath = package.cpath .. ";/usr/lib/i386-linux-gnu/lua/5.1/?.so"

-- we order luasql to require driver for mysql
luasql = require "luasql.mysql"

-- we order object "luasql" to ask for connection to mysql driver
env = assert (luasql.mysql())

-- we pass login info to the driver
con = assert (env:connect("linux_krozki","root","mypassword"))

-- we order connection to execute SQL query and save it to cursor
cur = assert (con:execute("SELECT * FROM stranke"))

-- we save cursor value to variable "vnos"
vnos = cur:fetch ({}, "a")

-- we continue until table has rows
while vnos do

  -- we print the variable "vnos"
  print(string.format("Ime: %s", vnos))

  -- we again transfer value from cursor to variable "vnos"      
  vnos = cur:fetch (vnos, "a")
end
