function fact(n)
  if n == 0 then return 1 end
  if n < 0 then return 0 end

  return n * fact(n-1)
end

--25! = 15511210043330985984000000 . From http://www.wolframalpha.com/input/?i=25!
local n = 15511210043330985984000000
assert(n == 15511210043330985984000000)
print(n)
assert(fact(4) == 24)
assert(fact(10) == 3628800)
print(fact(25))
assert(fact(25) == 15511210043330985984000000) --The problem!
