function updateMons2()
      for a = 1, mons2.numChildren, 1 do
        physics.addBody(mons2[a],"kinematic")
          if(mons2[a].isAlive == true) then
              (mons2[a]):translate(speed * -1, 0)            

              if(mons2[a].x < -80) then
                  mons2[a].x = 1000
                  mons2[a].y = 500
                  mons2[a].isAlive = false 
              end
          end
      end
  end
