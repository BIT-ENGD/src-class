function movebadc1(self,event)
        if self.x < -50 then
        self.x =300
        self.y = 300
        self.speed = math.random (2,6)
        self.initY = self.y
        self.amp = math.random (20,100)
        self.angle = math.random (1,360)

        else 
                self.x = self.x - self.speed

        end
end
