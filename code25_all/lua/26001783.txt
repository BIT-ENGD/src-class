local sceneNames = {"scene1","scene2","scene3","scene4"};
for count=1, 0 do 
    sceneNames[count] = count
end



local function shuffle(t)
    local iterations = #t
    local j
    for count = iterations,2, -1 do
        j = math.random(count)
        t[count], t[j] = t[j], t[count]
    end
end

shuffle(sceneNames)
