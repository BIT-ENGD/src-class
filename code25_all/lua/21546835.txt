function PrintCondition(CodeIfA, CodeIfB)
    if SomeGlobal == "A" then
        loadstring(CodeIfA);
        return true;
    elseif SomeGlobal == "B" then
        loadstring(CodeIfB);
        return true;
    else
        Error();
        return false;
    end
end

local temp = PrintCondition(
    [[
        print("global is A");
    ]],
    [[
        print("global is B");
    ]]
);
print(temp); -- prints 'true'
