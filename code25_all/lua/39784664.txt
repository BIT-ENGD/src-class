function rmsprop(x, dx, lr, alpha, epsilon, state)
    if not state.m then
        state.m = x.new(#x):zero()
        state.tmp = x.new(#x)
    end 
    -- calculate new (leaky) mean squared values
    state.m:mul(alpha)
    state.m:addcmul(1.0-alpha, dx, dx) 
    -- perform update
    state.tmp:sqrt(state.m):add(epsilon)
    x:addcdiv(-lr, dx, state.tmp)
end
