local locale = os.setlocale('')
require 'curses'

curses.initscr()
curses.cbreak()
curses.echo(0)
curses.nl(0)

local stdscr = curses.stdscr()
stdscr:clear()
stdscr:mvaddstr(0,0,'--->English,')
stdscr:mvaddstr(1,0,'--->Español')
stdscr:mvaddstr(2,0,'--->Русский (Russian)')
stdscr:mvaddstr(3,0,'--->日本語 (Japanese)')
stdscr:refresh()

stdscr:getch()
curses.endwin()

print(locale)
