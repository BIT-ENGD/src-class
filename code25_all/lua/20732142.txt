local rect = display.newRect(100, 100, 100, 100)

local moving, moving2
function moving()
    transition.to(rect, {time=500, x=300, y=100, onComplete=moving2})
end

function moving2()
    transition.to(rect, {time=500, x=100, y=300, onComplete=moving})
end

local rect = display.newRect(100, 100, 100, 100)
transition.to(rect, {time=500, x=300, y=100, onComplete=moving2})
