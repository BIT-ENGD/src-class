function read_file(file)
data = io.open(file, "r")
for char in data:lines() do
    if char == "print>" then
        print(string.sub(char, 6))
    end
end
data:close()
end

function run()
while true do
    print("Open a file")
    file = io.read()
    print("")
    read_file(file)
    print("")
end
end

run()
