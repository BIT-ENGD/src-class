function openFile(n, m)
  local f = io.open(n, m)
  if(not f) then
    return status(nil,"openFile: Nofile: "..tostring(n))
  end
  f.Read  = f.read
  f.Write = f.write
  f.Close = f.close
  f.Flush = f.flush
  return f
end
