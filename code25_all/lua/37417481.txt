local function file_Parse(input)
    local inputTbl = {}
    if input then
        for lines in input:gmatch("%g*%G*") do
            lines = lines:gsub("\r\n", "")
            table.insert(inputTbl, #inputTbl + 1, lines)
        end
    end
    return inputTbl
end
