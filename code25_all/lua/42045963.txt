function debugTable (t, indent)
    local ind = indent or "";
    local printFunc = print
    if (fibaro or {}).debug then
       function printFunc(...)
          return fibaro:debug(...)
       end
    end
    for k, v in pairs(t) do
        if(type(v) == "table") then
            -- recursive call if table
            debugTable(v, " - ");
        else
            printFunc(ind .. k .." ".. tostring(v));
        end;
    end;
end;

a = {};
c ={["33"] = 44,["55"]=43}
b = {["12"] = 30, ["11"]= 40,["66"]=c};

a["12"] = 10;
a["10"] = 11;
a["22"] = {10,["13"]=b};
a["11"] = 11;

> debugTable(a)
