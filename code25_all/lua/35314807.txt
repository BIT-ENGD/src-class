TouchListener
local function ballTouchEvent(e)

    local touchedBall = e.target
    local phase = e.phase

    if phase == "began" then

        log("Touch began Phase")    

    elseif phase == "moved" then

         log("Moved Phase")

    elseif phase == "ended" or phase == "cancelled" then

        log("Ended Phase")  

    end

    return true
end

ball:addEventListener("touch",ballTouchEvent)
