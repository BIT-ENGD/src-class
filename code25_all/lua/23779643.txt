local attacker = { 0,-1 }

local function test()

    local hitPattern = attacker

    print( "----------->> attacker", # attacker )

    --Set Loop Method
    if hitPattern[ # hitPattern ] == -1 then
        hitPattern[ # hitPattern ] = nil
    end

    print( "----->> attacker", # attacker )

end
test()
----------->> attacker 2
----->> attacker 1
