    -- Making grid
grid = {}
local i = 1
local ii = 1
mainx, mainy = love.graphics.getDimensions()
while(i<=mainx) do
    if(i%16==0) then
        while(ii<=mainy) do
            if(ii%16==0) then
                grid[i][ii] = nil
            end
            ii = ii + 1
        end
    end
    i = i+1
end
