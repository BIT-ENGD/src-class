local function fillColor(tile, color)
if color==0 then
    local g = { type="gradient", color1={ 0, 0, 0 }, color2={ 0, 1, 0 } }
    tile:setFillColor(g)
elseif color==1 then
    tile:setFillColor(0.1,0.1,0.1)
elseif color==2 then
    tile:setFillColor(0.5,0.5,0.5)
elseif color==3 then
    tile:setFillColor(0.75,0.75,0.75)
end
