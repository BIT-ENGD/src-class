function readformatEvent()

local read_data = {}
local duplicate
local unique_data = {}

for i=1,16 do
    read_data[i] = readResult(i):readData()
end

for i=1,16 do
    duplicate = 0

    for j=(i+1),15 do
        if read_data[i] == read_data[j] then
            duplicate = 1
        end
    end

    if duplicate == 0 then
        unique_data[i] = read_data[i]
    end
end

return unique_data

end
