List = {}

function List.new(...)
    local obj = { first = 0, last = -1 }

    if arg ~= nil then
        -- push any values passed in
        for k,v in pairs(arg) do
            List.pushleft(obj, v)
        end
    end

    return obj
end

function List.pushleft(list, value)
    local first = list.first - 1
    list.first = first
    list[first] = value
end
