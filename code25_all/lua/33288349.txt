function love.mousepressed( x, y, button )
  if button == "l" then
    for k, v in pairs(buttons) do
      local ins = insideBox( x, y, v.x - (v.w/2), v.y - (v.h/2), v.w, v.h)

      if ins then
        if v.action == "play" then
          loadState("game")
        end
      end
      if ins then
        if v.action == "quit" then
          love.event.quit()
        end
      end
    end
  end
end
