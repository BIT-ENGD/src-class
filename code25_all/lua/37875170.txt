local Renderer = {}

local num_of_layers = 2
local insert = table.insert
local remove = table.remove


function Renderer:create()
    local render = {}

    render.drawers = {}
    for i  = 0, num_of_layers do
        render.drawers[i] = {}
    end

    function render:addRenderer(obj, layer)
        local l = layer or 0
        insert(self.drawers(l), i, obj)
    end

    return render
end

return Renderer
