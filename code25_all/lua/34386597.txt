local limiteds = {
    test1 = {
        value = 999999999,
        itemid = -1,
        released = false
    }
}

function tablelength(T)  
    local count = 0
    for _ in pairs(T) do count = count + 1 end
    return count
end

function randomitem()
    local limited
    repeat
        local limited = limiteds[math.random(tablelength(limiteds))]
    until limited.released == false
end
