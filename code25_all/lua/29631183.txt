list = nil
list = {next=nil, value=value}

function appendToBack(list, value)
  local list = list
  if not list then 
    list = {next=nil, value=value}
  else
    local next = list.next
    while true do
      if not next then
        next = {next=nil, value=value}
        break
      else
        next = next.next
      end
    end
  end

  return list
end
