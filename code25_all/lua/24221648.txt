function DataModel.ObjectToTable(obj)
    children = {}

    for i,v in pairs(obj:GetChildren()) do
        table.insert(children, DataModel.ObjectToTable(v))
    end

    print(obj.Name)

    dmyself = {}
    dmyself.Name = obj.Name
    dmyself.Object = obj
    dmyself.Children = children

    print(#dmyself)

    return dmyself
end
