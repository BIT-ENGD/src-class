function drag(event)
if event.phase == "began" then
    firstx, firsty = event.x, event.y
end

if event.phase == "ended" then
    if firsty<event.y then
        local otherx, othery = event.x, event.y
        local x=(otherx-firstx)*4
        local y=(othery-firsty)*4
        shoot(x,y)
    end
end
end
