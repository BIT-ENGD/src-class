function(list_to_search_in, compared_value, invert)
    for k,v in pairs(list_to_search_in) do
        if invert and v ~= compared_value then
            table.remove(list_to_search_in, v)
        if not invert and v == compared_value then
            table.remove(list_to_search_in, v)
         end
    end
end
