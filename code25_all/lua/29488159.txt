local CharacterClass = 
{
 name = " ",
 hp = " ",
 velocity = " ";
}


function CharacterClass:new(o, nameCharacter, hpCharacter, velocityCharacter)
  local o = o or {}
  setmetatable( o, self )
  self.__index = self 
  self.name = nameCharacter
  self.hp = hpCharacter
  self.velocity = velocityCharacter

  return o
end
