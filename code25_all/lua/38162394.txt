function NewTrail()
ParticleSystem:createParticle(
    {
        flags = { "wall" },
        x = display.contentCenterX,
        y = display.contentCenterY,
        velocityX = 100,
        velocityY = 0,
        color = { 0, 0, 1, 1 },
        lifetime = 32.0
    }
)
end
timer.performWithDelay( 100, NewTrail, -1 )
