function love.load()
    Player = {
            x = 1, 
            y = 1,
            w = 50,
            h = 50
    }

    Player.self = love.graphics.rectangle("fill", Player.x, Player.y, Player.w, Player.h)    
end


function love.draw()
    love.graphics.draw(Player.self, Player.x, Player.y) --15 line
end
