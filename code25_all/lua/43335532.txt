function goFish()
    PressAndReleaseKey("d")
    Sleep(5)
    PressAndReleaseKey("d")
    PressAndReleaseKey("1")
    local x = GetRunningTime()
    while fishing do
        if(GetRunningTime() % 180000) == 0) then PressAndReleaseKey("b") end
        PressAndReleaseMouseButton(1)
        Sleep(4500)
        if(GetRunningTime()-x > 6000000) then
            x = getBait()
        end
    end
end
