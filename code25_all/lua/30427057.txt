-- Grid Vertical
local gx = 0 
-- Grid Horizontal
local gy = 0
-- Grid Loop
local g = 0
while g ~= 100 do
    print("Grid Loop: "..g) -- for testing purposes of the loop
    love.graphics.line( gx, 0, gx, 500)
    love.graphics.setColor( 255, 255, 255 )
    love.graphics.line( 0, gy, 1000, gy)
    love.graphics.setColor( 255, 255, 255)
    local gx=gx+50
    local gy=gy+50
    g=g+1
end
end
