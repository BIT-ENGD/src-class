function digitconverter (digit)
    if digit == "0" then
        cnumber = "ying2 "
    elseif digit == "1" then
        cnumber = "yi1 "
    elseif digit == "2" then
        cnumber = "er2 "
    elseif digit == "3" then
        cnumber = "san1 "
    elseif digit == "4" then
        cnumber = "si4 "
    elseif digit == "5" then
        cnumber = "wu3 "
    elseif digit == "6" then
        cnumber = "liu4 "
    elseif digit == "7" then
        cnumber = "qi1 "
    elseif digit == "8" then
        cnumber = "ba1 "
    elseif digit == "9" then
        cnumber = "jiu3 "
    end
    return cnumber
end

repeat
