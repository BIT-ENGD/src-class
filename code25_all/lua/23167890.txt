    tArgs = { ... }

  printUsage()
  return
end

--globals
local MAX_INV_SLOT = 16
local depth = tonumber(tArgs[1])
local bredth = tonumber(tArgs[2])

--Usage output
local function printUsage()
  print("Usage: <branch depth> <num branches>")
end

--wrap turtle.forward()
local function fwd()
 while turtle.forward() == false do
  turtle.dig()
 end
end
-- check / get fuels
local function checkFuel()
  while turtle.getFuelLevel() == 0 do
   for k=0, 16, 1 do
     if turtle.refuel(i) then
       return
     end
   end  
  return
end
--mine 1 wide 3 tall segment
local function step()
 fwd()
 turtle.digUp()
 turtle.digDown()
end

local function control()
  for i=0, bredth,1 do
    for j=0, depth,1 do
      checkFuel()
      step()
    end
    -- Turn Corner
    if (i % 2) == 0 then
      step()
      turtle.turnRight()
      step()
      step()
      step()
      turtle.turnRight()
    else
      -- Left U turn
      step()
      turtle.turnLeft()
      step()
      step()
      step()
      turtle.turnLeft()
    end
  end
  print("Should be done!")
end



control()
