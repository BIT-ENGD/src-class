function findFactors( num )
    local factors = { }
    local x = 0
    while ( x < num ) do
        x = x + 1
        if ( num % x == 0 ) then
            table.insert( factors, "Â±" .. x )
        end
    end
    return factors
end

                if num1 and num2 then
