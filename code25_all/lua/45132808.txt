Citizen = {name}

--Constructor
function Citizen:new(o, name)
  o = o or {}
  setmetatable(o, self)
  self.__index = self
  o.name = name
  return o
end

--Getters--
function Citizen:getName()
  return self.name
end

  __call = function(cls, ...)
  local self = setmetatable({}, cls)
  self:_init(...)
  return self
  end,
})

end
