function _say_(text)
    local temptext = ""
    local tbl = {}
    for word in string.gfind(text, "%a+") do
        if string.len(temptext.." "..word) >= 52 then
            table.insert(tbl, temptext)
            temptext = word
        else
            temptext = temptext.." "..word
        end
    end
    table.insert(tbl, temptext)
    for i = 1, table.getn(tbl) do
        print(tbl[i])
    end
end
