function Parent()
    local self = {}

    self.println = function(text) print(text) end
    return self
end


function Child()
    local super = Parent()
    local this = {}

    this.println = function(text)
        super.println(text) -- According to the debugger, here is the problem.
        print("Child")
    end

    for k, v in pairs(this) do
        super[k] = v
    end

    return this
end

local a = Child()
a.println("Hello!")
