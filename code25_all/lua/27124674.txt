object = {}

function object:new()
  self.elements = {}
  setmetatable(self, self)
  return self
end

function object:append(first, second)
  self.elements[#self.elements + 1] = {first, second}
end

function object:__index(k)
  return unpack(self.elements[k])
end

o:append(1,2)
o:append(3,4)

1
