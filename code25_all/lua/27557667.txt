text("Text", 0, 0, 0)
io.write("Enter the time in seconds that the timer will run: ")
local time = io.read('*number')

local function sleep(s)
   local clock = os.clock
   local t0 = clock()
   while clock() - t0 >= s do
   end
end

require('turtle')

function timer(time)
   local erase = snap()
   while time ~= 0 do
      text(time, 0, 0, 0)
      time = time - 1
      sleep(1)
      undo(erase)
   end
   text("Done", 0, 0, 0)
end

wait()
