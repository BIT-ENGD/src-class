# Introduction
grammar markdown.Markdown with org.eclipse.xtext.common.Terminals

generate markdown "http://www.Markdown.markdown"

Model:
  entities+=Entity*;

Entity:
  Section | Subsection | Paragraph
;

Section:
  '#'
    content+=TextPart
  '::'
;

Subsection:
  '##'
    content+=TextPart
  '::'
;

Paragraph:
  content+=TextPart
;

TextPart:
  text=Text
;

Text:
        (ID | WS | SINGLE_NL | MULTI_NL | ANY_OTHER | '\\[' | '\\]' | ',' | "-" | '\\:' | '\\%' | '\\#' | '\\##' )+;


terminal ID:
        ('a'..'z' | 'A'..'Z' | '_') ('a'..'z' | 'A'..'Z' | '_' | '0'..'9')*;

terminal SL_COMMENT:
        '%%' !('\n' | '\r')* ('\r'? '\n');

terminal MULTI_NL:
        '\r'? '\n' (/*(' ' | '\t')**/ '\r'? '\n')+;

terminal SINGLE_NL:
        '\r'? '\n';

terminal WS:
        ' ' | '\t';

terminal ANY_OTHER:
        .;
