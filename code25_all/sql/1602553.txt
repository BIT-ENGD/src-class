program_id (which related to table programs)

SELECT COUNT(*) as views, 
       `mp`.`title`, 
       `mp`.`program_date` as date 
FROM (`applications_media_views` as v) 
JOIN `media_programs` as mp ON `v`.`program_id` = `mp`.`program_id` 
GROUP BY `v`.`program_id` 
ORDER BY `views` desc
