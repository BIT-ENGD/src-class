ItemId INT
FirstName VARCHAR(200)
MiddleName VARCHAR(200)
LastName VARCHAR(200)
Address VARCHAR(1000)
ChildrenNames TEXT
SearchData TEXT

SELECT  *
FROM    [v_MyView] V
        LEFT OUTER JOIN CONTAINSTABLE
        (
          [v_MyView],
          (
           [FirstName],
           [MiddleName],
           [LastName],
           [Address],
           [ChildrenName],
           [SearchData]
         ), '"name" AND NOT "lastname"') FTS ON [FTS].[Key] = [V].[ItemId]
 WHERE   (ISNULL(RANK,0) > 0)
