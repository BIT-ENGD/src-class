   TableB.Name, 
   TableA.Charge, 
   ( 
     select Charge 
     from TableA 
     where (DateAdded >= '13/10/2009' and DateAdded < '14/10/2009')
   ) 
     from TableA

inner join 
   TableB on TableB.ID = TableA.ID

where 
   TableA.DateAdded >= '10/14/2009'

order by 
   Name asc
