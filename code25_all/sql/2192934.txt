LINQ(fetch)+TSQL(update) < LINQ(update)

declare @i int;
declare @nr int;
declare @p1 bit;
declare @p0 int;

set @i = 0;
set @nr = 1000000000;

while @i<10000
begin
  select @p0=id,@p1=bit_field from test_mono where name=@nr;
  set @p1 = 1- @p1;
  update test_mono set bit_field=@p1 where id=@p0;
  set @i = @i+1;
  set @nr = @nr-1;
end
