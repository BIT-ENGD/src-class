SELECT * FROM General JOIN Temp ON General.title = Temp.title JOIN Location ON Temp.title = General.title

SELECT
   General.id,
   General.url,
   General.title,
   General.description,
   Temp.id,
   Temp.url,
   Temp.title,
   Temp.description,
   Location.id,
   Location.url,
   Location.title,
   Location.description
FROM
   General INNER JOIN Temp
   ON General.id = Temp.id
   INNER JOIN Location
   ON Temp.id = Location.id
ORDER BY
   General.title
