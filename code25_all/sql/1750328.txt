  DECLARE
  @CPU_BUSY int
  , @IDLE int

  SELECT
  @CPU_BUSY = @@CPU_BUSY
  , @IDLE = @@IDLE

  WAITFOR DELAY '00:00:00:500'

  SELECT
  (@@CPU_BUSY - @CPU_BUSY)/((@@IDLE - @IDLE + @@CPU_BUSY - @CPU_BUSY) *
  1.00) *100 AS CPUBusyPct 

  SELECT
  @IO_BUSY = @@IO_BUSY
  , @IDLE = @@IDLE

  WAITFOR DELAY '00:00:00:500'

  SELECT
  (@@IO_BUSY - @IO_BUSY)/((@@IDLE - @IDLE + @@IO_BUSY - @IO_BUSY) *
  1.00) *100 AS IOBusyPct
