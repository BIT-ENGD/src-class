tChore(ChoreId int primary key, ChoreDescription nvarchar(15))

tDayChores(
    DayId int primary key,
    ChoreId int foreign key references tChore(ChoreId)
)

select top(1)
    DayId
from
    tDayChores DC
    inner join tChore C
        on C.ChoreId = DC.ChoreId
where
    C.ChoreDescription = 'vacuum'
