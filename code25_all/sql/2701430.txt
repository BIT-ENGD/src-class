forum_threads
SELECT 
  forum_threads.*, 
  forum_posts.thread_id 
FROM 
  forum_threads 
  INNER JOIN (
      SELECT MAX(id) AS id, thread_id as thread_id 
        FROM forum_posts 
    GROUP BY thread_id 
    ORDER BY id DESC
  ) forum_posts ON forum_threads.id = forum_posts.thread_id
