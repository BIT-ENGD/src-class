SELECT
  tbl2.* 
FROM
  (
   SELECT
     a, b, c, 
     ROW_NUMBER() OVER (PARTITION BY tbl1.s ORDER BY tbl1.Datetime1)) AS RowNo 
   FROM
     tbl1 
   WHERE
     tbl1.s in ('s1','s2','s2','s3','s1','s4')
  ) AS tbl2 
WHERE
  tbl2.rowno = 1
