score
    SELECT
       s.score
       , s.created_at
       , u.name
       , u.location
       , u.icon_id
       , u.photo
       , (SELECT COUNT(*) + 1 FROM scores WHERE score > s.score) AS rank
    FROM
        scores s
    LEFT JOIN
        users u ON u.uID = s.user_id
    ORDER BY
        s.score DESC
        , s.created_at DESC
    LIMIT 15

AND id > s.id
