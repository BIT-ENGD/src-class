Dim Items = From Item In DB.Items _

            Order By Item.PubDate Descending _

Dim Items = From Item In DB.Items _
            Where Item.CatID = CatID _
            And Item.ID > 600000 _
            Order By Item.PubDate Descending _
            Select Item Take 100
