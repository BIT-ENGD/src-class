CREATE PROC [dbo].[add_address1]
    @recno int OUTPUT,
    @clientID int,
    @street varchar (23),
    @city varchar (21),
    @state varchar (2),
    @zip varchar (9)
AS
    declare @s int;
    select @s = updated from is_clientindex where clientid = @clientID
    insert into is_address1
        (original_rec, clientID, street,city,state,zip)
        values (@s + 1, @clientID, @street,@city,@state,@zip);
    set @recno = @@IDENTITY;
