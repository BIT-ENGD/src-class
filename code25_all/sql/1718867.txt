CREATE TABLE course
    (COURSE_NO NUMERIC(8,0) NOT NULL
    ,DESCRIPTION VARCHAR(50) 
    ,COST NUMERIC(9,2) 
    ,PREREQUISITE NUMERIC(8,0)
    ,CREATED_BY VARCHAR(30) 
    ,CREATED_DATE DATE 
    ,MODIFIED_BY VARCHAR(30) 
    ,MODIFIED_DATE DATE 
    ,PRIMARY KEY (COURSE_NO)
    ,INDEX (PREREQUISITE)
    ) TYPE = INNODB;
