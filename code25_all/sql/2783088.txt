SELECT * FROM (
  SELECT 'a' AS a, '1' AS b
UNION 
  SELECT 'a' AS a, '2' AS b
UNION
  SELECT 'b' AS a, '1' AS b) AS a 
GROUP BY a.a 

SELECT *, GROUP_CONCAT(b) FROM (
  SELECT 'a' AS a, '1' AS b
UNION 
  SELECT 'a' AS a, '2' AS b
UNION
  SELECT 'a' AS a, '3' AS b
UNION
  SELECT 'b' AS a, '1' AS b
UNION
  SELECT 'b' AS a, '2' AS b
) AS a 
GROUP BY a.a
