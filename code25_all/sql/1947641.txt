GO ALTER PROCEDURE [dbo].[sp_GetTest]
  @TestId INT AS BEGIN
  SET NOCOUNT ON;

  SELECT T.TestId
        ,T.TestDetail
        ,T.isPassedTest     
    FROM Test T     
   WHERE T.TestId = @TestId
ORDER BY T.TestDetail

END
