       area,logo, url, email, map, 
       description, tag, catch_phrase, region
FROM   Results
WHERE  STYLE LIKE 'varStyle' 
   AND REGION LIKE 'varRegion' 
   AND BEDROOMS LIKE 'varBedrooms' 
   AND BATHROOMS LIKE 'varBathrooms'
