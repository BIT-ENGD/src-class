SELECT BranchNumber,QuantitySold
FROM Department 
UNION
SELECT BranchNumber,0
FROM ItemSale

var unionQuery = (from dept in Department
                            select new
                            {
                                dept.BranchNumber,
                                dept.QuantitySold,
                            })
                            .Concat(from item in ItemSale
                               select new
                               {
                                   item.BranchNumber,
                                   0
                               });
