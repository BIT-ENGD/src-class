    window.addEvent('domready', function() {
        var drop = $('large');
        var dropFx = drop.effect('background-color', {wait: false});
        $$('.item').each(function(item){
            item.addEvent('click', function(e) {
                drop.removeEvents();
                drop.empty();
                var a = item.clone();
                a.inject(drop);
                dropFx.start('7389AE').chain(dropFx.start.pass('ffffff', dropFx));
            });
        });
    });
