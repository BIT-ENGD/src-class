   function ExpandAll() {
        var elem = document.getElementById("repeaterTable");

        var rows = elem.rows;

        for (var i = 1; i < rows.length; i++) {
               rows.item(i).style.visibility = 'visible';
               rows.item(i).style.display = '';
        }
    }

    style.visibility = 'hidden';
    style.display = 'none';
