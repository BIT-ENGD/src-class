function pageLoadedHandler(sender, args) {  
    if (typeof(lastFocusedControlId) !== "undefined" 
         && lastFocusedControlId != "")   
    {  
        var newFocused = $get(lastFocusedControlId);  
        if (newFocused) {    
            focusControl(newFocused);
        }
    }

 function(id) { 
     return document.getElementById(id); 
 }
