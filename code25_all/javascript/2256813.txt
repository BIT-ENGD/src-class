function addSearchPattern(file, aStatus, aRule, aExpression, aCategory)
{
    $.getJSON(file, {
        status: aStatus,
        rule: aRule,
        expression: aExpression,
        categoryID: aCategory
    }, function(data){
        if(data.errors.length > 0)
        {
            var errorText = '';
            $.each(data.errors, function(i, errors){
                errorText += data.errors[i];
            })
            alert(errorText);
        }
    });
}
