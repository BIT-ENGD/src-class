gmarkers.each(function(mrk,ind){
    try {
        if (mrk.qa.image.indexOf('st1')!=-1){
            if (Browser.Engine.trident){
                mrk.Ir[0].zIndex = ind + 1000
            } else {
                mrk.Ir[0].setStyle('z-index',ind + 1000)
            }
        } else {
            if (Browser.Engine.trident){
                mrk.Ir[0].zIndex = ind
            } else {
                mrk.Ir[0].setStyle('z-index',ind )
            }

        }
    } catch (err){}
})
