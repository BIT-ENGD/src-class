getIframeWindow : function () {
    return {
        show :function(args) {
            args=args||{};
            win = new Ext.Window({ 
                width    :args.width||200,
                height   :args.height||200,
                items    :[{
                    border      : false,
                    xtype       : 'iframepanel',
                    defaultSrc  : args.src||'_blank.htm'
                }]
            });
            console.log(win.height);
            win.show();
        },
