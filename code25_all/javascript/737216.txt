$(".playThumb").fadeTo("normal", 1);

$(".playThumb").hover(function() {
    $(".playThumb").each(function() {
        if ( $(this) != $(this) ) {
            $(this).fadeTo("fast", 0.3);
        }
    });
}, function() {
    $(".playThumb").each(function() {
            $(this).fadeTo("fast", 1);
    });
});
