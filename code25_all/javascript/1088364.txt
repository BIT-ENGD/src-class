        {
            var itemPrice = 3;
            var totalChecked = 0;
            var veg = document.getElementsById("veg");
            for(j=0; j < veg.length;j ++)
            {
              if(veg[j].checked)
              {
                totalChecked += 1;
               }
            }
     return totalChecked

       if (totalChecked > 2) {
        totalChecked = totalChecked -2;

        itemPrice = itemPrice + totalChecked;

    }

}
