function nodeEmpty(nodeValue)
{
    var isEmpty = true;
    var length = nodeValue.length;
    for (var i = 0; i < length; i++)
    {
        if (nodeValue[i] != ' ' && nodeValue[i] != '\n' && nodeValue[i] != '\t')
        {
            isEmpty = false;
            break;
        }
    }

    return isEmpty;
}
