function parentA (base){
  function childA (exp){
    return Math.pow(base, exp);
  }

  alert(childA(2)); // call private function
}

parentA(4); // alerts 4^2 = 16


var parentB = function(str1){
  return {
    childB: function(str2){
      alert(str1 + str2);
    }
  };
}('foo');

parentB.childB('bar'); // alerts 'foobar'
