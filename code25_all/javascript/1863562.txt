function grabMetrics(url, locationAry, num){
alert(num);
$.ajax({
    type: "POST",
    url: "foo.php",
    data: url,
    success: function($this){
        $("#"+locationAry[num]).text($this);
        if(num<5){
            num++;
            grabMetrics('toggle='+num, locationAry, num);
        }
    }
});
}
