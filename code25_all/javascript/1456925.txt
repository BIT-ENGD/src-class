jQuery(document).ready(function () {
    // Hiding of prev & next arrows on movie pages.
    var arrows = false;

    if (jQuery('body').find('div.video'))
    {
        arrows = false;
    } else {
        arrows = true;
    }

    if (
        arrows == true
    ){
        jQuery('a.prev').show();
        jQuery('a.next').show();
    } else {
        jQuery('a.prev').hide();
        jQuery('a.next').hide();
    }

});
