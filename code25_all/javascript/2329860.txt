function checkSession(){
    $.ajax({url: "session.php", success: function(data){
         if( data == 1){
             var postFilen = 'msg.php';
             $.post(postFilen, function(data){
                 $("#msg").html(data).find("#message2").fadeIn("slow")
            }
         } else {
             $('#message2').hide();
         }
    }});
// setInterval('checkSession()',1000);
}
