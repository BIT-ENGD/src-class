function wopen(url, name, w, h)
{
    var win = window.open(url,
                  name, 
                  'width=' + w + ', height=' + h + ', ' +
                  'location=no, menubar=no, ' +
                  'status=no, toolbar=no, scrollbars=no, resizable=no');
    win.resizeTo(w, h); 
    win.focus();       
}
