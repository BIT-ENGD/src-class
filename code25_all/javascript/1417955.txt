$("#down").click(function() {
  var s = $("#slider"), val = s.slider("value"), step = s.slider("option", "step");
  s.slider("value", val - step);
});

$("#up").click(function() {
  var s = $("#slider"), val = s.slider("value"), step = s.slider("option", "step");
  s.slider("value", val + step);
});
