function btnOkClick()
{
    if(validateAll())
    {
        var behavior = $find('btnOkPopupBehavior');
        if (behavior)
        {
            behavior.show();
        }
        else
        {
            var lblOutput = $get('<%= lblOutput .ClientID %>');
            lblOutput .innerText = 'Couldn't find popup';
        }
    }
}
