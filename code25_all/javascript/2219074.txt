public void onCreate(Bundle savedInstanceState) {

        setContentView(R.layout.main);
        webview = (WebView) findViewById(R.id.webview);
        webview.setWebViewClient(new HelloWebViewClient());

        webview.loadUrl("http://ebay.com");         
    }
