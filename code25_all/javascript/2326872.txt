location.hostname
$('a').each(function() {

    var href = $(this).attr('href');

    if(this.hostname && this.hostname !== location.hostname) 
    {    
        $(this)
            .removeAttr('target')
            .attr('rel', 'external')                
            .attr('title', href)
            .click(function() {
                window.open($(this).attr('href'));
                return false;
            });
        }
});
