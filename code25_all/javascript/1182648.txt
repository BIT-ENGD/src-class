        var timer;

        $('#menubutton').click(function() {
            $('#menucontent').show();
        });

        $('#menubutton').mouseout(function() {
            timer = setTimeout('$("#menucontent").hide()', 500);
        });


        $('#menucontent').mouseover(function() {
            clearTimeout(timer);
        }).mouseout(function() {
            setTimeout('$("#menucontent").hide()', 300);
        });
