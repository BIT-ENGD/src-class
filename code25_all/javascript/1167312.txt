function LoadReplacementData(replacementData)
{
    var text = $("#textblock").html();

    $.each(replacementData, function(index, item)
    {
        var expression = new RegExp("\[sub:" + item.Key + "\]", "g");
        text = text.replace(expression, item.Value);
    });

    $("#textblock").html(text);
}
