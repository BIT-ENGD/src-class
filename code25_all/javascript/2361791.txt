function ClearForm(form)
{

}

function CuptoNum(str)
{
if((str >= 0) && (str <= 9))
return str;
switch(str.toUpperCase()) {
case "A": return 4;
case "B": return 5;
case "C": return 6;
case "D": return 7;
case "E": return 8;
case "F": return 9;
default:  alert('You must choose a number between 0 and 9 or a letter between A and F!');
return 'X';
}
