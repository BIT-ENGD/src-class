$().ready(function() {

          $('#dialog').dialog({
                           autoOpen:false,
                           title:  $(this).attr("title"),
                           modal: true, width: 450, height:"auto", resizable: false,
                           close: function(ev, ui) { $(this).remove(); },
                           overlay: {
                           opacity: 0.5,
                           background: "black"
                           }
                           });

   $(".mybutton").click(function(){

        $.post($(this).attr("href"), { },
            function(data) {
                $('#dialog').html(data);

            }

        );
           $('#dialog').dialog('open');
           return false;
      });

});
