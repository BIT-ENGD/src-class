function loadData() {
     var url = "/Testx.mvc/GetData";
     var id = "111111";
     var format = "html";

     $.ajax({
         url: url,
         type: "POST",
         dataType: format,
         data: "id=" + id,
         success: populateResults
     });
 }

 function populateResults(result) {
     $('#results').html(result);
 }
