        function parseURLtoVars(strLocation){
        var rArray = new Array();
        var key;
        var urlString = new String(strLocation);

        if (urlString.search(/\?/)>-1){
            var qArray = urlString.split('?')[1].split('&');

            if (qArray.length > 0){
                for (key in qArray){
                    var arVal = qArray[key].split('=');
                    if (arVal.length ==2){
                        rArray[arVal[0]] = arVal[1];
                    } else {
                        continue;
                    }
                }
                return rArray;
            } else {
                return false;
            }
        }
        return false;
    }
