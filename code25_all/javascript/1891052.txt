function areaMe(area) {
    var barea = $('#barea').val();
    if (barea.indexOf(area) != -1) {
        alert ("..." + barea + "..." + area + "...");
        barea.replace(area, "cu"); // Remove
        alert ("..." + barea + "..." + area + "...");
    }
    else {
        barea += area + ' '; // Include.
    }
    $('#barea').val(barea);
}
