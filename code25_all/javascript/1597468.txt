function parseNumber(number)
{
    if (number % 1 == 0)
        return parseInt(number);
    else 
        return parseFloat(number);
}

function parseNumber(number)
{
    if (number.indexOf(".") != -1)
        return parseFloat(number);
    else
        return parseInt(number);
}

function parseNumber(number)
{
    if (!number.matches(/([0-9]+.[0-9]+)/))
        return parseInt(number);
    else
        return parseFloat(number);
}
