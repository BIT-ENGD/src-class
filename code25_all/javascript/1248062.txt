function hideAllSubMenu(){
    var submenu = document.getElementsByTagName("div");
    for(var i = 0; i < submenu.length; i++)
    {
        if(submenu.item(i).className == "submenu_wrap")
            submenu.item(i).style.display = "none";
    }
}
