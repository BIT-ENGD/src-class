function CallPrint() {
        var prtContent = document.getElementById('<%= pnlDelete.ClientID %>');

        winPrint.document.write("<h3>Summary</h3><br />" + prtContent.innerHTML);
        winPrint.document.close();
        winPrint.focus();
        winPrint.print();
        winPrint.close();
