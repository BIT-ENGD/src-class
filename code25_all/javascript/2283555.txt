var locations = {
  'us-lax': {
    'name': 'Los Angeles'
  },
  'us-nyc': {
    'name': 'New York'
  }
};

// this is the order in which they'll appear
var all_locations = ['us-lax', 'us-nyc'];

function bar() {
  $('#foo').prepend('<h3>foo</h3>');
  for (i in all_locations) {
    location = locations[all_locations[i]];
    loc = $('<li></li>');
    loc.html(location['name']);
    loc.appendTo('#foo');
  }
};

$(document).ready(function() {
  bar();
});
