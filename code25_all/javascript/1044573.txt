    function setKeyHook(doc)
    {
        try{               
            if (doc)
                if (parent.TOP.handleKeypress){
                    doc.onkeydown = parent.TOP.handleKeypress;
                    logMessage('Attached handler');
                }
                else{
                    logMessage('No handleKeypress');
                }
            else
                logMessage('No doc');
        }
        catch (ex){
            logMessage(ex.toString());
        }
    }
