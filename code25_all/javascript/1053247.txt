var object1 = function(){ 
  //do something 
}
var object2 = function(){
  //do something else
}

var objects = {
  'o1' : object1,
  'o2' : object2  
};

var actions = [];

function addAction( actionName ){
 var object = objects[actionName];
 actions.push( function(){ new object(); } );
}

addAction( "o1" );
addAction( "o2" );

for( i in actions ){
  actions[i]();
}
