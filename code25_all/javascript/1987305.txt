        var memberStore = new Ext.data.JsonStore({
            proxy : new Ext.data.HttpProxy({
                url : url/getMembers,
                method : 'POST'
            }),
            fields : ['username'],
            root : 'members'
        });

//function that loads store when it is called
    function getMembers() {

        memberStore.load({
            //params : {action : "getmembers", groupname : "GROUP_Test"}
            jsonData : {
                action : "getmembers",
                groupname : "GROUP_Test"    
            }
        }); 
    }
