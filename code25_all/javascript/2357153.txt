<script type="text/javascript">
 $(document).ready(function(){

   $(".heading:not(#Container1)").click(function(){
     var c = $(this).next(".container");
     c.slideToggle("slow");
   });


 });

 $(document).ready(function(){
   $("img[alt='min']").click(function(){
     var c = $(this).next(".container");
     c.slideUp("slow");
   });

   $("img[alt='max']").click(function(){
     var c = $(this).next(".container");
     c.slideDown("slow");

   });
 });
</script>
