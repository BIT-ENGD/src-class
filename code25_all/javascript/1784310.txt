function FormSubmit(formid) {

this.initialize = function(formid) {
    this.formid = formid;
    $(formid).observe('submit', this.send);
}


this.send = function() {
    alert('sending '+this.formid)
    this.formid.onSubmit = true;
}

this.initialize(formid); 
}
