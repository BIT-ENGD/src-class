        var visible_status=0;
        function selectFn(sno){
        if(sno==1){
                $('#id2').hide();
                $('#id1').show();
                visible_status=1;
        }else if(sno==2){
                $('#id2').show();
                $('#id1').show();
                visible_status=2;
        }
        }

        function popitup(url) {
            newwindow=window.open(url+'?parent_status='+visible_status,'name');
            if (window.focus) {newwindow.focus();}
                return false;
        }
