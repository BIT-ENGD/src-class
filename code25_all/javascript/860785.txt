function getMyApp(appName) {                
    if (navigator.appName.indexOf ("Microsoft") !=-1) { 
        return window[appName];             
    } else {                        
        return document[appName];           
    }                           
}                           

function swfLayerToggle(layer, isOn) {          
    getMyApp("MAPublisherSWF").swfLayerToggle(layer, isOn); 
}
