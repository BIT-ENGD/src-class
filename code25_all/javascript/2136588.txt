            var over = false;
            $(function(){
                $("#s_1").hover(function(){
                    over = true;
                    swing_left_anim("#s_1");
                }, function(){
                    over = false;
                });
   $("#np_1").hover(function(){
                    over = true;
                    swing_left_anim("np_1");
                }, function(){
                    over = false;
                });
            });

            function swing_left_anim(obj){
                $(obj).animate({
                    rotate: '0deg'
                }, {
                    duration: 500
                });
                if (over) {
                    $(obj).animate({
                        rotate: '25deg'
                    }, 500, swing_right_anim(obj));

                }
            }

            function swing_right_anim(obj){
                $(obj).animate({
                    rotate: '-25deg'
                }, 500, swing_left_anim(obj));
            }
