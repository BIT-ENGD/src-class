 function validateMinTrans(sender, args) {
        var error = true;
        var txtMinTrans = document.getElementById('TxtMinTrans');
        var regexp = new RegExp("^[0-9]{1,40}(\.[0-9]{1,2})?$");
        if (txtMinTrans.value.match(regexp)) {
            alert("good");
        }
        else {
            alert("bad");
        }

        if (error)
            args.IsValid = false;
        else
            args.IsValid = true;
    }
