        fetch: function(site){
                var feed;
                var req = new Request({
                        method: this.options.method,
                        url: this.options.rssFetchPath,
                        data: { 'url' : site },
            onRequest: function() {
                                if (this.options.targetId) { $
(this.options.targetId).setProperty('html',
this.options.onRequestMessage); }
                        }.bind(this),
                        onSuccess: function(responseText) {
                                feed = responseText;
                        }
                });
                req.send();
                return feed;
        }
