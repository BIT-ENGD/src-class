        function doRequest(url){
            var req = new XMLHttpRequest();

            req.onreadystatechange = function(){
                if (req.readyState == 4) {
                    var str = req.responseText;
                    runtime.trace("deRequest result: " + str);

                        return str;
                }
            };
            req.open('GET', url, true);
            req.send(null);
        }
