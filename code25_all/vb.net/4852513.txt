    Dim fahrenheit, celsius As Double
    Console.WriteLine("Enter a value for fahrenheit tempature:")
    fahrenheit = Convert.ToDouble(Console.ReadLine())
    celsius = 5 / 9 * (fahrenheit - 32)
    fahrenheit = celsius * 9 / 5 + 32
