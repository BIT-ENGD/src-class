Public Function ReadUInt32() As UInt32
    Dim b(4) As Byte
    input.Read(b, 0, 4)
    Dim out As UInt32 = CUInt(If(IsBigEndian, b(0), b(3))) << 24
    out += CUInt(If(IsBigEndian, b(1), b(2))) << 16
    out += CUInt(If(IsBigEndian, b(2), b(1))) << 8
    out += CUInt(If(IsBigEndian, b(3), b(0)))
    Return out
End Function
