Function addtab()
    Dim myTabPage As New TabPage()
    Dim theweb As New WebBrowser

    myTabPage.Text = "TabPage Test" & (TabControl1.TabPages.Count + 1)
    TabControl1.TabPages.Add(myTabPage)

    theweb.GoHome()
    theweb.Parent = myTabPage
    theweb.Visible = True
    theweb.Dock = DockStyle.Fill
    Return True
End Function
