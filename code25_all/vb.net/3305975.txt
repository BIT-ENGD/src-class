Public Class Form1  
Dim i = 0  
Dim re = True  
Private Sub Timer_Tick(ByVal sender As System.Object, ByVal e As System.EventArgs)   Handles Timer.Tick  
    If i = 225 Then  
        re = False  
    ElseIf i = 0 Then  
        re = True  
    End If  
    If re = True Then  
        i += i  
    ElseIf re = False Then  
        i -= i
    End If  
    color1.BackColor = Color.FromArgb(i, i, i)  
End Sub  
End Class
