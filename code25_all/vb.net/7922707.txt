        Dim ren_date, mem_date As Date
        Dim renmon As String

        If ComboBox1.SelectedItem = "Silver" Then
            ren_date = mem_date.AddMonths(3)
            renmon = ren_date.ToString("MMMM")

            renmon = ren_date.ToString("MMMM")
        ElseIf ComboBox1.SelectedItem = "Gold" Then
