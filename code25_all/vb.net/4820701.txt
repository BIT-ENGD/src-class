  Private v_integer As Integer

  Public Sub New()

      v_integer = -1
  End Sub

  <DataMember()> _
  Public Property P_String() As String
    Get
        Return v_string
    End Get
    Set(ByVal value As String)
       v_string = value
    End Set
  End Property

  <DataMember()> _
  Public Property P_Integer() As Integer
    Get
        Return v_integer
    End Get
    Set(ByVal value As Integer)
       v_integer = value
    End Set
  End Property
End Class
