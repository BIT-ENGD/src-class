Class GreaterThan
Public Shared Sub Main()
    Try
        Dim Y As Integer = 12
        Dim X As Integer = 23
        If Y > X Then
            System.Console.WriteLine(Y)
        Else
            System.Console.WriteLine(X)
            Throw New Exception("Greater Than")
        End If
    Catch ex As Exception
        System.Console.WriteLine(ex.Message)
    End Try
End Sub
