Public Class Period

    Property FromDate As Date
    Property ToDate As Date

    Public Sub New(ByVal fromDate As Date)

        If fromDate > ToDate Then
            Throw New ArgumentException("fromDate must be less than or equal to toDate")
        End If

        _FromDate = fromDate
        _ToDate = ToDate

    End Sub


End Class
