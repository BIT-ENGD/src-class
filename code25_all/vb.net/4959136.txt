Dim arr As Variant
Dim unit As Integer
Dim i As Integer

unit = FreeFile
Open "SomeFile.dat" For Binary As unit

ReDim arr(1 To 10)
For i = 1 To 10
    arr(i) = i
Next
Put #unit, , arr
Close #unit

Imports Microsoft.VisualBasic

Dim mFileNumber as Integer = FreeFile()
FileOpen(mFileNumber, "SomeFile.dat", OpenMode.Binary, OpenAccess.Read)

Dim arr as Object
FileGetObject(mFileNumber, arr)
