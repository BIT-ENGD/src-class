Function Configuration() As ActionResult

        Dim configList As List(Of String) = New List(Of String)
        configList.Add("A")
        configList.Add("B")

        configList.Add("D")
        configList.Add("test")

        ViewData("cprotocols") = New SelectList(configList)

        Return View()
    End Function

                Return "You picked: wat even is C"
            Case "D"

        End Select

        Return [String].Empty
    End Function
