'Language
Private _Language As String = ""
Public Property Language() As String
    Get
        If _Language <> "" Then
            _Language = _Language.Remove(0, 9)
            _Language = _Language.Remove(_Language.Length - 3, 3)
        End If
        Return _Language
    End Get
    Set(ByVal value As String)
        _Language = "<![CDATA[" & value & "]]>"
    End Set
End Property
