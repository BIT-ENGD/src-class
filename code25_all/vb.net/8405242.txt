ListView
Public Sub SubmitLog(ByVal Data As String)

  If Data <> "" Then
    With lstLogs
      .Items.Add(New ListViewItem(New String(1) {Date.Now.ToString(), Data}))
      .Items.Item(.Items.Count - 1).EnsureVisible()
    End With
  End If

End Sub

 For I As Integer = 0 To 50
   SubmitLog(Path.GetRandomFileName())
 Next
