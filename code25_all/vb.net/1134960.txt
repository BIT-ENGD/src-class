Protected Function FormatIntegerToText(ByVal value As Object, ByVal nullvalue As String) As String
       ' Make sure value is not null... if so, return "-"
       If value = Null.NullInteger() Then
           Return nullvalue
       Else
           Return value.ToString()
       End If
End Function
