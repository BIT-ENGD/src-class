    Public Function parseText(ByVal str As String, ByVal tag As String) As String
        Dim match As Match = Regex.Match(str, "<" & tag & "\b[^>]*>(.*?)</" & tag & ">")
        If match.Groups.Count = 2 Then
            Return match.Groups(1).Value
        Else
            Return "0"
        End If
    End Function
