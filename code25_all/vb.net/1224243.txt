    If isPlay = True Then
        If isPaused = False Then
            btnPlay.Image = Image.FromFile("iconPause.png")
            isPaused = True
            isPlay = False
        End If
        GoTo Endline
    End If

    If isPlay = False Then
        If isPaused = True Then

            btnPlay.Image = Image.FromFile("iconPlay.png")
            isPaused = False
            isPlay = True
        End If
    End If
