Dim m_oMissing As Object = System.Reflection.Missing.Value
Dim m_oWord As Word.Application

Public Sub New()
    m_oWord = New Word.Application() 
    m_oWord.Visible = True        
End Sub

Public Sub CloseApp()
    m_oWord.Quit(False, m_oMissing, m_oMissing)
    Marshal.FinalReleaseComObject(m_oWord)
    GC.Collect()
    GC.Collect()
    GC.WaitForPendingFinalizers()
    GC.WaitForPendingFinalizers()
End Sub
