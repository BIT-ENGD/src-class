   Private Function GetWebDataGridOKButtonId() As String
        Dim ctls As ControlCollection = _
        WebDataGrid1.Controls(0).Controls(0).Controls

        Dim btn As Button
        Dim qry = From item As Control In ctls _
            Where item.ID = "ButtonOK" _
        Select item

        btn = qry.ToList()

        Return btn.ClientID

    End Function
