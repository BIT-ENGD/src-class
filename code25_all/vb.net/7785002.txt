        If Ai.SpriteRectangle.X < _sprite.SpriteRectangle.X Then
            MoveRight()
        ElseIf Ai.SpriteRectangle.X > _sprite.SpriteRectangle.X Then
            MoveLeft()
        ElseIf Ai.SpriteRectangle.Y < _sprite.SpriteRectangle.Y Then
            MoveDown()
        ElseIf Ai.SpriteRectangle.Y > _sprite.SpriteRectangle.Y Then
            MoveUp()
        End If
