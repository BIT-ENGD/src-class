Private Sub BtnNew_Click()
    Dim rsClone As Recordset
    Dim pVal As Integer

    Set rsClone = Me.RecordsetClone

    If Not (rsClone.BOF) Then
        DoCmd.GoToRecord , , acNewRec
        rsClone.MoveLast
        pVal = rsClone.AbsolutePosition + 2
        Me.CatgId.Value = pVal
        Me.CatgId.SetFocus
    Else
        rsClone.AddNew
        Me.CatgId.Value = 1
        Me.CatgId.SetFocus
    End If
End Sub
