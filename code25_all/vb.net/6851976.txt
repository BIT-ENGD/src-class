    'Dim declaration and calculation
    Dim MonthlyPayment_Double As Double = (LOAN_AMOUNT_Integer * RATE_Double / (1 - (1 + RATE_Double ^ -YEARS_Integer)))

    ResultLabel.Text = MonthlyPayment_Double
End Sub
