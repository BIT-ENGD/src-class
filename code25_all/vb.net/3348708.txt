Imports System.Windows.Forms

Public Class randtest
  Inherits ComboBox

  Public Sub New()
    setDefaultText()
    fillComboBox()
  End Sub

  Private Sub setDefaultText()
    Text = Rnd(10)
  End Sub

  Private Sub fillComboBox()

    For count = 0 To 5
      Items.Add(Rnd(10))
    Next

  End Sub
End Class
