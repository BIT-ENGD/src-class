Sub Main()
    Dim numberofdays As Integer
    Dim month As Integer
    Dim year As Integer

    Select Case month
        Case 4, 6, 9, 11
            numberofdays = 30
        Case 1, 3, 5, 7, 8, 10, 12
            numberofdays = 31
        Case 2
            Select Case year
                Case 2004, 2008, 2012, 2016
                    numberofdays = 29
                Case Else
                    numberofdays = 28
            End Select
    End Select
    Console.WriteLine("Please enter your month")
    Console.ReadLine()
End Sub
