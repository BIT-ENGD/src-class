 Sub GenerateNumbers()
    Dim numbers(2) As Integer
    Dim names(5) As String
    Dim x As Integer

    names(0) = "Cherries"
    names(1) = "Oranges"
    names(2) = "Plums"
    names(3) = "Bells"
    names(4) = "Melons"
    names(5) = "Bar"

    For x = 0 To 2
        numbers(x) = names(CInt(Int((6 * Rnd()) + 1)))
    Next x
End Sub
