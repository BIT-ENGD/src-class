ALTER FUNCTION [abc].[GET_PERSON_TYPE]
( 
@person_national_id CHAR(16)
)
RETURNS CHAR(1) AS
BEGIN 

declare @person_type CHAR(1)

SET @person_national_id = (
                SELECT p_type
                FROM abc.persons
                WHERE national_id = @person_national_id
                )

RETURN @person_national_id

END
