    $(function () {
        var itemToDelete
        $('#deleteDialog').dialog({
            autoOpen: false,
            width: 400,
            resizable: false,
            title: 'Delete Item',
            modal: true,
            open: function(event, ui) {
                //This calls the action if I don't try to pass in the variable
                $(this).load("<%: Url.Action("Delete") %>",
                    {  
                        item: itemToDelete 
                    });
                },
                buttons: {
                "Close": function () {
                    $(this).dialog("close");
                }
            }
        });

        $('#item-delete-button').click(function () {
            itemToDelete = $(this).find("input").val()
            $('#deleteDialog').dialog('open');
        });
    });
