Private Sub SendData(ByVal stringArray As String())
    SendData(GetMessageString(stringArray))
End Sub

Public Function GetMessageString(ByVal array As String()) As String
    Dim str As String = ""
    For i = 0 To array.Length - 1
        str = str & array(i)
        If i < array.Length - 1 Then
            str = str & "|"
        End If
    Next
    Return str
End Function
