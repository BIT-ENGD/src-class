Dim text as string = string.empty
Using sr As New StreamReader("C:\foo\AssemblyInfo.vb")
    text = sr.ReadToEnd()
    Dim fileVerReg As New Regex("^[^']*(AssemblyVersion[(""].)([^""]*)")
    Dim m As Match = fileVerReg.Match(text)
    If m.Success Then
         MsgBox(m.Groups(2).Value)
    End If
 End Using
