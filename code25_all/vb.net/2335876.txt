    Dim curRow As Data.DataRow = mySession.tWorksiteOtherMeasures.NewRow()

    curRow("lngWorksiteID") = getSelectedWorksiteID(getSelectedSiteID())

    curRow("lngPoints") = -1

    mySession.tWorksiteOtherMeasures.Rows.Add(curRow)
