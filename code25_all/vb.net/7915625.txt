Dim file As String = "C:\Users\test.txt"
    Dim Line As String

If System.IO.File.Exists(file) = True Then
        Dim objReader As New System.IO.StreamReader(file)
        Do While objReader.Peek() <> -1
            Line = Line & objReader.ReadLine() & vbNewLine
Loop
Next

 objReader.Close()
 Else

    End If
