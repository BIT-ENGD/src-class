 Public Function getXMLdeObjecto(ByVal obj As Object)
        Dim ser As New XmlSerializer(obj.GetType())
        Dim sb As New System.Text.StringBuilder()
        Dim writer As New System.IO.StringWriter(sb)
        ser.Serialize(writer, obj)
        Return sb.ToString()
    End Function
