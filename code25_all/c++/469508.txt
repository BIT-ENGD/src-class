class A
{
  virtual void func1();
}

class B : public A
{
}

class C : public A
{
  virtual void func1();
}

class D : public B, public C
{
}

int main()
{
  D d;
  d.func1(); // Causes warning
}

class D : public B, public C
{
  using B::func1();
}

class D : public B, public C
{
  virtual void func1() { B::func1(); }
}
