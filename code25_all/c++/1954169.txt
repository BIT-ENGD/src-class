ItemType BstClass::rRetrieve(node* trav, KeyType key, bool& inTree)
{
    if(trav == NULL)
    {
        inTree = false;
        ItemType garbage;
        return garbage - 1;
    }
    if(trav->data.key == key)
    { 
        inTree = true;
        return trav->data; 
    }
    else if (key < trav->data.key) 
        return(rRetrieve(trav->left, key, inTree)); 
    else
        return(rRetrieve(trav->right, key, inTree)); 
}// end rRetrieve

return garbage - 1;
