#include <iostream>

class A {
public:
    virtual ~A() {}
};

class B : public A {
};

class C : public A {
};

int main()
{
    B b;
    std::cout<< typeid(b).name()<< std::endl;       // class B

    A* a = dynamic_cast<A *>(&b);
    if (a) {
        std::cout<< "is child of A"<< std::endl;    // printed
    }

    C* c = dynamic_cast<C *>(&b);
    if (c) {
        std::cout<< "is child of C"<< std::endl;    // Not printed
    }
    getchar();
}
