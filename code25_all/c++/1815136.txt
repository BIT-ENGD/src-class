MSG mssg;

// run till completed
while (true) {

    // is there a message to process?
    while(PeekMessage( &mssg, NULL, 0, 0, PM_REMOVE)) {
        // dispatch the message
        TranslateMessage(&mssg);
        DispatchMessage(&mssg);
    }
    if(mssg.message == WM_QUIT){
        break;
    }
    // our stuff will go here!!
    Render();
    listeners->OnUpdate();
}
