void OnMapLoad(uint32 MapID)
{
    if (MapID == 1234)
    {
        printf("Map 1234 has been loaded");
    }
}

void SetupOnMapLoad(ScriptMgr *mgr)
{
    mgr->register_hook(HOOK_ON_MAP_LOAD, (void*)&OnMapLoad);
}
