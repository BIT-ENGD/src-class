#include <iostream>
using namespace std;


class Buffer {
public:
   operator char *() { cout << "operator const * called" << endl; return buff; }
private:
    char buff[1024];
};


int main(int, char**) {
    Buffer b;
    (char *)b;  // Buffer::operator char * is called here

    return 0;
}

(char *)b;
