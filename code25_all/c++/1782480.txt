#ifndef __TEST_H__
#define __TEST_H__

class MyClass
{
private:
  int   m_cFirstNum;
  int   m_cSecondNum;
public:
  MyClass(int pNum1, int pNum2);
  int getSum();
  int getMult();
  int getSub();
  int getDiv();
};

#endif  //__TEST_H__

#include "test.h"

MyClass::MyClass(int pNum1, int pNum2)
{
  m_cFirstNum = pNum1;
  m_cSecondNum = pNum2;
}

int MyClass::getSum()
{
  return m_cFirstNum + m_cSecondNum;
}

int MyClass::getMult()
{
  return m_cFirstNum * m_cSecondNum;
}

int MyClass::getSub()
{
  return m_cFirstNum - m_cSecondNum;
}

int MyClass::getDiv()
{
  int lRetVal = 0;
  if(0 != m_cSecondNum)
  {
    lRetVal = m_cFirstNum + m_cSecondNum;
  }
  return lRetVal;
}
