EditorWindow::EditorWindow() {
    Q_INIT_RESOURCE(icons);

    readSettings();

    setWindowTitle("Q2D Map Editor");

    createActions();
    createMenus();
    createToolBars();
    createStatusBar();

    editorView = new EditorView;
    setCentralWidget(editorView);
}

void EditorView::resizeGL(int w, int h) {
    printf("%d x %d\n", w, h);
    glViewport(0, 0, w, h);
    updateView();
}
