bool quack::popFront(int& nPopFront)
{   
    nPopFront = items[top+1].n;
    if ( count >= maxSize ) return false;
    else
    {
        items[0].n  = nPopFront;
        intFrontPtr = &items[0].n;
        intBackPtr  = &items[count-1].n;
    }
    for (int temp; intFrontPtr < intBackPtr ;)
    {
         ++intFrontPtr;
         temp = *intFrontPtr;
        *intFrontPtr = temp;
    }
    return true;
}
