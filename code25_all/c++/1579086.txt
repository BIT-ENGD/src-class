#include <iostream>
#include <fstream>

#include <cstring>
#include <cerrno>

#include <time.h>

using namespace std;

int main(int argc, char* argv[])
{
    ofstream file;
    file.open("test.txt");
    if (!file) {
        cout << strerror(errno) << endl;
    } else {
        cout << "All is well!" << endl;
    }

    for (int i = 0; i < 10; i++) {
        file << i << "\t" << time(NULL) << endl;
    }

    file.flush();
    file.close();
    return 0;
}
