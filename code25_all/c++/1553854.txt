// template.h
template<typename T>
class Templ {
public:
  static int templStatic;
};

template<typename T> Templ<T>::templStatic = 0;

// unit1.cpp
#include "template.h"

int method1() {
  return Templ<void>::templStatic++;
}

// unit2.cpp
#include "template.h"
int method2() {
  return Templ<void>::templStatic++;
}

// main.cpp
#include <iostream>
int method1();
int method2();

int main(int argc, char** argv) {
  std::cout << method1() << std::endl;
  std::cout << method2() << std::endl;
}
