class Foo : public IFoo {...};

template <typename T>
class Singleton
{
public:
static T* Instance() {
   if (m_instance == NULL) {
      m_instance = new T();
   }
   return m_instance;
}
private:
static T* m_instance;
};

IFoo::Instance()->foo();
class IFoo : public Singleton<IFoo> {...};
class Foo : public IFoo, public Singleton<Foo> {...};
