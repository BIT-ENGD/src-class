#include <iostream>
#include <fstream>
#include <cstdlib>

int main() {
    std::fstream f1("/tmp/test");
    if (!f1) {
        std::cerr << "f1 failed\n";
    } else {
        std::cerr << "f1 success\n";
    }
    FILE *f2 = fopen("/tmp/test", "w+");
    if (!f2) {
        std::cerr << "f2 failed\n";
    } else {
        std::cerr << "f2 success\n";
    }
}
