template<typename A, typename B>
class SpecialModel<A, B> {
};

template<typename M>
class Worker<M> {

    Worker(M& model);

    void work();
 };

template<typename A, typename B>
class Worker<SpecialModel<A, B> > {

    Worker(SpecialModel& model);

    void work();
};

template<typename W>
class Manager<W> {

    Manager() {};

    static void getThingsDone(W worker) {
        ...;
        worker.work();
        ...;
    };
};
