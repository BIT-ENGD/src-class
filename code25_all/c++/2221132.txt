//
// Create a brand new registry key
//
LONG Registry::CreateRegister( std::string path )
{
    HKEY hKey;  
    DWORD dwDisposition;

    LONG openRes = RegCreateKeyEx( HKEY_CLASSES_ROOT,
                                   path.c_str(),
                                   0, 
                                   NULL,
                                   REG_OPTION_NON_VOLATILE,
                                   KEY_ALL_ACCESS,
                                   NULL,
                                   &hKey,
                                   &dwDisposition );    

    RegCloseKey( hKey );

    return openRes;
}
