    int position = 0;

    double *myNegArray = nVector;
    double *myMatrix = fromVector;

    for(int i = 0; i < m*n; i++)
        if(*(*(myMatrix+i)) < 0){
            *(myNegArray+position) = *(*(myMatrix+i));
            position++;
        }

    return myNegArray;
}
