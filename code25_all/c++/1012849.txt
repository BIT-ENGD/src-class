class Rectangle
{
   float w;
   float h;
   const float area() {return w*h;}
};

class Poker : Rectangle
{
    int style;  // Diamond, Club, ....
    int point;  // A~10, J, Q, K
};

class BusinessCard : Rectangle
{
    string name;
    string address;
    string phone;
};

Rectangle* rec;
rec = new Poker();
delete rec;
rec = new BusinessCard();
