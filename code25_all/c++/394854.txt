template <class T>
class Uncopyable
{
  protected:
    Uncopyable() {}
    virtual ~Uncopyable() {}
  private:
    Uncopyable(const Uncopyable &);
    T & operator=(const T&);
};

class Entity : private Uncopyable<Entity> { }

class Entity has pointer data members
but does not override Entity(const Entity&)
or operator=(const Entity&)
