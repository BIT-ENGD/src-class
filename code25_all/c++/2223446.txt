struct VariantMap {
    string name;
    map<string, Variant> children;
    bool isArray;

    VariantMap(VariantMap & variantMap) {
        name = variantMap.name;
        children = variantMap.children;
        isArray = variantMap.isArray;
    }

    VariantMap() {
        isArray = false;
    }
};
