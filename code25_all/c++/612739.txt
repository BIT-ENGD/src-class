#include <iostream>

using namespace std;

char* tolower(char* src);

int main (int argc, char * const argv[])
{
    char* hello = "Hello, World!\n";

    cout << tolower(hello);
    return 0;
}

char* tolower(char* src)
{
    int ascii;
    for (int n = 0; n <= strlen(src); n++)
    {
        ascii = int(src[n]);
        if (ascii >= 65 && ascii <= 90)
        {
            src[n] = char(ascii+32);
        }
    }

    return src;
}
