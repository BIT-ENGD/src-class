string temp = "1234567890123456";
string str;

cout << str.capacity() << endl;   

str.reserve(16);

// capacity is 31 on my computer    

str += temp;    
cout << str.capacity() << endl;    

str.reserve(16);    
cout << str.capacity() << endl;     
// can't release. The capacity is still 31.
