vector<string> split(const string& s, const string& w) // w is second argument
{
    vector<string> words;
    string altered;
    for(int i = 0; i < s.length(); ++i)
    {
        altered+=s[i];
        if(i == (s.length()-1)) words.push_back(altered);
        else if(s[i] == ' ')
        {
            words.push_back(altered);
            altered = "";
        }
    }

    return words;
}
