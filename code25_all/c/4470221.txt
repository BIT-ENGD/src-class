typedef struct{
    char name[30];
    int empno;
    int sal;
}empd, * empdPtr;

int main(){
    int x = 1;
    char y = 2;
    int z = 3;
    empd e;
    empdPtr ep = &e;

    printf("sizeof ep = %d e = %d \n",sizeof(ep),sizeof(e));
    printf("Address of e = %u, ep = %u x = %u y = %u z = %u\n",&e,&ep,&x,&y,&z);
    printf("Address of e.name = %u e.empno = %u e.sal = %u \n",&e.name,&e.empno,&e.sal);
    return 0;
}
