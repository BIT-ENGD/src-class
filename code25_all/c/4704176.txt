#include<stdio.h>

int main(int argc,  char *argv[])
{
freopen(argv[1], "r", stdin);

for(int i=1;i<=n;++i)
{
  if((i%3==0)&&(i%5==0))
    {
      printf("Hop\n");
    }

      printf("Hoppity\n");
    }
  else if(i%5==0)
    {

    }
}
return 0;
}
