#include <stdio.h>

int main (int argc, const char * argv[]) {

    int c, nl;

    nl = 0;

    while ((c = getchar()) != EOF) {

        if (c == '\n') {
            ++nl;   
        }



    }

    printf("new lines=> %d\n", nl);

    return 0;
}
