int
evaluateExpresion ( EXPRESSION * e)
{
__asm
{
mov eax,dword ptr [e] 
movsx ecx,byte ptr [eax] 
test ecx,ecx 
jne salto1
mov eax,dword ptr [e] 
mov eax,dword ptr [eax+4] 
jmp final
salto1:
mov esi,esp 
mov eax,dword ptr [e] 
mov ecx,dword ptr [eax+8] 
push ecx  
mov edx,dword ptr [e] 
movsx eax,byte ptr [edx] 
push eax  
mov ecx,dword ptr [e] 
mov edx,dword ptr [ecx+4] 
call edx  
add esp,8
final:
} 
}
