#include <stdio.h>
#define LENGTH 33

int main( int argc, char*argv[ ] )

{
 unsigned int number, base, remainder, x, i;
 char result[LENGTH];

 puts( "Enter a decimal value, and a desired base: " );
 scanf( " %u", &number );
 scanf( " %u", &base );

 x = number;

 for( i=0; i < LENGTH; i++ )
 {
  remainder = x % base;
  x = x/base;
  result[i] = remainder;
 }

 printf( "%u equals  %s  (base-%u) \n", number, result, base );

}
