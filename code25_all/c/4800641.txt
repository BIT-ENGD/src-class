#include <stdio.h>
#include <stdarg.h>

int add(int x, int y, ...) {
    va_list intargs;
    int temp = 0;

    va_start(intargs, y);
    int i;
    for (i = 0; i < 3; i++) { /* How can I loop through any number of args? */ 
        temp += va_arg(intargs, int);
    }
    va_end(intargs);

    return temp + x + y;
}

int main() {
    printf("The total is %d.\n", add(1, 2, 3, 4, 5));
    return 0;
}
