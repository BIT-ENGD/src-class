    #include <stdio.h>

    main()
    {
        int i;
        if (fork()) { /* must be the parent */
            for (i=0; i<1000; i++)
            printf("\t\t\tParent %d\n", i);
        }
        else { /* must be the child */
            for (i=0; i<1000; i++)
                printf("Child %d\n", i);
        }
    }
