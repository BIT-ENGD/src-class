#include <stdio.h>

int fibonacci(int number)
{
  int retval;
  if (0 == number){
    return 0;
  }
  if (1 == number){
    return 1;
  }
  return(fibonacci(number-1) + fibonacci(number-2));
}

int main()
{
  int number = 0;
  int fibvalue = 1;
  while (1){
    printf("please enter the number\n");
    scanf("%d", &number);
    fibvalue = fibonacci(number);
    printf("computed fibonacci value %d\n", fibvalue);
  }
  return 1;
}
