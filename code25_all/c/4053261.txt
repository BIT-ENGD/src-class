#include <stdio.h>

int main(void)
{
        int x=1,y=1;
        int sum=0;
        while(x<100) 
        {
                while(y<x) 
                {
                        if(x%y==0) {
                                sum=sum+y;
                                y++;
                        }
                }
                x++;
                sum=0;
                y=1;
        }
        return 0;
}
