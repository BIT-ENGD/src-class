int duplicate (struct card hand[])
{
    int i = 0, j = 0, duplicate = 0;

    for (i = 0; i < 5; i++)
    {
        for (j = i + 1; j < 5; j++)
        {
            if (hand[i].suit == hand[j].suit && hand[i].face == hand[j].face)
            {
                duplicate++;
            }
        }
    }
    return duplicate;
}
