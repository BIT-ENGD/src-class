int is_refreshing_needed (char * path)
{
    int refresh_content;
    struct stat file;
    stat(path, &file);
//File does not exist or Permission Denied or the file has not been modified
    if( errno == 2 || errno == 13 || file.st_atime > file.st_mtime ) {  
        refresh_content=0;
    }
    else { 
        refresh_content=1;  
    }
    return refresh_content; 
}
