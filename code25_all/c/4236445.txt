#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main(int argc, char *argv[]) {
   int c;
   int i;
   int limit;

   sscanf(argv[1], "%d", &limit);
   srand((unsigned int)time(NULL));

   for (i = 0; i < limit; i++) {
      c = rand();
      if (c != 9 && c != 10 && c <= 32 && c >= 127) {
         i--;
      }
      else {
         putchar(c);
      }
   }
   return 0;
}
