main()
{

   int  i,j,mul_count[16]={0},mul;

   int *ptr;

   ptr =&mul_count;

   for(i=1;i<=4;i++)    
   {
         for(j=1;j<=4;j++)

         mul = j*i;

         ptr = (ptr+(mul*4));

         mul_count[mul] = ++*ptr;
   }

   for(i=1;i<=16;i++)    
      printf("mul_count[%d]=%d\n",i,mul_count[i]);

}
