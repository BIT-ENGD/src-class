//CHILD
typedef struct Child{
    int id;
}Child;
Child* newChild(){
    Child *aChild = malloc(sizeof(Child));
    aChild->id = 0;
    return aChild;
}

//PARENT
typedef struct Parent{
    int id;
    Child **children;
}Parent;

Parent* newParent(){
    Parent *aParent = malloc(sizeof(Parent));
    aParent->id = 0;
    aParent->children = malloc(sizeof(Child*) * 5);//ARRAY OF 5 CHILDREN?
    for(i=0; i<5; i++){
        aParent->children[i] = newChild();
    }
    return aParent;
}
