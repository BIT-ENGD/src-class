$ cat size.c
#include<stdio.h>
#include<math.h>

int main() {

    printf ("max unsigned long long = %llu\n",
    (unsigned long long)(pow(2, 64) - 1));
}
$

$ gcc size.c -o size
$ ./size
max unsigned long long = 18446744073709551615
$
