#include <stdio.h>

double IncomeTAX(double i)
{
    double tax;

    if(i <= 120000)        
        tax = 0;    
    else        
        tax = (i - 120000) * 15/100;    

    return tax;
}


int main(void)
{    
    double a, b;

    printf  ("Enter the annual income: ");
    scanf   ("%lf", &b);

    a = IncomeTAX(b);

    printf ("Income Tax is %.2lf\n", a);    
}
