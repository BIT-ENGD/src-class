char* fun(char *s) {
  char buffer[50];
  int i=0;
  while(*s){
    if(isdigit(*s)){
      buffer[i++]=*s;
    }
    s++;
  }

  buffer[i]='\0';
  return buffer;
}

int main(){
  char *s="o34";
  char *p="off";

  p=fun(s);
  while(*p){
    printf("%c",p);
    p++;
  }
  //printf("%s",fun(&s[0]));
  //puts(fun(s));
  getchar();
}
