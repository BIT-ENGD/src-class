int main(int argc, char **argv) {
    FILE *file_pointer;
    file_pointer = fopen("text.txt","r");
    if(fseek(file_pointer, 0, -1)) {
        puts("An error occurred");
    }
    else {
        char buffer[100];
        fgets(buffer, 100, file_pointer);
        puts("The first line of the file is:");
        puts(buffer);
    }
    fclose(file_pointer);
    return 0;
}
