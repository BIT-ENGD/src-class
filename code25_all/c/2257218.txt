/* Demonstrates printer output. */

#include <stdio.h>

main()
{ 
 float f = 2.0134;

 fprintf(stdprn, "This message is printed.\n\n");
 fprintf(stdprn, "And now some numbers:\n\n");
 fprintf(stdprn, "The square of %f is %f.", f, f*f);

 /* Send a form feed */

 fprintf(stdprn, "\f");
}

FILE *printer;
printer = fopen("PRN", "w");
