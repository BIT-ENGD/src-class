  FILE *file = fopen("randwords", "r");
  if(file)
  {
    char line[64];
    while(fgets(line, sizeof line, file))
    {
      ch = getch();
      clear();
      if(ch == 'q') break;
      if(ch == 'z') s = 1;
      move(LINES/2, (COLS - 20)/2);
      for(i=0; i < strlen(line); i++)
      {
        if(!s && line[i] >= '0' && line[i] <= '9') break;
        addch(line[i]);
      }
      s = 0;
    }
    fclose(file);
    system("rm randwords");
  }
