#include <stdio.h>

int STRING_SIZE=4;
int getString(char *);

int getString(char *str)
{
   printf("\nEnter a string of length < %d: ", STRING_SIZE);
   fgets(str, STRING_SIZE, stdin);
   fflush(stdin);
   printf("\n\n%s\n\n",str);
   return 0;
}


int main(void)
{

   char str1[1024];
   char str2[1024];

   getString(str1);
   getString(str2);

   fprintf(stdout, "%s\n", str1);
   fprintf(stdout, "%s\n", str2);

   return 0;
}
