typedef struct _s
{
  int *r;

}s;

void main()
{
   s *u;

   u=malloc(sizeof(s)*8);
   u->r=malloc(sizeof(int)*8);
   u->r[5]=y;

   printf("%d\n",u->r[5]);
   getch();
}
