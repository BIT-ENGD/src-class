    void read_vector(int* v, int n)
    {
        int value, i = 0, j = 0, k;
        char num, str[9];

        do 
        {
            num = getchar();

            if (num > 47 && num < 58) 
            {
                while (i < 10) 
                {    
                    str[i] = num;
                    i++;
                    num = getchar();

                   if (i = 9 || num < 47 || num > 58) 
                   {
                       str[i] = '\0'
                       j++;

                       for(k = 1; k <= strlen(str); k++) 
                       {
                           value += str[k - 1] * pow(10, (strlen(str) - k));
                       }

                       v[j] = value

                       if(j = n)
                           return;
                   }
               }
           }
       } 

       while (1);
   }
