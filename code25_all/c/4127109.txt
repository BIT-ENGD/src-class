#include  <stdio.h>
#include  <stdlib.h>
#include  <errno.h>
#include  <unistd.h>
#include  <sys/types.h>
#include  <sys/wait.h>
#include  <string.h>
int main(void)
{
    pid_t pid;
    char ch;
    switch(pid = fork()) {
    case -1:
         perror("fork"); /* something went wrong */
         exit(1);         /* parent exits */
    case 0:
     FILE *fpaste,*fcopy;
     fpaste=fopen(xyz,"w");
     fcopy=fopen(abc,"r");
     while(1)  
         {  
            ch = getc(fcopy);  
            if(ch==EOF)  
        {  
            break;  
        }  
        else  
             putc(ch,fpaste);  
     }  
     fclose(fcopy); 
     fclose(fpaste);
     }
     exit(0);
   default:
         printf("PARENT: I'm outta here!\n");
    }
    return 0;
}
