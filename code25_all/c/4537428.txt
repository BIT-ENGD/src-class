const char* get_value(void *obj)
{
    /* Cast to the correct structure type */    
    if(obj == origin) {
        /* Is a origin structure */
        origin_t *origin = (origin_t*)obj;
    } 
    else if(obj == session) {
        /* Is a session structure */
        session_t *session = (session_t*)obj;
    }
}
