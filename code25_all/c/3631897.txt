#include "rs232.h"
#include <time.h>
#include <stdio.h>


void main() {
    int res = OpenComport(16, 9600);
    printf("Open port result: %d\n\n\n", res);
    unsigned char *buf;

    while(1) {
        int n = PollComport(16, buf, 255);
        printf("n: %d\n", n);
        sleep(1);   
    }

    CloseComport(16);
}
