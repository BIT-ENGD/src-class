void printMonth (int* startDay, int* days)
{
    int weekDay;

    printf("Sun Mon Tue Wed Thu Fri Sat\n");
    printf("--- --- --- --- --- --- ---\n");

    for (weekDay = 0; weekDay < *startDay; weekDay++)
        printf("    ");

    for (int dayCount = 1; dayCount <= *days; dayCount++)
    {
        if (weekDay > 6)
        {
            printf("\n");
            weekDay = 1;
        }
        else
            weekDay++;
        printf("%3d ", dayCount);
    }

printf("\n--- --- --- --- --- --- ---\n");

return;
}
