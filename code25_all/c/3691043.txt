char* fstrstr(char *s1,char *s2)
{
    int i=0,flag=0;
    char *s4,*s3;
//  s4 for retaining the value of s2
    s4 = s2;
    while(*s1 != '\0' && *s2 != '\0')
    {
        if(*s1 == *s2)
        {
            *(s3+i) = *s1;
            s2++;
            s1++;
            i++;
            flag = 1;
        }
        else
        {
            i = 0;
            s1++;
//          Initialize s2 again from its address
            s2 = s4;
            flag = 0;
        }
    }
    if(flag == 1)
    {
        while(*s1 != '\0')
        {
            *(s3+i) = *s1;
            i++;
            s1++;
        }
        *(s3+i) = '\0';
    }
    if(flag == 1)
        return (s3);

    if(flag==0)
    {
        *s3 = NULL;
        return (s3);
    }
}
