#include <stdio.h>

#define WORD "jumble"
#define JUMBLED "mleujb"

int main()
{
    char string[6];
    int i = 0;
    printf("The jumbled word is ");
    printf(JUMBLED);
    printf("\nCan you guess the original: ");
    while(i == 0)
    {
        scanf("%d", string);
        if (string == "exit")
        {
            return;
        }
        if(string == WORD)
        {
            i++;
            printf("Kudos! You've guessed the word!");
        }
        else
        {
            printf("English please, good sir. Guess again.\n");
        }
    }
}
