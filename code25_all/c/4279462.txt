#include <stdio.h>

int main(){
    FILE* speaker = fopen("/dev/audio", "wb");
    FILE* music = fopen("bugsbunny2.wav", "rb");
    char data[16];
    while(!feof(music)){
        fread(data, sizeof(char), sizeof(data), music);
        fwrite(data, sizeof(char), sizeof(data), speaker);
    }
    return 0;
}
