#include <stdio.h>

struct card 
{
    unsigned int value;
    unsigned int suit;
};


struct board
{
    struct card deck[52];
};


int main()
{
    struct board b;
    b.deck[0].value = 3;
    b.deck[0].suit = 4;

    printf("Card Value: %d Suit: %d \n", b.deck[0].value, b.deck[0].suit);

    return 0;
}
