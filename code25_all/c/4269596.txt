#include <stdio.h>
#include <sys/types.h>
#include <unistd.h>
#include <sys/wait.h>
#include <stdlib.h>
int main()
{
    int array[3][3];
    int sumoro = 0,sumort = 0,sumorth = 0;

    pid_t pid = fork();

if (pid < 0) {
    printf("fork faild");
    exit(1);
}
else {

    if (pid == 0)
    {
        for (int i = 0; i < 3 ; i++) {
            for (int j = 0; j < 3; j++) {
                array[i][j] = rand()%9;
                if (i == 0)
                sumoro += array[0][j];
                if (i == 1) 
                sumort += array[1][j];
                if (i == 2)
                sumorth += array[2][j];
            }
        }
    }
    else {
        waitpid(pid, NULL, 0);
        int sumofall = sumoro + sumort + sumorth;
        printf("sum of all equal : %d ", sumofall);
    }


return 0;

    }
}
