bool EnterName(struct SymTab *ATable,
              const char *Name,
              struct SymEntry * *AnEntry)
{
              char name = *Name;
              unsigned hashval = hash (&name);
              struct SymEntry *ptr;

          ptr = ATable->Contents[hashval];

          while(ptr != NULL)
          {
                    if(strcmp(ptr->Name, &name)) 
                    {
                                 AnEntry = &ptr;
                                 return true;
                    } 
          }

          ptr = malloc(sizeof(struct SymEntry));
          ptr->Name = &name;
          AnEntry = &ptr;
          return false;
