int config_setting_lookup_string(const config_setting_t *setting,
                                 const char *name, const char **value)
{

  if(! member)
    return(CONFIG_FALSE);

  if(config_setting_type(member) != CONFIG_TYPE_STRING)
    return(CONFIG_FALSE);

  const char *string = config_setting_get_string(member);
