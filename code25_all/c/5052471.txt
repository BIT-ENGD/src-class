typedef struct leaf_node_t {
    int type;
    int key;
    int *data_ptr[2]; //This will point to real data.
} leaf_node_t;

typedef struct internal_node_t {
    int type;
    int key;
    typedef union{
        struct internal_node_t* iptrs[2];
        leaf_node_t* lptrs[2];
    } node_ptr;
} internal_node_t;
