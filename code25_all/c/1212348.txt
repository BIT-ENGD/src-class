#define _POSIX_SOURCE
#include <pwd.h>
#include <sys/types.h>
#include <sys/wait.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sched.h>
#include <unistd.h>
#include <string>
#include <time.h>

using namespace std;


int main(int argc, char *argv[])
{
    int uid;
    struct passwd *p;

    if ((p = getpwuid(uid = getuid())) == NULL){
            perror("getpwuid() error");
            exit(1);
    }

    printf("***************************************\n");
    printf("Executing user: %s (%d)\n", p->pw_name, p->pw_uid);
    printf("***************************************\n");

    seteuid(1000);

    if ((p = getpwuid(uid = getuid())) == NULL){
            perror("getpwuid() error");
            exit(1);
    }

    printf("***************************************\n");
    printf("Executing user: %s (%d)\n", p->pw_name, p->pw_uid);
    printf("***************************************\n");

    return 0;
}
