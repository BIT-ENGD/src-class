{
    int n_args = 0, i = 0;

        if (isspace(line[i++]))
            n_args++;
    }

    for (i = 0; i < n_args+1; i++)
        command = (char**) malloc (n_args * sizeof(char*));

    i = 0;
    line = strtok(line," \n");
    while (line != NULL)
    {
        command[i++] = (char *) malloc ( (strlen(line)+1) * sizeof(char) );
        strcpy(command[i++], line);
        line = strtok(NULL, " \n");
    }
    command[i] = NULL;
}
