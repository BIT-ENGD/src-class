#include <stdio.h>
#define MAXDIG   50
void main() {
   FILE     *f;
   int      j, i = 0;
   char   dig[MAXDIG];
   if ( !(f = fopen("H:\1\first", "r")) ) {
      printf("error opening file\n");
   }
   while (!feof(f)) {
      fscanf(f, "%c\n", &dig[i++]);
      if (i >= MAXDIG)
         break;
   }
   for (j=0; j < i; j++)
      printf("%c\n", dig[j]);
}
