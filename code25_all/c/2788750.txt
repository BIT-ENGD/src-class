#include <stdio.h>

__declspec(dllexport) void
print_to_stream (FILE *stream)
{
  fprintf (stream, "OK!\n");
}

#include <stdio.h>

__declspec(dllimport) void print_to_stream (FILE *stream);

int
main (void)
{
  print_to_stream (stdout);
  return 0;
}

cl /LD file-dll.c

cl file-test.c file-dll.lib

cl /LD /MD file-dll.c
cl /MD file-test.c file-dll.lib
