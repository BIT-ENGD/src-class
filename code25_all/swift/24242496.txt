func traverseCollection<T : Sequence>(collection: T)
{
    var generator = collection.generate()
    while let item = generator.next()
    {
        if let collectionItem = item as? Collection
        {
            traverseCollection(collectionItem)
        }
        else
        {
            println("item:", item)
        }
    }
}

       }
}
