class FQ {

    var rating: Float?
    var phone: String?

    init(rating: Float?, phone: String?)
    {
        self.rating = rating
        self.phone = phone
    }
}

FQ.match() -> FQ?
let fq: FQ? = FQ.match(self.place!.name, location: self.place!.location)

if fq != nil && fq!.phone != nil {
    self.phone.text = fq!.phone
} else {
    self.phone.text = ""
}
