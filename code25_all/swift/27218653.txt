import UIKit

let wv = UIWebView(frame: CGRectMake(0, 0, 255, 255))
let url = NSURL(string: "http://www.google.com")
let req = NSURLRequest(URL: url!)
wv.loadRequest(req)
wv // This is for previewing in the Playground
