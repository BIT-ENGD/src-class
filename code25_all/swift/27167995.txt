class SomeClass {
    var optionalVar: String?

    func checkVar() {
        println(optionalVar!)
    }

    func createVar() -> String {
        if let tempVar = optionalVar {
            return tempVar
        } else {
            println("create optional")
            return "123"
        }
    }
}

let instance = SomeClass()
instance.optionalVar = instance.createVar()
instance.checkVar()
