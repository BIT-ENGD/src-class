@IBAction func saveButton(sender: AnyObject) {

    //create sound
    var sound = Sound()
    sound.name = self.soundObjectName.text


    //add sound to tableView
    self.previousViewController.sounds.append(sound)


    //dismiss viewController

    self.previousViewController.modalTransitionStyle = UIModalTransitionStyleCrossDissolve


    self.dismissViewControllerAnimated(false, completion: nil)


}
