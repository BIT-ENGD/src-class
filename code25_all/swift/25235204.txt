// User.swift
class User: Equatable {
    let id: Int

    init(_ id: Int) {
        self.id = id
    }
}

func == (lhs: User, rhs: User) -> Bool {
    return (lhs.id == rhs.id)
}

// ArrayExtension.swift
extension Array {

    public func contains(object: T) -> Bool {
        for x in self {
            // Compile error here "T is not convertible to User"
            if (x == object) {
                return true
            }
        }

        return false
    }
}
