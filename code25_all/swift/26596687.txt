import Foundation

protocol Prototype {
    func Clone<T>() -> T
}

class myColor: Prototype {

    var red: Int?
    var green: Int?
    var blue: Int?

    init () {}

    func Clone<myColor>() -> myColor {

        let newColor = myColor()

        newColor.red = self.red
        newColor.green = self.green
        newColor.blue = self.blue


        return newColor
    }
}

let newColor = myColor()
