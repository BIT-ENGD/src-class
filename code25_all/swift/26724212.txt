func addNodeToScene(x: CGPoint,y: CGPoint)
    {
        let node = SKSpriteNode(color: self.colorOfSpriteNode, size: CGSize(width: 50, height: 50))

        node = String(format: "node%d", self.nodeIdentifier)
        ++self.nodeIdentifier

        node.position = CGPoint(x: x, y: y)

        addChild(node)
    }
