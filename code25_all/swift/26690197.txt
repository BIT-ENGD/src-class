class Array2D<T> {

    let rows: Int
    let columns: Int

    var array: Array<T?>

    init(rows: Int, columns: Int) {
        self.rows = rows
        self.columns = columns

        array = Array<T?>(count:rows * columns, repeatedValue: nil)
    }

    subscript(column: Int, row: Int) -> T? {
        get {
            return array[(row * columns) + column]
        }
        set {
            array[(row * columns) + column] = newValue
        }
    }
}
