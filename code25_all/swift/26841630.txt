if var threads = self.threadsInfo?.threads? {
            switch self.threadsSortType
            {
            case .Score:
                threads.sort {
                    $0.score > $1.score
                }
            case .Views:
                threads.sort {
                    $0.views > $1.views
                }
            case .PostsCount:
                threads.sort {
                    $0.postsCount > $1.postsCount
                }
            default:
                assert(false, "Unknown threads sort type")
            }
        }
