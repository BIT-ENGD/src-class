class Array2D<T> {
let columns: Int
let rows: Int
let array: [T?] 

init(columns: Int, rows: Int) {
    self.columns = columns
    self.rows = rows
    array = [T?](count: rows*columns, repeatedValue: nil)
}

subscript(column: Int, row: Int) -> T? {
    get {
        return array[row*columns + column]
    }
    set {
        array[row*columns + column] = newValue  // Error here
    }
}}
