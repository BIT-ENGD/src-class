    let Left = CGPoint(x: -1, y: 0)
    let Top = CGPoint(x: 0, y: -1)
    let Right = CGPoint(x: 1, y: 0)
    let Down = CGPoint(x: 0, y: 1)

    let TopLeft = CGPoint(x: -1, y: -1)
    let TopRight = CGPoint(x: 1, y: -1)
    let DownLeft = CGPoint(x: -1, y: 1)

    let None = CGPointZero
}
