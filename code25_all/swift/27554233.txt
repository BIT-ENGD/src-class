class Point {
var x: Int = 0
var y: Int = 0

init(){
    x = 0
    y = 0
}

init(x: Int, y: Int) {
    self.x = x
    self.y = y
  }
}

class Line: Point {
var pointA: Point
var pointB: Point

init(p1x: Int, p1y: Int, p2x: Int, p2y: Int) {
    self.pointA(p1x, p1y)
    self.pointB(p2x, p2y)
  }
}

var lineA = Line(p1x:10, p1y:20 , p2x:20 , p2y:40)
