func JSONStringify(jsonObj: AnyObject) -> String {
    var e: NSError?
    let jsonData = NSJSONSerialization.dataWithJSONObject(
        jsonObj,
        options: NSJSONWritingOptions(0),
        error: &e)
    if e != nil {
        return ""
    } else {
        return NSString(data: jsonData, encoding: NSUTF8StringEncoding)
    }
}
