def showWell(well):
    print('The Well')
    for i in well:
        print(i)

def fillNumbers(matrix):
    for i in range(len(matrix)):
        for j in range(len(matrix[i])):
            matrix[i][j]=j+1
    return matrix

def makeWell(rows, collumns):
    i = 0
    well=[]
    while i<rows:
        well+=[collumns[:]]
        i+=1
    well=fillNumbers(well)
    return well
wellPlate=makeWell(r,wellC)

def checkForX(well):
    xs=0
    for i in range(len(well)):
