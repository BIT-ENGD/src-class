override func tableView(tableView: UITableView, heightForRowAtIndexPath indexPath: NSIndexPath) -> CGFloat{
        let label:UILabel = UILabel(frame: CGRectMake(0, 0, tableView.frame.width-20, CGFloat.max))
        label.numberOfLines = 0
        label.lineBreakMode = NSLineBreakMode.ByWordWrapping
        label.font = dataTextLabel.font
        label.text = dataTextLabel.text

        label.sizeToFit()
        return label.frame.height+28
}

@IBOutlet var dataTextLabel: UILabel!
