func mapView(mapView: GMSMapView!, didTapMarker marker: GMSMarker!) -> Bool {
    if marker.title == nil{
        makeApiRequest(["id": marker.userData],
            success: {
                (response: Dictionary<String, JSONValue>?) in
                if let name = response?["name"]?.string{
                    marker.title = name
                }
            },
            failure: {
                (error: NSError) in
        })
    }
}
