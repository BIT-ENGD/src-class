var tick:(() -> ())?
var tickLengthMillis = TickLengthLevelOne
var lastTick:NSDate?

override func update(currentTime: CFTimeInterval) {
    if lastTick == nil {
        return
    }
    var timePassed = lastTick!.timeIntervalSinceNow * -1000.0
    if timePassed > tickLengthMillis {
        lastTick = NSDate.date()
        tick?()
    }
}
