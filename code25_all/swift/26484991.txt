public class Var<T> {
    let _get : Void -> T
    init(_ f : Void -> T) {
        _get = f
    }
    public var value : T {
        get {
            return _get()
        }
    }
}

public class Value<T> : Var<T> {
    var _val : T
    public init(_ t: T) {
        _val = t
        super.init({ [unowned self] in self._val })  // <--- Error
    }
}
