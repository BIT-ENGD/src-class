for i in 1...100 {
    switch (i) {
    case _ where (i % 3 == 0) && (i % 5 == 0):
        println("FizzBuzz")
    case _ where i % 3 == 0:
        println("Fizz")
    case _ where i % 5 == 0:
        println("Buzz")
    default:
        println(i)
    }
}
