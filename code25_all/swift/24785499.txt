func exchange<T>(inout data:[T], i:Int, j:Int)
{
    let temp = data[i];
    data[i] = data[j];
    data[j] = temp;
}

func swapLeft<T: Comparable>(data: [T], index:Int)
{
    for i in reverse(1...index)
    {
        if (data[i] < data[i - 1])
        {
            exchange(&data, i, i - 1);
        }
        else
        {
            break;
        }
    }
}
