import Cocoa

var arr: [[Int]] = [[1,2], [3,4], [5,6]]

func flatten<S: SequenceType, SS: SequenceType, E where S.Generator.Element == SS, SS.Generator.Element == E>(seq: S) -> [E] {
    var result = [E]()

    for subseq: SS in seq {
        for elem: E in subseq {
            result += [elem]
        }
    }

    return result
}

var flattened = flatten(arr)
println(flattened)
