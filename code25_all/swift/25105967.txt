@objc class Palette {
  var palette: [CGColor]

  init() {
    palette = [CGColor]()
    buildPalette()
  }

  func buildPalette() {

    let rgb = CGColorSpaceCreateDeviceRGB()

    // omitting the loop for simplicity
    let color = CGColorCreate(rgb, [1.0, 1.0, 1.0, 1.0])
    palette.append(color)   // crashes here
  }
}
