class Client
{
    var name:String
    var account:Account!

    init(name:String)
    {
        self.name = name
        self.account = Account(client: self)
    }

    deinit
    {
        println("Client::deinit")
    }
}

class Account
{
    var client:Client
    var balance:Int

    init(client:Client)
    {
        self.client = client
        self.balance = 0
    }

    deinit
    {
        println("Account::deinit")
    }
}

var client:Client! = Client(name: "larryhou")
client = nil
