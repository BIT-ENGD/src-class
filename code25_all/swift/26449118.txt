dispatch_source_set_timer(timerDispatchObj, macTime, 10, 1)


dispatch_source_set_event_handler(timerDispatchObj, {

print("hi siva")
})


if((timerDispatchObj) != nil){

dispatch_resume(timerDispatchObj)

}
