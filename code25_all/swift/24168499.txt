func convertDataToJSON(data: NSData) -> NSDictionary {
    var error: NSError?
    var jsonDictionary: NSDictionary = NSJSONSerialization.JSONObjectWithData(data, options: .MutableContainers, error: &error) as NSDictionary

    return jsonDictionary
}
