func initializeWeights(w : [Int64], N : Int64)
    {
       for var i=0; i < Int(N)+1; i++
       {
          w[i] = Int64(((i + 1) % 4) * 10);
       }
    }

func trainWeights(target : Int64, net : Int64, eta : Float, w : [Int64], x : [Int64], N : Int64)
{
    for var i=0; i < Int(N+1); i++
    {
        var deltaW = eta * Float(x[i] * (target - net));
        w[i] = w[i] + deltaW;
    }
}
