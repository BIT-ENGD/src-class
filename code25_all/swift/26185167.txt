func textField(
  tf: UITextField,
  shouldChangeCharactersInRange range: NSRange,
  replacementString str: String
) -> Bool
{
  var tgt = String( tf.text )
  tgt.replaceRange( range.toRange(), with: str )   // PROBLEM HERE
  NSLog( "realtime: %@", tgt )
