// Dog model
class Dog: RLMObject {
    var name = ""
    var owner = Person()
}

// Person model
class Person: RLMObject {
    name = ""
    birthdate = NSDate(timeIntervalSince1970: 1)
    dogs = RLMArray(objectClassName: Dog.className())
}

Person.objectsWhere("ANY dogs.name name BEGINSWITH 'B'")
