import Foundation

class Thing : Printable {
    var name:String
    var description:String { return name }

    init(name:String) {
        self.name = name
    }
}

extension Thing {
    var nameWithArticle:String {
    return "a" + name
    }
}

let bigfoot = Thing(name: "BigFoot")
println(bigfoot.nameWithArticle)
