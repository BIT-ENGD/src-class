class Vector{
    var x:Int
    var y:Int
    init(x _x:Int, y _y:Int){
        self.x = _x
        self.y = _y
    }
}

func *(lhs:Vector, rhs:Vector)->Int{
    return lhs.x * rhs.y + rhs.x + rhs.y
}

func +(lhs:Vector, rhs:Vector)->Vector{
    return Vector(x: lhs.x + rhs.x, y: lhs.y + rhs.y)
}

var v1 = Vector(x: 6, y: 1)
var v2 = Vector(x: 3, y: 1)
