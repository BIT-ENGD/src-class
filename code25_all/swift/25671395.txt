// extension method on the String class
func findIndexOf(char : String) -> Int? {

    for var i = 0; i < countElements(self); ++i {
        var ch = String(Array(self)[i])
        if String(ch) == char {
            return i
        }
    }

    return nil
}

// extension method on the String class
func findIndexOf(char : String) -> Int? {

    for (index, ch) in enumerate(self) {
        if String(ch) == char {
            return index
        }
    }

    return nil
}
