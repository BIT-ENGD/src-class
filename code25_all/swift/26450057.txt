extension Character {

    var hexValue : UInt {
        let zero : Character = "0"
        let nine : Character = "9"
        let a : Character = "a"
        let f : Character = "f"
        let A : Character = "A"
        let F : Character = "F"

        if self >= zero && self <= nine {
            return self - zero
        }
        if self >= a && self <= f {
            return self - a + 10
        }
        if self >= A && self <= F {
            return self - A + 10
        }

        return 0
    }
}
