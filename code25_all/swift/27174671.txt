class Parent {
    func breakMethod() {
        println("p break")
    }
    func breakMethod(#variable: Int) {
        println("p breakpoint " + String(variable))
    }
}

class Child: Parent {
    override func breakMethod() {
        println("c break")
    }
    override func breakMethod(#variable: Int) {
        println("c breakpoint " + String(variable))
    }
}

let p = Parent()
let c = Child()

p.breakMethod()
