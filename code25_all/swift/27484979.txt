extension Dictionary {
   func toDataJSON() -> NSData? {

       return JSON(self as Dictionary<String, AnyObject>)
                   .toString(pretty: false)
                   .dataUsingEncoding(NSUTF8StringEncoding, allowLossyConversion: false)
   }
}
