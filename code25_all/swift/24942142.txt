    var definitions: NSSet = NSSet()

    func addDefinitionsObject(value: AnyObject) {
        self.mutableSetValueForKey("definitions").addObject(value)
    }

    func removeDefinitionsObject(value: AnyObject) {
        // this method is never called

}
