var img = UIImage()
let userImageFile = object["image"] as PFFile

    (imageData: NSData!, error: NSError!) -> Void in
        if !error {
           println("Imaging!")
           img = UIImage(data:imageData)
           self.pic.image = img
        } else {
           print("Nooooo")
        }
}
