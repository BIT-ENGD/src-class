func checkPlayerDead() {
    if (self.Runner.position.y < 0) {
        let reveal: SKTransition = SKTransition.revealWithDirection(SKTransitionDirection.Down, duration: 0.5)
        let gameOverScene: SKScene = GameOverScene.sceneWithSize(self.frame.size)
        self.scene.view.presentScene(gameOverScene)
    }
}
