class Person {

    var name: String? {
    didSet {
        println("Person name was set.")
    }
    }

    init() {}
}

class Employee: Person {

    override var name: String? {
    didSet {
        println("Employee name was set.")
    }
    }

}

var person = Person()
person.name = "Bob"

var employee = Employee()
employee.name = "Sally"  // results in "Ambiguous use of 'name'" compile error
