class human {
    var name:String
    var height:Integer
    var hairColor:String
    var health:Integer

    init(name:String, height:Integer, hairColor:String) {
        self.name = name
        self.height = height
        self.hairColor = hairColor
        self.health = 100
    }

    func applyDamage(amount:Integer) -> Integer{
        self.health = self.health - amount
        return self.health
    }
}
