class MyClass {
    var name:String?

    required init() {
        println("init")
    }
}

extension MyClass {
    convenience init(name: String) {
        self.init()
        self.name = name
    }
}

protocol MyProtocol {
    init(name:String)
}

extension MyClass : MyProtocol {
    convenience init(name: String) {
        self.init()
        self.name = name
    }
}

required
final
