func finder<T:Equatable>(array:[T], valueToFind:T) -> Int? {
    for(index, value) in enumerate(array) {
        if(value == valueToFind) {
            return index
        }
    }
    return nil
}

var testArray = ["Dog", "Cat", "Mouse", "Tiger"]

finder(testArray, "Cat")

var testArray = [1, 2]

finder(testArray, 1)

var testArray = ["Dog", "Cat", "Mouse", "Tiger", 2]

finder(testArray, "Cat")
