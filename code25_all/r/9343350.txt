x = c(1,2,3,4,1,2,3,4)
f = c(rep('a',4),rep('b',4))
y = c(2,4,5,6,9,4,2,10)
z = data.frame(x=x,f=f,y=y)

x1 = c(2,2.5,3.1,2.8,3.4,3.5)
y1 = c(1,1.2,1,2,3,2)
f1 = c(rep('a',3),rep('b',3))
z1 = data.frame(x=x1,f=f1,y=y1)

p = ggplot() +
 layer(
 data=z,
 mapping = aes(x,y),
 geom="line"
) +
layer(
 data = z1,
 mapping = aes(x,y),
 geom="polygon",
 color="blue"
) +
facet_grid(f ~ .)
