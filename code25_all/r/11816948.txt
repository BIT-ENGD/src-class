x <- c(10,11,15,8,16,12,20)
y <- c(10,14,18,25,28,30,35)

d <- c(x,y)
f <- as.factor(c(rep("a",7), rep("b",7)))

# Unequal variance:

t.test(x,y)$p.value
t.test(d~f)$p.value

# Equal variance:

t.test(x,y, var.equal=TRUE)$p.value
t.test(d~f, var.equal=TRUE)$p.value

anova(lm(d~f))[[5]]
summary(aov(lm(d~f)))[[1]][5]
summary(lm(d~f))[[4]][8]
