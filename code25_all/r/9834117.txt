prob <- c(0.10, 0.20)
for (j in seq(prob)) {
  range <- c(2,3)
  for (i in seq(range)) {
    sample <- c(10,20)
    for (k in seq(sample)) {
      data <-  Simulation(X =1,Y =range[i], Z=sample[k] ,p = prob[j])
      filename <-  paste('file',i,'txt')
      if (j == 1) {
        write.table(data, "Desktop/file2.txt", col.names= TRUE) 
      } else {
        write.table(data,"Desktop/file2.txt", append = TRUE, col.names = FALSE)
      }
    }
  }
}
