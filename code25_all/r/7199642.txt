  cat(c("c=", c*1e6, "micro.mol/L\n",
          "m=", m*1e3, "mg\n",
          "M=", M, "g/mol\n",
          "V=", V*1e3, "mL\n"))
  ## mol/L, g, g/mol, L
 invisible(list(c=c, m=m, M=M, V=V))

}
