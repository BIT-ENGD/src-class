className <- class {
  # private variables
  var1 <- "standardvalue"

  var2 <- TRUE
  # Constructor
  constructor (v1, v2) {
    var1 <- v1
    var2 <- v2
  }

  # Method 1
  function sum() {
    var1 + var2
  }

  # Method 2
  function product() {
    var1 * var2
  }
}

# Create Object
numbers <- className(10,7)

# Call functions of the Object
numbers -> sum()      # Should give "17"
numbers -> product()  # Should give "70"
