set.seed(1); u = matrix(runif(1000), nrow=1000)
u[u > 0    & u <= 1/12]   <- 1
u[u > 1/12 & u <= 2/12]   <- 2
u[u > 2/12 & u <= 3/12]   <- 3
u[u > 3/12 & u <= 4/12]   <- 4
u[u > 4/12 & u <= 5/12]   <- 5
u[u > 5/12 & u <= 6/12]   <- 6
u[u > 6/12 & u <= 7/12]   <- 7
u[u > 7/12 & u <= 8/12]   <- 8
u[u > 8/12 & u <= 9/12]   <- 9
u[u > 9/12 & u <= 10/12]  <- 10
u[u > 10/12 & u <= 11/12] <- 11
u[u > 11/12 & u < 12/12] <- 12
table(u); u[1:10,]
