library(ggplot2)

# sample data
sampleData = data.frame( 
 v1=c('a','b','c','d','e', 'f','g', 'h', 'i','j'), 
 v2=c(1:10)     
)    
sampleData$Names = data.frame( Names = paste(sampleData$v1, sampleData$v2, sep="") )    
sampleData$Values = c(1:10)

# make plot
x = sampleData$Values
y = sampleData$Names 

qplot(      
    x, y, data = sampleData,
    geom="bar"  
)
