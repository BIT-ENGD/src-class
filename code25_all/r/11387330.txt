y = a*b*c*x/((1 - c*x)(1 - c*x + b*c*x))

0 < a < 100
0 < b < 100
0 < c < 100

f(x[2]) > 20
f(x[3]) < 40

best = function(p,x,y){sum(abs(y - p[1]*p[2]*p[3]*x/((1 - p[3]*x)*(1 - p[3]*x + p[2]*p[3]*x))))}
p = c(1,1,1)
x = c(.1,.5,.9)
y = c(5,26,35)
optim(p,best,x=x,y=y)

optim(p,best,x=x,y=y,method="L-BFGS-B",lower=c(0,0,0),upper=c(100,100,100))
