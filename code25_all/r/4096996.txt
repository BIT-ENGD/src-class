function (tag, key = lastkey, parse = TRUE) 
{
    params = list(method = "tag.gettopartists", tag = tag, api_key = lastkey)
    ret = getForm(baseurl, .params = params)
    doc = xmlParse(ret, asText = TRUE)
    if (parse) 
        doc = p.tag.gettopartists(doc)
    return(doc)
}
