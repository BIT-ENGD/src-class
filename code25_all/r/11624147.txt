FUN <- function(n) {
    plot.new()   
    mtext("wrong", side = 3, cex=1.5, adj = 0, padj = 1, col = "red") 
    wrong <- letters[1:n]
    text(0, .8, paste(wrong, collapse = "\n"), offset=.3, cex=1.5)
}

FUN(5)
FUN(10)
