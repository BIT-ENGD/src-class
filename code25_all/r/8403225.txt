dates <- chron(1:1825, origin=orig, out.format=c(dates="d/m/y", times="h:m"))

for (date in dates)
{
  s = chron(date, "00:00:00", origin=orig)
  e = chron(date, "23:59:59", origin=orig)

  aeronet_day = window(aeronet, start=s, end=e)
}
