foo<-data.frame(x=c(1:10), 
                y=c("red", "red", "red", "blue", "blue", 
                    "blue", "yellow", "yellow", "yellow", 
                    "green"),
                z=Sys.Date()+c(1:10))

foo$y<-as.character(foo$y)
foo$z<-as.numeric(foo$z)

foo<-convert.magic(foo, c(character, character, numeric))
