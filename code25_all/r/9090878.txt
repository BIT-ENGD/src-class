largestpalndrome3 <- function(){
products3 <- c()
i <- 100
while(i <= 999){
    j <- i
    while(j <= 999){
        products3[i] <- i*j
        j <- j+1
    }
    i <- i+1
}
palindromes <- c()
i <- 1
for(prod in products3){
    if(prod<100000){
        prodcopy <- prod
        o <- prodcopy %% 10
        prodcopy = prodcopy%/%10
        t <- prodcopy %% 10
        prodcopy = prodcopy%/%10
        h <- prodcopy %% 10
        prodcopy = prodcopy%/%10
        th <- prodcopy %% 10
        prodcopy = prodcopy%/%10
        tth <- prodcopy %% 10
        hth <- prodcopy%/%10
        if(o==hth & t==tth & h==th){
            palindromes[i] <- prod
        }
    }
    if(prod>100000){
        prodcopy <- prod
        o <- prodcopy %% 10
        prodcopy = prodcopy%/%10
        t <- prodcopy %% 10
        prodcopy = prodcopy%/%10
        h <- prodcopy %% 10
        prodcopy = prodcopy%/%10
        prodcopy = prodcopy%/%10
        tth <- prodcopy %% 10
        prodcopy = prodcopy%/%10
        hth <- prodcopy%%10
        m <- prodcopy%/%10
        if(o==m & t==hth & h==tth){
            palindromes[i] <- prod
        }
    }
    i <- i + 1
}
}
