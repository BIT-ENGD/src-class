# generate some data to plot
x <- 1:5
y1 <- rnorm(5)
y2 <- rnorm(5,20)

# set margins
par(mar=c(5,4,4,5)+.1)

# plot first x/y line
plot(x,y1,type="l",col="red")

#plot second x/y line
par(new=TRUE)
plot(x, y2,,type="l",col="blue",xaxt="n",yaxt="n",xlab="",ylab="")
axis(4)
mtext("y2",side=4,line=3)
