#Load packages
library(RODBC)
library(xlsReadWrite)

#Set working directory
setwd("C:/iDashboards/ivizgroup/excel")

#set "filename" to the path and name of Access DB to import
filename <- "C:/ZZZ DATA FILES/LGHTest2.accdb"

#Import data from access into "ch"
ch <- odbcConnectAccess2007(filename)

#Save table called "LGH Equipment Schedule" to "dat"
dat<-sqlFetch(ch, "LGH Equipment Schedule")

#Write "dat" to an Excel file
write.xls(dat, "LGH Equipment Schedule.xls")

#Close access database connection
odbcClose(ch)
