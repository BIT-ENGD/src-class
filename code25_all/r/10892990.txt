a <- function(x) {
  message("The input is ", deparse(substitute(x)))
}

a("foo")
# The input is "foo"

tmplist <- list(x1 = 1, x2=2)
a(tmplist)
# The input is tmplist

  a(y)
}

b("foo")
# The input is y

  message("The input is ", deparse(eval(substitute(substitute(x)), parent.frame())))

a1(tmplist)

  a1(y)
}
