cdpl<-function(df,dep,indep){
    attach(df)

    cdplot(dep~indep,xlab=deparse(substitute(indep)),ylab=deparse(substitute(dep)))
    g<-indep
    ec<-ecdf(indep)
    lines(knots(ec),as.numeric(names(table(ec(g)))),col="red",lw=3)
    detach(df)
    }
