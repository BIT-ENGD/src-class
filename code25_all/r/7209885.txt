expect_identical
expect_equal
library(R.oo)
library(testthat)

setConstructorS3( "MyClass" , function( )
{
    extend( Object() , "MyClass" ,
            .param1 = rnorm(10)
          )
} )

test1 = MyClass()
test2 = MyClass()
expect_identical(test1, test2) # error: Objects equal but not identical
expect_equal( test1, test2) # no error reported
