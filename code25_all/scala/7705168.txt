def add1: Int => Int => Int = a => b => a + b
def add2(a: Int)(b: Int) = a + b

scala> add1(1)(2)
res2: Int = 3

scala> add2(1)(2)
res3: Int = 3

scala> :t add1
Int => Int => Int

scala> :t add2
(a: Int)(b: Int)Int

add1
add2
scala> add1(1)
res4: Int => Int = <function1>

scala> add2(1)(_)
res5: Int => Int = <function1>
