class User(id: Long, name: String)

class DisplayIt {
  def display = {
    val users = List(new User(0,"John"), new User(1, "James"))
    "#name *" #> users.map(_.name) &
    "#id *" #> users.map(_.id.toString)
  }
}


//In the html:
<div class="lift:DisplayIt.display">
  <div class="one-user">
    User <span id="name"> has the id <span id="id">
  </div>
</div>
