class Functional {

  object Timer {
    def oncePerSecond(callback: () => Unit) {
      while (true) { callback(); Thread sleep 1000 }
    }
    def timeFlies() {
      println("time flies like an arrow...")
    }
    def run() {
      oncePerSecond(timeFlies)
    }
  }

  def main(args: Array[String]) {
    Timer.run();
  }
}
