  def referer(implicit request: Request[AnyContent]) =
    request.headers.get(REFERER).getOrElse(mainUrl)
  def locale(l: String) = Authenticated { user =>
     implicit request =>
     Redirect(referer).withCookies(Cookie(LANG, if (l == "de" || l == "en") l else "de"))
  }
