var base: Array[Boolean]
  def getMatrix(matrix: DenseMatrix[Double], pred: Boolean, m: Int, n: Int): DenseMatrix[Double] = {
    var a = DenseMatrix.zeros[Double](m, n)
    var scanPos: Int = 0
    var insertPos: Int = 0
    for (el <- base) {
      if (el == pred) {
        a(::, insertPos) := matrix(::, scanPos)
        insertPos += 1
      }
      scanPos += 1
    }
    return a
  }
