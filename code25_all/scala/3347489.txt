class FullLineIterator() extends LineIterator {   
    // Don't strip \r?\n
    override def getc() = iter.hasNext && {
        val ch = iter.next
        if (ch == '\n') { 
            sb append ch
            false
        } else if (ch == '\r') {            
            sb append ch

            if (iter.hasNext && iter.head == '\n') {
                iter.next
                sb append iter.head
            }

            false
        } else {
            sb append ch
            true
        }
    }
}
