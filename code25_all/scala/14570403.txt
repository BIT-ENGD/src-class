  def split[X](n: Int, xs: List[X]): (List[X], List[X]) = {
    if (n <= 0) (Nil, xs)
    else if (n >= xs.size) (xs, Nil)
    else {
      val grouped = xs.grouped(n).toList
      return (grouped(0), grouped(1))
    }
  }
