object Log {

  def get[T](implicit manifest : Manifest[T] ) = {
    LoggerFactory.getLogger( manifest.erasure.getName )
  }

  def getByName( name : String ) = {
    LoggerFactory.getLogger(name)
  }

}

object SimpleFuture {
  val log = Log.get[SimpleFuture[Throwable,Nothing]]
}

manifest.erasure
