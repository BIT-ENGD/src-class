scenario("Incorrect password") {
  given("user visits", the[AdminHomePage])
  then(the[SignInPage], "is displayed")      

  when("username", "admin", "and password", "wrongpassword", "are entered")
  then(the[SignInPage], "is displayed")      
  and("Error message is", "Sign in failed")
}
