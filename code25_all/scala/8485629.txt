import scala.actors.Actor


case class FooBar()


class ParentActor extends Actor {



def act {

  loop {

    react {

      case  f: FooBar =>

        println("Parent Foo")

      case _ =>

        println("Parent something")

    }

  }

}

}

class ChildActor extends ParentActor {

  override def act {

    loop {

      react {

        case i: Integer =>

          println("Child int")

        case default =>

          println("Child Base")
          super ! default


      }

    }


  }


}

super ! Message
