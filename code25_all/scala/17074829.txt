package graphs

class City(val x: Int, val y: Int) {
  def dist(other: City): Double = {
    val xdist  = x - other.x
    val ydist  = y - other.y
    floor(sqrt(xdist * xdist + ydist * ydist) + 0.5)
  }
}

object City {
//  def apply ( x: Int,  y: Int) = new City(x, y)
  def apply = new City(_, _)
}
