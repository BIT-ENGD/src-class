   val removeVowels: Boolean = false
   val dropFirstCharacter: Boolean = true

def normalize(string: String) = {
  var s = string

  if (makeLowerCase) {
    s = s.toLowerCase
  }

  if (removeVowels) {
    s = s.replaceAll("[aeiou]", "")
  }

  if (dropFirstCharacter) {
    s = s.drop(1)
  }

  s
}
