object ActorAction {

  def apply[A](actorRef: ActorRef, msg: AnyRef, timeout: Timeout = 5 seconds)(f: A => Result)(implicit m: Manifest[A]): Action[AnyContent] = {
    Action {
      AsyncResult {
        ask(actorRef, msg, timeout).mapTo[A].asPromise.map(f)
      }
    }
  }

}

  }
