def populateBoard(data:Array[Array[Char]]) {

    Board.resize(data(0).length, data.length)

    for(y <- 0 to data.length-1) {
        val row = data(y)
        for(x <- 0 to row.length-1) {
            Board.putObjectAt(x,y,GamePieceFactory.createInstance(row(x),x,y))
        }
    }

}
