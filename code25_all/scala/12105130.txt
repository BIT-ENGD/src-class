Map[Char,Int] = Map(h -> 1, e -> 1, o -> 1, l -> 2)

val str = "hello"
var counts = new scala.collection.mutable.HashMap[Char,Int]
for (i <- str) {
    if (counts.contains(i))
        counts.put(i, counts(i) + 1)
    else
        counts.put(i, 1)
}

> str.groupBy(_.toChar).map{ p => (p._1, p._2.length)}
