#!/usr/bin/env perl
use strict;
use warnings;

my $ifh;

my $ofh;

my $line;
my $recnum = 0;
my $ifn  = "factor data 1.txt";
my $ofn  = "try.txt";
open ($ifh, "<$ifn") || die "can't open $ifn";
open ($ofh, ">$ofn") or die "can't open $ofn";
my $a = "\/\/ ";

while ($line = <$ifh>)
{
chomp $line ;

    if ($line =~ m/$a\$/)
        {

         print "$ofh $line\n";

         $recnum++;
  }
         else
 {
        print "$ofh $recnum $line\n";
  }
 }
       close ($ifh);
       close ($ofh);
