use strict;
use warnings;
use Data::Dumper;

my $fh = \*DATA;


while(my $line = <$fh>) {



        chomp($line);
        $line =~ s/\s+//g;

        push(my @ArrLines, $line);

        my @SortedArr = sort @ArrLines;

        foreach my $val (@SortedArr) {

                print "$val\n";
        }
}
