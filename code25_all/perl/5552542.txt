#!/usr/local/bin/perl

 use warnings;
 use strict;

 print "Enter minutes \n";
 my $min;
 my $n;

 $min = <STDIN>; # number of minutes to time
 chomp $min;
 for ($n = $min; $n >= 1; $n--){ # counts down the number of minutes
 print "$n minutes remaining\n";
 sleep(60); # counts down 60 seconds
 }
 print "*********************\n";
 print "*********************\n";
 print "* Minutes elapsed = $min \n";
 print "*********************\n";
 print "*********************\n";
