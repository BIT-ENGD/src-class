sub parsear{

my %operations=(
'echo' => \&echo,
'status' => \&status,
'echo5' => \&echo5,
);

my $op= $_[0];
if ($operations{$op}){
    $operations{$op}->(@_);
    print "it exists\n";
}else{
    print "incorrect command.\n";
    }
}

}
