use strict;
my $v = "UUUUUCCNNCCCCNNNCUUUNNNNN";
printf("before: %s \n", $v);
if($v =~ m/([UN]+)$/)
{
  my $length = length($1);
  substr($v, (length($v) - $length), $length) = "T" x $length;
}
printf(" after: %s \n", $v);

> ./test
before: UUUUUCCNNCCCCNNNCUUUNNNNN 
 after: UUUUUCCNNCCCCNNNCTTTTTTTT
