sub starterThread{
  $SIG{'KILL'}=sub{threads->exit();};
  $log->info("Starter Thread started");
  while() {
    my $programName=$startQueue->dequeue();
    $log->debug("programName:$programName");
    qx/$programName/;
  }
}
