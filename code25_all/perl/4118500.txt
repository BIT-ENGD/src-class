sub ftp_update {
    print "Checkin\' $_[0] \n";
    $ftp->cwd($_[0]);
    chdir $_[0];
    my @ls = $ftp->ls;

    foreach (@ls) {         
        ftp_update($_) if $ftp->isdir($_);

        next unless (! -e $_);

        if ($ftp->isfile($_)) {
            print "Gettin\' $_\n";
            $ftp->get($_);
        }               
    }
    $ftp->cdup;
    chdir "..";
}
