sub patch_check {
  my $pline;
  my $sline;
  while (<SYSTEMINFO>) {
    chomp($_);
    $sline = $_;

    while (<PATCHLIST>) {
      chomp($_);
      $pline = $_;
      print "sline $sline pline $pline underscoreline $_ "; #troubleshooting

      print "$sline - $pline\n";
      if ($pline =~ /($sline)/) {
        #print " - match $pline -\n";
      }
    } #end while
  }
}
