[me@unixbox1:~/perltests]> cat testopensimple.pl
#!/usr/bin/perl

use strict;
use warnings;

# |<command> for writing
# <command>| for reading

testopen("| pwd");
testopen("pwd |");
testopen("| hostname");
testopen("| cd");
testopen("| sh");
testopen("| sleep 2");

sub testopen {
        my $command = shift;

        print "Exists: " . (-e $command ? "true" : "false") . "\n";
        print "testopen($command)\n";
        eval {
                open(my $fh, $command) or die "$! $@";
                my $data = join('', <$fh>);
                close($fh) or die "$! $@";
                print $data . "\n";
        };
        if ($@) {
                print $@ . "\n";
        }
}
