sub calc_usage {
    use Date::Manip::Date;

    my $date_from;
    my $date_to;
    my $delta;

    my $i;
    for $i (0 .. $#DATE_HOLDER) {
        $date_from = new Date::Manip::Date;
        $date_to = new Date::Manip::Date;

        $date_from->parse($DATE_HOLDER[$i][0]);
        $date_to->parse($DATE_HOLDER[$i][1]);

        $delta = $date_from->calc($date_to, "business");

    }

}
