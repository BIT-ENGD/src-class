#!/usr/local/bin/perl
use warnings;
use 5.012;

my %hash;
$hash{one} = 0;
$hash{two} = 2;
$hash{three} = 0;

print ": ";
my $aw = <>;
chomp $aw;

given ( $aw ) {
    when ( 'cat' ) { say '$aw eq cat' }
    when ( 'mouse' ) { say '$aw eq mouse' }
    when ( 'sheep' ) { say '$aw eq sheep' }
    when ( !( !$hash{$_} ) ) { say '$hash{$_} is true' } ###
    default { say 'something else' }
}
