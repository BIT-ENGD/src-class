#! /usr/bin/perl

use strict;
use warnings;
use Spreadsheet::ParseExcel::Simple;

my @data;
my $xls = Spreadsheet::ParseExcel::Simple->read('mylargefile.xls');
foreach my $sheet ($xls->sheets) {
     while ($sheet->has_data) {  
         @data = $sheet->next_row;
         print "@data \n";
     }
}
