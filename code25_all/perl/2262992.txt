#!/usr/bin/perl --
use strict;
use CGI;

print <<HTTP;
Status: 200 OK
Content-Type: text/html

HTTP

print <<HTML; 
<html>
    <head>
        <title>Parameters<title>
    <head>

    <body>
     <table border="1" cellpadding="5" cellspacing="1">
     <tr>

     <th>Name</th>
     <th>Value</th>

     </tr>

HTML

 my $query = new CGI;

 my($name, $value);

 foreach $name ( $query->param)
 {
    print "<tr>";
    print "<td>$name</td>";

    foreach $value($query->param($name))
    {
        print "<td>$value</td>";
        print "</tr>";
    }



}
