#!/usr/local/bin/perl

use warnings;    
use strict;

my $inputfile = "file1.txt"; 

open FH,$inputfile;          
my @results = <FH>;   
close FH;           

print "\n---------type n--------------\n" ;

foreach my $line (@results) { 
    if ($line =~ m/(^Mgn\d\.qna.*)/i)            
    {
        print "$1\n";  
    }    
}

print "\n---------type p--------------\n" ;

foreach my $line (@results) {
   if ($line =~ m/(^Mgp\d\.qpa.*)/i)     
       print "$1\n"; 
   }
}
