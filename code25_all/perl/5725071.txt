#!/usr/bin/perl -w
use MIME::Lite::TT;

# SendTo email id
my $email = 'tester@mydomain.org';

my %params;
$params{ips} = "1.2.3.4";

# create a new MIME Lite based email
my $msg = MIME::Lite::TT->new
(
Subject => "HTML email test",
From    => 'admin@mydomain.org',
To      => $email,
Type    => 'text/html',
Template => 'test.txt',
TmplParams => \%params 
);

$msg->send();
