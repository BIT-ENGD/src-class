#!/usr/bin/perl

use warnings;
use strict;

print "Enter day";
chomp (my $day = <STDIN>);
print "Enter month";
chomp (my $month = <STDIN>);
print "Enter year";
chomp (my $year =  <STDIN>);
print "Enter time in hour:minute format";
chomp (my $time = <STDIN>);
my @timeIs = split(/:/,$time);
my $hour = $timeIs[0];
my $min = $timeIs[1];
my $date = `date $month$day$hour$min$year`;
print $date;
