use strict;
use warnings; #Always use these!

my $world = 1;
for my $top (1 .. 100) {
    for my $left (1 .. 100) {
         print MYFILE "\#world$world \{
                background: url(/images/1.png) 0 0 no-repeat;
                float: left;
                width: 1%;
                height: 2%;
                position: absolute;
                top: $top\%;
                left: $left\%;
                z-index: -1;
                margin-top: -10px;
                margin-left: -10px;
            \}

            \#world$world:hover \{
                background-position: 0 -20px;
                cursor: pointer;
            \}"; 
        $world++;
    }
}
close (MYFILE);
