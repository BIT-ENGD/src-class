package XMSP::File;
use parent 'IO::File';

sub new {
    my ($self,@args) = @_;
    my $object = {};

    bless($object,$self);
    $object->SUPER::new(@args);

    return $object;
}

sub open {
    my ($self,@args) = @_;
    $self->SUPER::open(@args);
}

sub close {
    my ($self,@args) = @_;
    $self->SUPER::close(@args);
}
1;
