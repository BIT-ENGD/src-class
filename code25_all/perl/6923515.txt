 my $b = "SS ";
 my $_ = " ";
 while ($line = <$ifh>)
 {
 chomp 
 if ($line =~ m/$a/g); {
  $line = $_;

  push @data, $line;

 while ($line = <$ifh>) 
{
$line .= $_;

push @data, $line;

last if 
($line =~ m/$b/g);
}

}
push @data, $line; }


print @data;
