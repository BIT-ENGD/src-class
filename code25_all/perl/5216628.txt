$a=<>;$b=<>;
@mul=("$a","$b");

for($i=0;$i<scalar(@mul)-1;$i++) {
    $source=$mul[$i];
    print "\n\nComparision of source:  $mul[$i]\n";
    print "------------------------------------";
    for($j=$i+1;$j<scalar(@mul);$j++) {
        $sample=$mul[$j];
        print "\n$sample ";
        print "\n------\n";
        $t=mutate($source,$sample);
        print $t;
    }
}

sub mutate {
    my ($s1,$s2)=@_;
    $temp="";
    for($k=0;$k<length($s1);$k++) {
        $seq1=substr($s1,$k,1);
        $seq2=substr($s2,$k,1);
        if($seq1 ne $seq2) {
            $temp.="[$seq1($k)/$seq2($k)]";
        } 
    }
    return $temp;
}
