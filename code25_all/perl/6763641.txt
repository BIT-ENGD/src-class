#!/usr/bin/perl

use warnings;
use strict;
use Fcntl ':flock'; # Import LOCK_* constants

my $file = 't';
my $can_lockin = 1;

open(my $fh, '>', $file) or die "Could not open '$file' - $!";
flock($fh, LOCK_EX) or die "Could not lock '$file' - $!";

print $fh $can_lockin;
sleep 100000000;
