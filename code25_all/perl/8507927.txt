sub coerce {
    my $self = shift;

    my $coercion = $self->coercion;

    unless ($coercion) {
        require Moose;
        Moose->throw_error("Cannot coerce without a type coercion");
    }

    #return $_[0] if $self->check($_[0]);

    return $coercion->coerce(@_);
}
