######################################
sub GetItems
######################################

{
    my $classes = shift;
    my %items = ();
    foreach my $ref (@_)
    {
        foreach my $id (keys %$ref) 
        { 
            foreach my $cui (keys %{$ref->{$id}}) { $items{$cui} = 1 }
        }
    }

    my @items = sort { $a <=> $b } keys %items;

    open(VAL, "> $classes.items");
    for my $i (0 .. $#items)
    {
        print VAL "$items[$i]\n";
        $items{$items[$i]} = $i + 1;
    }
    close VAL;
    return \%items;
}
