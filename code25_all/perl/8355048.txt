use strict;
use Win32;
use Win32::GUI;

my $win=Win32::GUI::Window->new(
        -name => 'wino',
        -text => 'window',
        -left => 375,
        -top  => 400,
        -width =>380,
        -height =>260,
);
my $but=$win->AddButton(
      -text=>"Add Control",
      -onclick=>\&add_control,
);

 $win->Show();
 Win32::GUI::Dialog();

 sub add_control(){

   my $mchk=$win->AddCheckbox(
    -text=>"run_time_con",
    -pos=>[180,145],
   );

   $mchk->Show();

 }
