#!/usr/bin/perl

use strict;
use warnings;

open (TEMPTABLE,"temp_data") or die "Cannot open file\n";

my @temp_table_data=< TEMPTABLE > ;
chomp @temp_table_data;

my $total_sum;
for(my $i=0;$i<=$#temp_table_data;$i++)
{
print "$temp_table_data[$i]\n";
my @col=split('==',$temp_table_data[$i]);
for(my $m=0;$m<1;$m++)
{
$total_sum+=$col[$m];
}    
}
print "$total_sum\n";

OUTPUT:
1==0==2
5==3==2
7==1==0
13
