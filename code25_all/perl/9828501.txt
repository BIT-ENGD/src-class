my $source = "/KEYS.txt";
my $data= "/claims.dat";
my @array;
my $arraySize = scalar (@DESYarray);

open (DAT, $data) or die "Cannot open file!";
open (LOG, ">>/output.log");

open (TXT,$source);
while (my $searchValues = <TXT>) {
    push (@array, $searchValues);
}
close (TXT);


while (my $line = <DAT>) {      
for (my $x = 0; $x <= $arraySize; $x++) {
    if (my $line =~ /$array[$x]/) {
        print LOG $line;
    }
}
}

close (DAT);
close (LOG);
