#!/usr/bin/perl

use strict;
use warnings;
use File::Find;
use Data::Dumper;
use List::MoreUtils qw/ uniq /;

my $localdir = 'images/p/';
my @filefound;

find(
  sub {push @filefound, $File::Find::name if /.jpg$/ },
  $localdir
);

foreach (@filefound){
  my @result = split('/images/p/',@filefound);
  foreach (@result) { print "$_  \n";} 
}
