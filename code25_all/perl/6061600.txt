#!/usr/bin/perl 

use strict;
use warnings;
use English qw( -no_match_vars );

use Tk;
use Tk::Tree;

my $mw   = MainWindow->new();
my $tree = $mw->Tree(
    -width     => '25',
    -browsecmd => \sub {
        my ($numbers) = @ARG;
        print $numbers. "\n";
    }
);

foreach (qw(one two three four five six )) {
    $tree->add( $ARG, -text => $ARG ); #populates the tree
}

$tree->pack();

MainLoop();
