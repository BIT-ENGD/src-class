XML::Simple
#!/usr/bin/perl
use strict;
use warnings;
use XML::Simple;

sub writexmlout {
  my $hashref = {
        'Contracts' => {
           'ShortName' => '123231123'
         },
         'CallStates' => {
            'CallStatesSPR' => {
           'Name' => 'Acknowledged'
             }
         },
         'ContractElements' => {
             'ShortName' => 'test'
         },
         'Calls' => {
             'CHD' => {
                 'FirstName' => 'Ron',
                 'LastName' => 'Rich',
                 'Tel' => '(123)  456-4567',
                 'Sign' => 'B2B',
                 'Email' => 'ron.rich@mavenir.com'
             },
             'SPCallID' => '12',
             'Remarks' => 'Ticket successfully Ebonded',
             'CustCallID' => '6199999'
         }
  };

  # Create object
  my $xml = new XML::Simple( NoAttr => 1,
                             SuppressEmpty => 1,
                             XMLDecl => 1 ,
                             RootName=>'CallData');

  my $result = $xml->XMLout($hashref);
  print $result;
}

writexmlout();
