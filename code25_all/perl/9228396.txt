use DateTime; 
use DateTime::Format::Strptime;  
my $val = "20090103 12:00";  
my $format = new DateTime::Format::Strptime(                 
                    pattern => '%Y%m%d %H:%M',          
                                            time_zone => 'IST',                 );  
my $date = $format->parse_datetime($val);  
print $date->strftime("%Y%m%d %H:%M %Z")."\n";  
$date->set_time_zone('EST');  
print $date->strftime("%Y%m%d %H:%M %Z")."\n";
