    }

dlete(\@array);




sub dlete {
    my $array = shift;
    my $not_complete = 1;
    my $index;
    my $len = ((scalar @$array) - 2);
    while ($not_complete) {
        $not_complete = 0;
        foreach $index (0 .. $len) {
            if (@$array[$index] eq @$array[$index+1] && -M @$array[$index] > -M @$array[$index + 1]) {

                unlink "@$array[$index]";

                $not_complete = 1;
              }
       }
    }
}
