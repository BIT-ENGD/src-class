    #!/usr/bin/perl
use File::MultiTail;

$path="my/path";
$files="*.log";

sub first{
    $True=0;
    my $file=MultiTail->new(OutputPrefix => 'f',
    Files => ["$path"],
    RemoveDuplicate => $True,
    ScanForFiles => "0");

    while(defined(my $line=$file->read)) {
        print"$line\n";
        $file->update_attribute(Files => ["$path"]);
    }

}

&first();
