use Web::Query 'wq';
my $football_matches = wq($mech->content)
    ->find('tr.match')
    ->map(sub {
        my (undef, $e) = @_;
        return 'v' eq $e->find('td.score')->text
            ? [
                $e->attr('id'),
                map { $e->find("td.$_")->text }
                  (qw(tournament dateTime homeTeam score awayTeam prices))
            ]
            : ();
    });
