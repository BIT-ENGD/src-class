%params = (
  "help" => "",
  "no_inline" => "",
  "no_dupl" => ""
);

&paramError if (!GetOptions(
  "help" => \$params{"help"},
  "no-inline" => \$params{"no_inline"},
  "no-duplicates" => \$params{"no_dupl"},
));

> script.pl --no-inline # ok, valid parameter
> script.pl --no-inline blahblah # blahblah is not valid
> script.pl --no-i # not valid
