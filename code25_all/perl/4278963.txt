use 5.012;
use warnings;

my @list = '01' .. '95';
my $col = 9;

my $row = int( ( $#list + $col ) / $col );
my $d = @list % $col;
my @arr;

my $i = 0;
for ( 0 .. $col ) {
    $i = 1 if $d and $_ >= $d;
    $arr[$_] = [splice( @list, 0, $row - $i )];
}

for my $e ( 0 .. $row - 1 ) {
    my @list;
    for my $c ( 0 .. $col - 1 ) {
    next if $d and $e == $row-1 and $c >= $d; 
        push @list, $arr[$c][$e];
    }
    say "@list";
}
