 package Person;
 use strict;

 my($NAME, $AGE, $PEERS) = ( 0 .. 2 );

 sub new {
    my $self = [];
    $self->[$NAME] = undef;
    $self->[$AGE] = undef;
    $self->[$PEERS] = [];
    bless($self);
    return $self;
 }

 sub name {
    my $self = shift;
    if (@_) { $self->[$NAME] = shift }
    return $self->[$NAME];
 }

 sub age {
    my $self = shift;
    if (@_) { $self->[$AGE] = shift }
    return $self->[$AGE];
 }

 sub peers {
    my $self = shift;
    if (@_) { @{ $self->[$PEERS] } = @_ }
    return @{ $self->[$PEERS] };
 }

 1;
