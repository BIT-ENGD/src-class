#!/usr/bin/perl
use strict;
use warnings;


my $file = '/kismetLOGS/PcapDumpKismet/WEPMACS.txt';
open my $info, $file or die "Could not open $file: $!";

while( my $line = <$info>) 
  {
    print $line;
    system("/usr/local/bin/aircrack-ng", "-b $line", "*.pcap *.pcapdump");
    last if $. == 0;
  }

close $info;
exit;
