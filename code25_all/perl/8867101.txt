#input: filename (i.e: http://test.com/hi4324.gif)
sub getFileName
{
  my $input = shift;  
  (my $filename = $input)  =~ m/(\w+\.(jpg|png|gif|jpeg)$)/i;  
  return $filename;
}

#input: filename (i.e: hi4324.gif)
sub getExtension
{
  my $input = shift;  
  (my $extension = $input)  =~ m/\w+\.(jpg|png|gif|jpeg)$/i;  
  return $extension;
}
