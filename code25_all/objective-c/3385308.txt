+ (UIImage*)imageForTag
{
    static UIImage *imgTag;

    if(imgTag == nil)
    {
        NSString* imageName = [[NSBundle mainBundle]
                           pathForResource:@"tag" ofType:@"png"];
        imgTag = [[[UIImage alloc]
                           initWithContentsOfFile:imageName] autorelease];
    }
    return imgTag;
}

    }

    }
    else if(indexPath.row == 1)
