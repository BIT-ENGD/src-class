-(UIImageView)wrapImage:(NSString *)imagePath
{
  UIImage *image = [UIImage imageWithContentsOfFile:[[NSBundle mainBundle]
                                    pathForResource:imagePath 
                                             ofType:nil]];
  UIImageView *view = [[UIImageView alloc] initWithImage:image];
  view.hidden = YES;
  return view;
}
