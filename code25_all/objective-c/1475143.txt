@interface AppController : NSObject
{
    NSArray *myArray;
}

- (void)init
{
    [super init];
    self.myArray = [[NSArray alloc] init];
    return self;
}

- (void)init
{
    [super init];
    NSArray *myTempArray = [[NSArray alloc] init];
    self.myArray = myTempArray
    [myTempArray release];
    return self;
}
