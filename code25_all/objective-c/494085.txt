@interface Bar : NSObject {
    NSMutableArray *item;
}
@end

@interface Bar : NSObject {
    NSMutableArray *items;
}
@end

- (void)setValue:(id)value forUndefinedKey:(NSString *)key {
    if([key isEqualToString:@"item"]) {
        if(items == nil) {
            items = [[NSMutableArray alloc] init];
            [items retain];         
        }
        [items addObject:value];
    }
}
