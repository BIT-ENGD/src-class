- (NSString *)deviceType
{
    NSString *deviceName = @"iphone";
    if([[UIScreen mainScreen] respondsToSelector:@selector(scale)])
    {
        if (UI_USER_INTERFACE_IDIOM() == UIUserInterfaceIdiomPad){
            deviceName = @"ipad";
        }
        else {
            deviceName = @"iphone4";
        }
    }
    return deviceName;
}
