-viewWillAppear
-viewWillDisappear
message sent to deallocated instance
BOOL
- (void)viewWillAppear:(BOOL)animated {
    isOnScreen = YES;
    [super viewWillAppear:animated];
}

- (void)willWillDisappear:(BOOL)animated {
    isOnScreen = NO;
    [super viewWillAppear:animated];
}

if (isOnScreen) [self doSomething];
