-(IBAction)letsstartGame {

ViewController * sl = [[ViewController alloc] initWithNibName:@"ViewController" bundle:[NSBundle mainBundle]];  
self.viewLink = sl;
[sl release];

[mainMenu stop];
[mainMenu setCurrentTime:0.0];

[self presentModalViewController:viewLink animated:NO];

[viewLink release];
