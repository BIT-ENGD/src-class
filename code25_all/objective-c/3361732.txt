-(void)SpinAction
{
BOOL win = NO;
int numInRow = 1;
int lastValue = -1;
for(int i = 0;i < 5;i++)
{
    int newValue = random() % [self.column1 count];
    if(newValue == lastValue)
        numInRow++;
    else
        numInRow = 1;
    lastValue = newValue;
    [picker selectRow:newValue inComponent:i animated:YES];
    [picker reloadComponent:i];
    if(numInRow >= 3)
    {
        win = YES;
    }

}
if(win == YES)
{
    winLabel.text = @"!!!You Won!!!";
}
else
{
    winLabel.text = @"";
}
}
