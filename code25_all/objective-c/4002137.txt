#import <Cocoa/Cocoa.h>


@interface myClass : NSObject {
    int a;
    int b;
}

-(void) setvara:(int)x;
-(void) setvarb:(int)y;
-(int) add;

@end

#import "myClass.h"


@implementation myClass

-(void)setvara:(int)x{
    a=x;    
}

-(void)setvarb:(int)y{
    b=y;    
}

-(int)add{
    return a+b; 
}
@end

#import <Cocoa/Cocoa.h>
#import <stdio.h>
#import "myClass.m"

int main(int argc, const char* argv[])
{
    myClass* class = [[myClass alloc]init];

    [class setvara:5];
    [class setvarb:6];

    printf("The sum: %d", [class add]);

    [class release];

}
