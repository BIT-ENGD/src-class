 [self.view addSubview:drawImage];

 CGContextSelectFont(context, "Courier", 40,  kCGEncodingMacRoman);

 CGContextSetFillColor(context, white);

 drawImage.image = UIGraphicsGetImageFromCurrentImageContext();
