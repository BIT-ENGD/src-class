'True means no duplicates'
Public Shared Function checkDuplicate(ByVal list As ArrayList) As [Boolean]
    Dim [set] As New HashSet(Of Integer)
    For i As Integer = 0 To list.Count - 1
        Dim val As Boolean = [set].Add(list(i))
        If val = False Then
            Return val
        End If
    Next
    Return True
End Function
