@interface CoolClasss : NSObject
  +(void)staticMethod;
@end

@implementation CoolClass
  static int integer=0;
  +(void)staticMethod {
    NSLog(@"integer: %d", integer);
    integer++;
  }
@end

int main (int argc, const char * argv[]) {
  for (int i=0; i<10; i++) {
    [CoolClass staticMethod];
  }
  return 0;
}
