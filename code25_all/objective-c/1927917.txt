- (IBAction) btnClickMe_Clicked:(id)sender {

    NSURL *url = [NSURL URLWithString:@"http://www.xxx.com/index.php?keyword=",kw];
    NSURLRequest *request = [NSURLRequest requestWithURL:url];
    [iMessageLabel loadRequest:request];

}
