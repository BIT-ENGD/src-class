string list = "Fred,Sam,Mike,Sarah";

string newList = "'Fred','Sam','Mike','Sarah'";

int number = 0;
string newList = "";
foreach (string item in list.Split(new char[] {','}))
{
    if (number > 0)
    {
        newList = newList + "," + "'" + item + "'";
    }
    else
    {
        newList = "'" + item + "'";
    }
    number++;
}
