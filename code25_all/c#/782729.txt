class Foo
{
    public Foo()
    {
        Size = true;
    }

    private bool _size;

    protected bool Size
    {
        get { _size; }
        set { _size = value; }
    }
}

class CrazyFoo : Foo
{
    public void First()
    {
        if (!Size)
            return;
    }

    public void Second()
    {
        if (!Size)
            return;
    }
    public void Finished()
    {
        if (!Size)
            return;
    }
}
