public static byte[] GetSHA256Hash(this FileInfo file)
{
    using (FileStream fs = new FileStream(file.FullName, FileMode.Open))
    {
        using (SHA256 hasher = new SHA256Managed())
        {
            return hasher.ComputeHash(fs);
        }
    }
}
