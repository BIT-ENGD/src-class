           {%>
        <div class="Question">
            <div class="QuestionTitle">

                      case QuestionTypeEnum.PhoneNumber:%>
                <%=Html.TextBox("somename")%>
                <% break;
                      case QuestionTypeEnum.Email:%>
                <%=Html.TextBox("somename")%>
                <% break;
                      case QuestionTypeEnum.Date:%>

                <% break;
                      case QuestionTypeEnum.YesNo:%>
