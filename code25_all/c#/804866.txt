    public User ValidateUser(string Username, string Password)
    {
        DataContext db = new DataContext();

                var query = from u in db.Users where (u.Username == Username && u.Password == Password) select u;

                if (query.Count() != 0)
                {
                    return query.First();
                }
                else
                {
                    return new User { UserID = -1 };
                }
    }
