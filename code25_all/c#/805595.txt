string[] languageCodsList= service.LanguagesAvailable();

                   new Dictionary<string, string>(languageCodsList.Length);

{
    CultureInfo cul = new CultureInfo(cod);
    list.Add(cod, cul.DisplayName);
}
languageList.DataSource = list;
