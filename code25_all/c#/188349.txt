    {

        using (Graphics g = Graphics.FromImage(bufl))
        {
            g.FillRectangle(Brushes.Black, new Rectangle(0, 0, pf.Width, pf.Height));

            pf.CreateGraphics().DrawImageUnscaled(bufl, 0, 0);
