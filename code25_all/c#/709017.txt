    public enum Language
    {
        English = 1,
        French = 2,
        German = 3
    }

    public class Locale
    {
        public string EnglishValue { get; set; }
        public string FrenchValue { get; set; }
        public string GermanValue { get; set; }
    }

    public class Locale
    {
        public string StringValue<T> Where T : Language
    }
