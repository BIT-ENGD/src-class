    protected override void OnRender(DrawingContext dc)
    {
        base.OnRender(dc);

        DrawGrid(dc);
        DrawSelector(dc);
    }

    private void DrawSelector(DrawingContext dc)
    {
        if (Selecting)
        {
            dc.DrawRectangle(new SolidColorBrush(Color.FromArgb(75, 0, 0, 255)), new Pen(Brushes.Blue, 1.5), SelectionRect);
        }
    }
