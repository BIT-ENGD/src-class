private int Status
{
    get
    {
        try
        {
            return (int)ViewState[@"__Status"];
        }
        catch
        {
            return 0;
        }
    }
    set
    {
        ViewState[@"__Status"] = value;
    }
}

private int Status
{
    get
    {
        if (ViewState[@"__Status"] is int)
        {
            return (int)ViewState[@"__Status"];
        }
        else
        {
            return 0;
        }
    }
    set
    {
        ViewState[@"__Status"] = value;
    }
}
