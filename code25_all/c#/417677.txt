public static string tocap(string s)
{    
    if (s.Length == 1) return s.ToUpper();
    string s1;
    string s2;

    s1 = s.Substring(0, 1).ToUpper();
    s2 = s.Substring(1).ToLower();

    return s1+s2;
}
