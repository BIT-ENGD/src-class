 do
  if not (null  list) then
   if (head list) == element then
     ['A'..'Z'] !! (length ['A'..'Z'] - length (tail list ) -1)
   else
     lowerToUpperImpl element (tail list) litereMari litereMici
  else
    '0' --never to be reached
