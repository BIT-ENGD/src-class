proceed char word progress = let zprog = zip progress [0..] in
    foldl (\a x -> a ++ [fst x]) "" $ map (f char word) zprog where    
        f c w el = 
            if c == w !! (snd el) then
                (c, snd el)
            else
                el

proceed char = zipWith combine where
    combine x y
        | x == char = char
        | otherwise = y
