class DebugMonad m where
    pdebug :: String -> m ()

instance DebugMonad (ST s) where
    pdebug _ = return ()

instance DebugMonad IO where
    pdebug = putStrLn

    v <- newRef initV
    modifyRef v (+1)
    pdebug "debug"
    readRef v
