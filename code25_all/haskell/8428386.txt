fib' n = f n 0 1
f 0 a b = a
f n a b = f (n-1) b (a+b)

fib' 1 = f 1 0 1
       = f 0 1 1
       = 1
fib 1  = 1 (By definition)
therefore
fib' n = fib n for n = 1

fib' k  = fib k
f k 0 1 = fib k

fib' (k+1)  = fib (k+1)
f (k+1) 0 1 = fib k + fib (k-1)

f n (fib p) (fib (p+1)) = fib (p+n)

p = 0
fib' (n+p)  = fib (n+p)
f (n+p) 0 1 = fib (n+p)
