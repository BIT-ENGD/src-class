interp env (While b1 b2)  = do
  c1 <- interp env b1
  case c1 of
    (VBool True)  -> do
                         interp env b2
                         interp env (While b1 b2)
    (VBool False) -> return VNil

interp env ( For s a1 a2 a3)= do 
  v1 <- interp env a1
  v2 <- interp env a2 
  if  (v1<v2 ) then do 
    x <- interp env a1
    let env' = extendEnv x e1 env 
    interp env' a3 
  else return VNil
