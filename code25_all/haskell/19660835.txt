<meta>
    <sampleQuery>sample1</sampleQuery>
    <sampleQuery>sample2</sampleQuery>
</meta>

parseMeta =
  proc x -> do
    meta          <- deep (isElem >>> hasName "meta") -< x
    sampleQueries <- getText <<< getChildren <<< deep (hasName "sampleQuery") -< meta
    returnA -< Meta sampleQueries
