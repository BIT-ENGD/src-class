pyths
-- takes an Int and returns a list of pythagorean triples whose components 
---- are at most the given Int
pyths :: Int a => a -> [(Int, Int, Int)]
pyths n = [(x, y, z) | x <- f, y <- f, z <- f, x^2 + y^2 == z^2] 
  where f = factors n
