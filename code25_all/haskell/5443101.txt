data Polindrom = Polindrom { a :: Integer,
                             b :: Integer,
                             c :: Integer } deriving (Show)


euler_4 = [ p | y <- [10..20],
                z <- [10..20],
                let x = y * z,
                let p = Polindrom { a = y, b = z, c = x },   
                let s = show x,            
                s == reverse s]

}.sort { |a, b|
  b[2] <=> a[2]
}[0]
