howManyEqual :: Int -> Int -> Int -> Int
howManyEqual x y z
    | x == y && x == z && y == z = 3
    | 

howManyAboveAverage :: Int -> Int -> Int -> Int
howManyAboveAverage x y z
    | x > average(x y z) && y > average(x y z) && z > average(x y z) = 3
    where
    average a b c = ((a + b + c) / 3)
