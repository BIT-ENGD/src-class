data Quality = Odd | Even deriving Show

qual :: Integer -> Quality
qual x = case odd x of
             True -> Odd
             _ -> Even

type Qualifier = (Integer, Quality)

mkQ :: Integer -> Qualifier
mkQ x = (x, qual x)

plusQ :: Qualifier -> Qualifier -> Qualifier
plusQ (x, _) (y, _) = (x+y, qual (x+y))
