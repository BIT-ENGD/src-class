import Control.Monad
import qualified Data.HashTable as H
import System.Environment

main = do
  [size] <- fmap (fmap read) getArgs
  m <- H.new (==) H.hashInt
  forM_ [1..size] $ \n -> H.insert m n n
  v <- H.lookup m 100
  print v

:t getArgs ---> getArgs :: IO [String]
:t main    ---> main :: IO ()

getArgs :: IO [String]
<prog> 145
Just 100
main 145
