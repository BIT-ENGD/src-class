instance Num Complex where
  (C u v) + (C x y) = (C (u+x) (v+y))
  (C u v) * (C x y) = (C (u*x) (-v*y)) + (C (v*x) (u*y))
  (C u v) - (C x y) = (C (u+(-x)) (v+(-y)))
  negate
  abs       (C x y) = C (root (x*x + y*y)) 0
  signum    (C x y) = if (x==0 && y==0) then 0 else 1
  fromInteger    n  = C (fromInteger n) 0
