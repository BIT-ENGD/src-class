import System.IO
main = do        
       handle <- openFile "phone.txt" ReadMode  
       contents <- hGetContents handle 
       name<-getLine       
       putStrLn $ [findNumber name contents]

findNumber x [] = "not found"
findNumber x ((y,z):ys) = if x==y then z else findNumber x ys
