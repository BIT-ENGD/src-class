cut (x:xs) | x > 0 = x : (cut xs)
       | otherwise = []

first  (xs) = ( (foldl (+) 0 (cut          (xs)))   ) : []
second (xs) = ( (foldl (+) 0 (cut (reverse (xs))))  ) : []

test (xs) = first(xs) ++ second(xs)

cut [] = []
cut (x:xs) | x > 0 = foldl (+) 0 ( x : cut xs) : []
       | x == 0 = (cut xs)
