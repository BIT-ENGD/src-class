import Control.Monad (replicateM)

transpose :: [[a]]->[[a]]
transpose ([]:_) = []
transpose x = (map head x) : transpose (map tail x)

readMany :: Read a => IO [a]
readMany = fmap (map read . words) getLine

parse :: IO ([[Int]])
parse = do
        [m] <- readMany
        xss <- replicateM m readMany
        let matrix = transpose xss
        return (matrix)

main :: IO ()
main = do     
          parse

          print ??
