import Data.Fixed

main = do
    print x
  where
    x :: Pico
    x = read "12" -- error: no instance for 'Read Pico'

type Pico = Fixed E12

data E12 = E12

instance HasResolution E12 where
    resolution _ = 1000000000000

instance (HasResolution a) => Read (Fixed a) where
    readsPrec _ = readsFixed
