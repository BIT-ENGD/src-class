import System.IO
import Data.List
import Data.Char
printlist :: Show a => a -> IO ()
printlist x = putStr (show x)
main = do
     handle <- openFile "/usr/local/share/corpus" ReadMode
     text <- hGetContents handle
     let wlist = words text
         clist = map (\k -> take ((k + 15) - k + 1).drop (k - 10))(elemIndices "word" wlist)
printlist clist
