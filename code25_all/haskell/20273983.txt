classify xs
classify :: String -> String
classify (x:xs)
        |x == '+' = "PLUS"
        |x == '-' = "MINUS"
        |x == '*' = "MULT"
        |x == '/' = "DIV"
        |x == '(' = "LP"
        |x == ')' = "RP"
        |isAlpha x = "VAR"
        |isDigit x = "CONST"
        |otherwise = error "Cannot determine lexeme"
