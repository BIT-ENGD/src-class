compareTo :: [String] -> [String] -> Bool
compareTo x y
            |length (intersect x y) == length x     =True
            |otherwise                              =False 

removeDuplication :: [[String]] -> [[String]]
removeDuplication (x:xs) = nubBy compareTo (x:xs)
