import Data.Array.IO                                                                                          

data SomeData = SomeData {                                                                                    
  array :: IO (IOArray Int Float)                                                                             
}                                                                                                             

main = do                                                                                                     
  let d = SomeData { array <- newArray (1,255) 1.0 :: IO (IOArray Int Float) } 
  a <- readArray (d array) 1                                                                                  
  writeArray (d array) 1 64                                                                                   
  b <- readArray (d array) 1                                                                                  
  print (a,b)
