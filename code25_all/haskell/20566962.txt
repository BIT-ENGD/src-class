function main() {
    player = new Player()
    lastUpdateTime, lastDrawTime = getTime()
    while (!quit) {
        deltaTime = getTime() - lastUpdateTime
        player.update(deltaTime)
        if (getTime() - lastDrawTime > 1/60) {
            player.draw()
            lastDrawTime = getTime()
        }
        lastUpdateTime = getTime()
    }
    delete player
}
