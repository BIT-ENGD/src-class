data TeX
  = TeXt String
  | TeXmath String
  deriving (Show,Read,Eq)
teX (TeXt t)    = t
teX (TeXmath t) = "$$" ++ t ++ "$$"

class TeX t where
  teX :: t -> String

newtype TeXt = TeXt String
  deriving (Show,Read,Eq)
instance TeX TeXt where
  teX (TeXt t) = t

newtype TeXmath = TeXmath String
  deriving (Show,Read,Eq)
instance TeX TeXmath where
  teX (TeXmath t) = "$$" ++ t ++ "$$"
