hurdyGurdy c = concat [ if n `elem` "AEOUYaeouy" then [n] else [n, 'o', n] | n <- c]

unRovarsprak ("") = ("")
unRovarsprak (c1:'o':c2:xs)
       | isConsonant c1 = c1 : unRovarsprak xs
   | otherwise   = xs
unRovarsprak (c1:xs) 
   | c1 `elem` vokaler = c1 : unRovarsprak xs
   | otherwise      = xs
