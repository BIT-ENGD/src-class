import Control.Monad.ST
import Data.STRef

fourty_two = do
  x <- newSTRef (42::Int)
  readSTRef x

main = (print . runST) fourty_two -- (1)

main = (print . runST) $ fourty_two -- (2)

main = ((print . runST) $) fourty_two -- (3)

main = (($) (print . runST)) fourty_two -- (4)
