> leaves ' :: (Tree a) -> [a]
> leaves ' b = lvl b []
>              where lvl Nil as = as
>                    lvl (Leaf a) as = a:as
>                    lvl (Br l r) as = lvl l (lvl r as)

> data MultiTree a = Nil
>       | Leaf a
>       | Br [MultiTree a]  deriving Show

> multileaves :: MultiTree a -> [a]
> multileaves b = lvl b []
>             where lvl Nil as = as
>               lvl (Leaf a) as = a:as
>               lvl (Br a) as = Br a (map (multileaves (a:b) as)
