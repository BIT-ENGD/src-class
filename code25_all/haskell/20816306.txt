import Data.Char 
mywords s
       | dropWhile isSpace s == "" = []
       | dropWhile isSpace s == s' = w : mywords s'' 
            where (w, s'') = break isSpace s'

import Data.Char 
mywords s = if dropWhile isSpace s == "" then [] else w : mywords s'
            where (w, s') = break isSpace s

import Data.Char
mywords s                 =  case dropWhile isSpace s of
                                "" -> []
                                s' -> w : mywords s''
                                      where (w, s'') =
                                             break isSpace s'
