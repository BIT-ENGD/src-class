data Data = T | F
    deriving (Eq,  Show, Ord)

type PhoneBook = [(String, Data)] 

testBook :: Book
testBook = [("aT", T), ("bF", F)]

lookup1 :: PhoneBook->String->Data
lookup1 [] key = Nothing
lookup1 ((k,v):xs) key = if key == k
                            then Just v
                            else lookup1 xs key
