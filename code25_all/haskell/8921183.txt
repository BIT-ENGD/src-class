class numberClass
{
    private:
    int value;
    public:
    int read()
    {
        return value;
    }
    void load(int x)
    {
        value = x;
    }
    void increment()
    {
        value= value +1;
    }
};

int main()
{
    numberClass num;
    num.load(5);
    int x=num.read();
    cout<<x<<endl;
    num.increment();
    x=num.read();
    cout<<x;
}
