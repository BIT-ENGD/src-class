encodeChar :: Int -> [Char] -> Char -> Char
encodeChar x y z = eChar (offsetChar x y) z

encodeString :: Int -> [Char] -> String -> String
encodeString _ _ "" = ""
encodeString x y z = encodeChar x y (head z) ++ encodeString x y (tail z)
