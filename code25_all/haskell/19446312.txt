updown :: (Ord a) => [a] -> [[a]]
updown []         = [[]]
updown list       = smaller list

bigger :: (Ord a) => [a] -> [[a]]
bigger []         = [[]]
bigger [x]        = [[x]]
bigger list       = [n:p | n <- list,p <- smaller (filter (/=n) list),n < p!!0]

smaller :: (Ord a) => [a] -> [[a]]
smaller []        = [[]]
smaller [x]       = [[x]]
smaller list      = [n:p | n <- list,p <- bigger (filter (/=n) list),n > p!!0]
