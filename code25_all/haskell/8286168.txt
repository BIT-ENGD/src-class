Data Card = Card Int deriving (Show, Eq)

Type Cards = [Card]

 instance Ord Card where
    x > y   |ix == iy   = False
            |ix == 0    = True
            |iy == 0    = False
            |otherwise  = (ix > iy)
        where 
            ix = label x
            iy = label y

 (Card x) > (Card y) :: x,y are Int 

 [(Card x)] > [(Card y)] :: x,y are Int
