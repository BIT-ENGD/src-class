data Move = N | S | W | E | X
newtype Moves = Moves [Move]

createMoves:: [Char]-> Moves
createMoves (x:xs) =  if xs==[] then Moves [createMove(x)]
                      else Moves [createMove (x)] 

createMove:: Char-> Move
createMove (x) = if x=='N' then N
                 else if x=='S' then S
                 else if x=='W' then W
                 else if x=='E' then                      
                 else X
