intersperse myChar lists
    | lists == []          = ""  
    | otherwise            = attach myChar lists
        where   attach myChar (x:[]) = x 
                attach myChar (x:xs) = x ++ (myChar : []) ++ attach myChar xs 


| lists == (x:[]) = lists

| lists == (_:[]) = lists

| length lists == 1    = lists
