initialCards :: GameState
initialCards = [[Card s1 r1, Card s2 r2] | s1 <- [minBound .. maxBound]::[Suit], r1 <- [minBound.. maxBound]::[Rank], 
                                       s2 <- [minBound .. maxBound]::[Suit], r2 <- [minBound .. maxBound]::[Rank], 
                                       s1 /= s2, r1 /= r2]
