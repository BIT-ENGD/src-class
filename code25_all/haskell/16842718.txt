data Expression = Constant Integer
        | Negate Expression
        | Add Expression Expression
        | Multiply Expression Expression
   deriving Show

eval e = case e of
    Constant c -> c
    Negate e    -> -(eval e)
    Add e1 e2     -> (eval e1) + (eval e2)
    Multiply e1 e2 -> (eval e1) * (eval e2)
