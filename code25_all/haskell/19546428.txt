strReplace :: (Char, Char) -> String -> String -> String {- Original (y) Parsed (z) -}
strReplace _ "" y = y
strReplace x y z = if (y !! 0) == fst x then strReplace x (drop 1 y) (z:([snd x])) else        strReplace x (drop 1 y) (z:(y!!0))

*** Expression     : z : [snd x]
*** Term           : z
*** Type           : [Char]
*** Does not match : Char
