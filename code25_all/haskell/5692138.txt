import IO
import Char

isInteger "" = False
isInteger (a:b) =
  if length b == 0 && isDigit(a) == True then True
  else if isDigit(a) == True then isInteger(b)
  else False

main = do
    q <- getLine
    let x = read q
    if isInteger x == False then putStrLn "not integer"
    else putStrLn "integer"
