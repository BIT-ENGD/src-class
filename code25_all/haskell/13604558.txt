st1 = [("x", 2), ("y", 3), ("z", 3)]

update str val st = (str, val) : foldr step [] st where
    step x acc
        | fst x == str = acc
        | otherwise = x:acc

assign (str, val) env = update str val env

let st2 = update "x" 1 st1
