    public function match($path, $partial = false) 
    {
        if (false === $this->_doesModuleExists($path)) {
            return parent::match($path, $partial);
        }
        return array(
            'module' => 'static',
            'controller' => 'serve',
            'action' => 'view',
            'static_path' => $path
        );
    }
