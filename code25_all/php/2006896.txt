function checkOrphans($segment, $id, $db_table){
$data = array();
$this->db->select('id,name');
$this->db->where($id,id_clean($segment));
$Q = $this->db->get($db_table);
if ($Q->num_rows() > 0){
   foreach ($Q->result_array() as $row){
     $data[$row['id']] = $row['name'];
   }
}
$Q->free_result();  
return $data;
