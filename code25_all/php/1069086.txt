function showFields($selClient) 
    {
        $result = mysql_query("SELECT * FROM project WHERE projectid = $selClient");
        $values = array();
        while($row = mysql_fetch_array($result)) 
        {
            $values[] = array($row['clientname'],$row['prospect'],$row['salesperson']);
        }
        return $values;

  }
