<?php

class Conf {
    function Conf( $filename ) {
        $this->__construct( $filename );
    }

    function __construct( $filename ) {
        echo $filename;
    }

    function getInstance( $filename ) {
        static $instance = null;
        if ( $instance == null ) {
            $instance = new Conf( $filename );
        }
        return $instance;
    }
}
