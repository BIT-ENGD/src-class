function all_images(&$post){
$content = $post->post_content;
if(preg_match_all('/<img[^>]+src="(.*?)"[^>]*>/', $content, $results)){
    $i = 0;
    $count = count($results);
    $count = $count - 1;
    while($i < $count)
    {
        foreach($results as $result){
            echo $result[$i];
        }
        $i++;
    }
  }
}
