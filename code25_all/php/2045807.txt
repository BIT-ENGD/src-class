function getTotalDuration ($durations) {
    $total = 0;
    foreach ($durations as $duration) {
        $duration = explode(':',$duration);
        $total += $duration[0] * 60;
        $total += $duration[1];
    }
    $mins = floor($total / 60);
    $secs = str_pad ( $total % 60, '2', '0', STR_PAD_LEFT);
    return $mins.':'.$secs;
}
