protected function _initRoutes()
{
    $this->router = $this->frontController->getRouter();
    $route = new Zend_Controller_Router_Route(
        ':username',
        array(
            'module'     => 'default',
            'controller' => 'view',
            'action'     => 'profile',
            'username'   => ':username'
        )
    );
    $this->router->addRoute('profile', $route);
}
