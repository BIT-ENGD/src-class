if ($handle = opendir('photos')) {
  $count = 0;
  $list[] = array();
  while (false !== ($file = readdir($handle))) {
    if ($file != "." && $file != "..") {
      $list[$count] = $file;
      $count ++;
    }
  }
  closedir($handle);
  asort($list);
  $sorted_list = array();
  $sorted_list = array_values($list);
  foreach ($sorted_list as $i => $value) {
    echo "<li><img src=\"photos/$sorted_list[$i]\" alt=\"$sorted_list[$i]\" title=\"\"></li>\n";
  }
}
