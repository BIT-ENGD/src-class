function get_dirs($path = '.') {
$dirs = array();
foreach (new DirectoryIterator($path) as $file) {  //this is line 462
    if ($file->isDir() && !$file->isDot()) {
        $dirs[] = $file->getFilename();
    }
}

return $dirs;
