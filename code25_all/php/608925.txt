<?php

namespace String; // string helper functions

    function match($string1, $string2)
    {

        $string1 = trim($string1);
        $string2 = trim($string2);

        if (strcasecmp($string1, $string2) == 0) // are equal 
        {
            return true;
        } else {
            return false;
        }
    }


?>
