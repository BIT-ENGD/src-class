class Card
  attr_accessor :answer, :question
  def initialize(answer = "", question="")
    @answer = answer
    @question = question
  end

  def quiz
    puts "What does #@question mean?"
    answer = gets.chomp
    if answer == @answer
      puts "Right"
      return true
    else
      puts "Wrong"
      return answer
    end
  end
end

class Cardlist
  attr_accessor :Cards
  def initialize(Cards = [])
    @Cards = Cards
  end
  def quiz
    Cards.each do |w|
      w.quiz
    end
  end  
end

class Card < ActiveRecord::Base
  belongs_to :cardlist
end

class Cardlist < ActiveRecord::Base
  has_many :cards
end
