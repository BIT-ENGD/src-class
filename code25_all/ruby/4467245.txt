require 'rubygems'
require 'sinatra'
require 'candy'
require 'haml'

Candy.db = "Miroir"

class Post
    include Candy::Piece
end

class Posts
    include Candy::Collection
    collects :post
end

get '/' do
    @posts = Posts.all
    haml :index
end

!!! 5
%html
    %body
        %strong Posts
        %ul
        - @posts.each do |post|
            %li= post
