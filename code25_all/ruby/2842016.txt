
class Router :: Mongrel::HttpHandler

  def process(req, res)
    status, header, body = [200, {"Content-type"=>"text/html"}, Model.all.to_xml]

    res.start(status) do |head, out|
      header.each_pair { |key, value|  head[key] = value }
        out.write body
      end
  end

end
