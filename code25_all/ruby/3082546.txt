clean_link = link.attributes['href'].gsub(' ', '%20')
link_pieces = clean_link.scan(/http:\/\/(?:www\.)?([^\/]+?)(\/.*?\.mp3)/)

host = link_pieces[0][0]
path = link_pieces[0][1] 

Net::HTTP.start(host) do |http|
    resp = http.get(path)
    open("temp.mp3", "wb") do |file|
        file.write(resp.body)
    end
end
