class Game < ActiveRecord::Base
  has_one :home_team
  has_one :away_team
end

class Team < ActiveRecord::Base
  has_many :games
end

class Game < ActiveRecord::Base
  has_many :teams, :through => :game_teams, :limit => 2
end

class Team < ActiveRecord::Base
  has_many :games, :through => :game_teams
end

class Game_Teams < ActiveRecord::Base
  belongs_to :game
  belongs_to :team
end
