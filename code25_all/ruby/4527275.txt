def h(n)
    if n==0
        value = 0
    else
        c_n = $number[n-1].to_i
        value = ( h(n-1) * $A + c_n ) % $B
    end
    value
end

n, $A, $B, G = gets.scan(/\d+/).map! { |x| x.to_i }

(1..10**n).to_a.each {|x|
    $number = x.to_s.rjust(n, '0')
    if ( G == h(n) )
        puts $number
        break
    end
}
