def odd_or_even n
  if n%2 == 0
    return :even
  else
    return :odd
  end
end

simple = [1,2,3,4,5]

simple.map do
  |n| odd_or_even(n)
end
# => [:odd,:even,:odd,:even,:odd]

# does not work
simple.map do |n|
  if n%2 == 0
    return :even
  else
    return :odd
  end
end

# Desired result:
# => [:odd,:even,:odd,:even,:odd]
