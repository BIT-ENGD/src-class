var xhr = new XMLHttpRequest();
xhr.open("GET", "http://localhost:9292", true);
xhr.send();
xhr.onreadystatechange = function() {
    if(xhr.readyState == 4) {
      alert(xhr.status);
    }
}

run lambda { |env|
    response = Rack::Response.new("testing");
    response.finish
}
