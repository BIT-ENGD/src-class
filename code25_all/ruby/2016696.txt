module Log
  class Logger
    def info(msg, mod = '')
      puts "Module: #{mod}  Msg: #{msg}"
    end
  end # class Logger
end # module Log
$log = Log::Logger.new

module Foo
  class Bar
    def do_something
      # Do not pass in self.class.name.
      # We want the output to look like:

      $log.info "I did something!"
    end
  end # class Bar
