def split_array(array,size)
    index = 0
    results = []
    if size > 0
        while index <= array.size
            res = array[index,size]
            results << res if res.size != 0
            index += size
        end
    end
    return results
end

[1,2,3,4,5,6]
split_array([1,2,3,4,5,6],3)
[[1,2,3],[4,5,6]]
