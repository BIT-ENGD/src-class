$(document).ready(function() {
        $('.slides').each(function() {
        var $this = $(this), $ss = $this.closest('.slideshow');
            var prev = $ss.find('a.prev'), next = $ss.find('a.next');
            $this.cycle({
                prev: prev,
                next: next,
                fx: 'scrollLeft', 
                speed:  'fast', 
                timeout: 0 
            });
    });

    });
