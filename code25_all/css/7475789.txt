$(document).ready(function(){



$("#inner").css("overflow-x", "hidden"); 

var xPos = $('#scroller li:last').position();
var pos = '-' + xPos.left + 'px';
alert(pos);



$('#next').click(function(){

if(("#scroller ").css("margin-left") > pos ){



    $('#scroller').animate({

            marginLeft: "-=133px"



    }, 200)





}


});


$('#prev').click(function(){

    if($("#scroller").css("margin-left") < "0"){

        $('#scroller').animate({

            marginLeft: "+=133px"



    }, 200)


    }
});
