$(document).ready(function(){
$.getJSON("URL-GOES-HERE", function(data){
  $.each(data.items, function(i,item){
    $("<img/>").attr("src", item.media.m).addClass("thumb").appendTo(".flickr")
      .wrap("<a href='" + item.link + "'></a>").wrap("<div class='flickr-thumb' />");
  });
});
});
