    <script>
    $(document).ready(function(){

        $('.elem').mouseenter(function(){
            if ($(this).is(':animated')){
                $(this).stop();
            }
            $(this).hide().addClass('active').fadeTo(750,1);
        });

        $('.elem').mouseleave(function(){
            if ($(this).is(':animated')){
                $(this).stop();
            }
            $(this).fadeTo(750, 0, function(){
                $(this).removeClass('active').css({'filter' : 'none', 'display' : 'block'});
            });
        });

});
    </script>
