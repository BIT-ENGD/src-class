$(document).ready(function()
{
   $('#main-section').load('Home.html'); // Default
   $('.NavClick').click(function(e)
   {
      var url = $(this).attr('href');
      if(!url.match('^http'))
      {
         e.preventDefault();
         $('#main-section').load(url);
      }
   });
});
