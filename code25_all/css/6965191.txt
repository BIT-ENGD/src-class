rotateBanner = function() {
    var active = $('ul.rotating-banner li.active');
    var next = active.next();
    if (next.length === 0) {
        next = $('ul.rotating-banner li:first');
    }

    active.removeClass('active');
    next.addClass('active');
}
setInterval(rotateBanner, 1000);
