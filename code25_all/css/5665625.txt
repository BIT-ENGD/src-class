#stylesheets_controller.rb
class StylesheetsController < ApplicationController

  def user_css
  if user_signed_in?
    @user = current_user
  end

  respond_to do |format|
    format.css
  end
end

end

#user_css.css.erb
<%=
  if @user # User log in ?
    background_color = @user.color_index
  else
    background_color = "666"
  end
%>

.cadre {
   background-color : #<%= background_color %>;
   height: 50 px;
   width: 50 px;
}
