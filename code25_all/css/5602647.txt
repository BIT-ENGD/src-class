$(document).ready(function(){
    $("img:not('.Copy, .Show_Card')").live('dblclick', function(){
        var limit = $(this).parent().size();
        if (limit < 4) {
            var $clone = $(this).clone(true).addClass('Copy');
            $(this).parent().find('.DCT_Card:last').before($clone);
        }
    });
});
