jQuery('a').hover(function()
{
    jQuery(this).animate({fontSize : '+=5'});
}, function()
{
    jQuery(this).animate({fontSize : '-=5'});
});

jQuery('a').hover(function()
{
    jQuery(this).stop().animate({fontSize : '+=5'});
}, function()
{
    jQuery(this).stop().animate({fontSize : '-=5'});
});
