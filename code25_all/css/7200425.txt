$('iframe')
    .mouseover(function () {
        var html = $('html');
        html.data('previous-overflow', html.css('overflow'));
        html.css('overflow', 'hidden');
        html.css('margin-right', '17px');
    })
    .mouseout(function () {
        // un-lock scroll position
        var html = $('html');
        html.css('overflow', html.data('previous-overflow'));
        html.css('margin-right', '0px');
    });
