$("form#featuredItems").submit(function() {

    // we want to store the values from the form input box, then send via ajax below
    var $form = $(this);        
    $.ajax({
            type: "POST",
            url: "ajax/featuredItems.php",
            data: $form.serialize(),
            success: function(){
                showAction('Featured Items Saved');
                }
        }); // End .ajax function
    return false;
    }); //End submit function()
