$('div.box table a').click(function (e) {

    stopEvent(e);
    var self = $(this);

    if ($("#" + self.text()).css('display', 'none')) {
        $("#" + self.text()).show('slow');

        return;
    }
    else if ($("#" + self.text()).css('display', 'table-row')) {
        $("#" + self.text()).hide();

        return;
    }

});
