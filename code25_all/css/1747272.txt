$(function() {
    var drawer = $('#drawer');
    $('a').click( function() {
        drawer.html("<center>Hey Man!<br>This is a message.</center>");
        drawer.slideDown( function() {
                drawer.css("backgroundColor", "orange");
                setTimeout(function() { drawer.css("backgroundColor", "yellow"); }, 1000);
