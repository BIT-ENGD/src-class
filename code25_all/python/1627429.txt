from sqlalchemy import Table, Column, Integer, ForeignKeyConstraint

Base = declarative_base()

    Base.metadata, 
    Column('id', Integer, primary_key=True))

t2 = Table('t2', 
    Base.metadata, 
    Column('id', Integer, primary_key=True),
    Column('fkey', Integer),
    ForeignKeyConstraint(['fkey'], [t1.c.id]))

    __table__ = t1.join(t2)

class ST2(Base):
    __table__ = t1.join(t2).alias()
