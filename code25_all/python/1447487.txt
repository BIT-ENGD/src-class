class FH:

    def __init__(self, fh):
        self.fh = fh

    def close(self):
        self.fh.close()

    def seek(self, arg):
        self.fh.seek(arg)

    def fix(self, s):
        return ' '.join(s.split())

    def next(self):
        return self.fix(self.fh.next())

    def __iter__(self):
        for line in self.fh:
            yield self.fix(line)
