def PrintProductsBelowNumber(number):
    number1 = 1
    number2 = 1
    while number1 <= number:
        while number2 <= number:
            print(number1, "*", number2, "=", number1 * number2)
            number2 += 1
        number1 += 1

PrintProductsBelowNumber(2)

1 * 1 = 1
1 * 2 = 2

1 * 1 = 1
1 * 2 = 2
2 * 1 = 2
2 * 2 = 4
