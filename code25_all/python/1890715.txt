#! /usr/bin/python

import gtk

def test():
    retval = True
    def cb(widget):
        retval = False
        gtk.main_quit()

    window = gtk.Window(gtk.WINDOW_TOPLEVEL)
    button = gtk.Button("Test")
    button.connect("clicked", cb)
    button.show()
    window.add(button)
    window.show()

    gtk.main()
    return retval

if __name__ == "__main__":
    print test() # prints True when the button is clicked
