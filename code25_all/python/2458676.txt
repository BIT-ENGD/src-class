class File(file):

    def __init__(self, filename, *args, **kwargs):
        self.abspath = os.path.abspath(filename)
        file.__init__(self, filename, *args, **kwargs)

f = File('test','rb')
os.chdir('some_directory')
f.abspath # absolute path can be accessed like this
