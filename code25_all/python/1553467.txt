likes = {
    'colors': ['blue','red','purple'],
    'foods': ['apples', 'oranges']
}

favorites = {
    'colors':'yellow',
    'desserts':'ice cream'
}

for key in favorites:
    if key in likes:
        likes[key].append(favorites[key])
    else:
        likes[key] = list(favorites[key])

for key in favorites:
    try:
        likes[key].append(favorites[key])
    except KeyError:
        likes[key] = list(favorites[key])
