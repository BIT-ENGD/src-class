class Foo(db.Model):
    id = db.StringProperty()
    bar = db.StringProperty()
    baz = db.StringProperty()

foos = db.GqlQuery("SELECT * FROM Foo")

def toDict(self):
    return {
         'id': self.id,
         'bar': self.bar,
         'baz': self'baz
       }

fooDict = {}

for foo in foos:
    fooDict[foo.id] = foo.toDict()

return simplejson.dumps(fooDict)
