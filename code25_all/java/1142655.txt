private Object addVertex(String label, int posX, int posY) {
    Vertex newVertex = new Vertex();
    this.getModel().beginUpdate();
    try {
        newVertex = insertVertex(parent, null, label, posX, posY, 80, 30);
    }
    finally {
        this.getModel().endUpdate();
    }
    return newVertex;
}
