public Mutex getMutex( String id )
{                                  
    Mutex key = new MutexImpl( id );
    synchronized( mutexMap )
    {
        WeakReference<Mutex> ref = mutexMap.get( key );
        if( ref == null )
        {
            mutexMap.put( key, new WeakReference<Mutex>( key ) );
            return key;
        }
        Mutex mutex = ref.get();
        if( mutex == null )
        {
            mutexMap.put( key, new WeakReference<Mutex>( key ) );
            return key;
        }
        return mutex;
    }
}
