class SomeData1 {
    public int x;
    public int y;
}

class SomeData2 {
    int getX();
    void setX(int x);

    int getY();
    void setY(int y);

    private int x;
    private int y;
}

// a function in a class
public int f(SomeData1 d) {
    return (3 * d.x) / d.y;
}

// a function in a class
public int f(SomeData2 d) {
    return (3 * d.getX()) / d.getY();
}
