public static int[] countLetters(String line)
{
    int[] count = new int[26];

    for (int i = 0; i<line.length(); i++)
    {
        if (Character.isLetter(line.charAt(i)))
            count[(int)(line.charAt(i) - 'a')]++;
    }

    return count;
}
