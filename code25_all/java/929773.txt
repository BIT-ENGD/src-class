package org.totalbeginner.tutorial;

public class Punkt {

    public double x;
    public double y;

    Punkt(double xkoord, double ykoord){
        this.x = xkoord;
        this.y = ykoord;
    }

    public double getX() {
        return x;
    }

    public double getY() {
        return y;
    }    
}
