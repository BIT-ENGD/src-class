public class MatrixMultiply{

    public static void main(String[] args) {
        main();
    }


    static void main() {
        int[] vec = { 1, 2, 3, 4, 5 };
        System.out.println( mean( vec ) );
    }

    static int mean(int[] v) {
        int total = 0;

        for (int i = 0 ; i < v.length ; i++ ) {
            total = total + v[i];
        }
        return total / v.length;
    }
}
