public class Negsum {

    static double findSum(double array[], int n){
        double sum=0;
        if(array[n]>0) 
            return 1;
        else
            return array[n-1] + findSum(array, n-1);
    }

    public static void main(String args[]){
        double array[]={15, -7, -19, 8, 5, -6, -1};
        System.out.println(findSum(array, 5));
    }
}
