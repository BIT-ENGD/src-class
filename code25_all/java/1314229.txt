page
----
id int
topic varchar(128)
content varchar(1024)

page_l10n
---------
id int fk
topic varchar(128)
content varchar(1024)
locale_id int fk

locale
------
id int
name varchar(8)

SELECT
    COALESCE (hl.topic,h.topic) , COALESCE(hl.content, h.content)
    FROM page h 
    LEFT JOIN (page_l10n hl, `locale` l) ON (h.id=hl.id AND l.id=hl.locale_id) AND l.locale = 'ja_JP'
