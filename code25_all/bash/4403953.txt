#!/bin/bash

string="This is my string. I want THIS TEXT"
echo -e "string: $string"

length=${#string}
echo "length: $length"

start=$(($length-9))
echo "start: $start"

text=`expr substr $string $start 9`
echo "text: $text"

exit 0
