# check every line of check_list file presents in my_prog output
MY_LIST=`./my_prog`
for l in $(cat check_list); do
    if ! echo -n "$MY_LIST" | grep -q -x "$l"; then
        die "Bad line: '$l'"
    fi
done

# check every line of check_list file presents in my_prog output
./my_prog > my_list
for l in $(cat check_list); do
    if ! grep -q -x "$l" "my_list"; then
        die "Bad line: '$l'"
    fi
done
