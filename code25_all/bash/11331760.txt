startJob (){
    if [ $per -lt 10 ]; then
    folder=/storage/users/avityo/prog/0$1.$2$3
    else
    folder=/storage/users/avityo/prog/$1.$2$3
    fi

for per in $(seq 70 -5 0); do
    for seed in {580..583}; do
