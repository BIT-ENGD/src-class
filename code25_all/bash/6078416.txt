inDir() {
    if [ $# -gt 1 ]; then
        local dir="$1"
        local cwd=`pwd`
        shift

        if [ -d "$dir" ]; then
            cd "$dir" && "$@"
            cd "$cwd"
        fi
    fi
}

inDir /tmp { [ -e testFile ] && touch testFile }

cd
