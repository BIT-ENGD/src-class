#!/bin/bash

#remove the file name, and change every space into \space
shorter=`echo "$1" | sed 's/\/[^\/]*$//' | sed 's/\ /\\\ /g'`

echo $shorter
cd $shorter

$shorter
. script /my/path
./script /my/path
$shorter
/home/jack/my\ folder/subfolder
/usr/bin/script : line 4 : cd: /home/jack/my\: no file or folder of this type
cd '$shorter'
cd "'"$shorter"'"
