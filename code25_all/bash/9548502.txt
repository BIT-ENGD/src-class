while read line
 do
   if [ "/$e" == "$line" ]
    then
      exit 1
    elif [[ "${e:4}" == "$line" ]]
     then
       exit 1
    else
      echo "$line"|grep "$e"
      if [ $? -eq 0 ]
        then
          exit 1
     fi

     if [[ "${e:0:4}" == "/etc" ]]
      then
        echo "$line"|grep "\'${e:4}\'"
        if [ $? -eq 0 ]
         then
          exit 1
        fi
     fi
  fi
done < <(cat $EXCLUDEFILE)
