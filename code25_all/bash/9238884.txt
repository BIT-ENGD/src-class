    /usr/bin/sendEmail -v -f ip_watcher@xxxoo.com \
    -s smtp.gmail.com:587 -xu ip_watcher@xxxoo.com -xp xxxxxx \
    -t xxx@xxxxx.com \
    -o tls=yes \
    -u "$NEWIP" \
    -m "$NEWIP  $TIME" -a

else 
     logger "New IP is the SAME with old. not sending ..."
fi
