virtualenv:
    if [ ! -f $(PYTHON) ]; then \
        version=`$(VIRTUALENV) --version`;\
        check="1.8.2";\
        winner=`echo -e "$${version}\n$${check}" | sed '/^$$/d' | sort -nr | head -1`;\
        if [[ "$(winner)" = "$(version)" ]]; then \
            $(VIRTUALENV) $(ROOT_DIR)/lib/virtualenv; \
        else \
            echo "Please upgrade your virtualenv>=1.8.2";\
            exit 1;\
        fi \
    fi
