#!/bin/bash
fileslist="$1"
while read fn
do
  if [ ! -f `find . -type f -name $fn ` ];
  then
   echo $fn
  fi
done < $fileslist

#!/bin/bash
fileslist="$1"
foundfiles="~/tmp/tmp`date +%Y%m%d%H%M%S`.txt"
touch $foundfiles
while read fn
do
  `find . -type f -name $fn | sed 's:./.*/::' >> $foundfiles`
done < $fileslist
cat $fileslist $foundfiles | sort | uniq -u
rm $foundfiles
