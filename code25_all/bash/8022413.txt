#!/bin/bash

domain=(`cat /etc/trueuserdomains | cut -d: -f1`)
user=(`awk '{print $2}' /etc/trueuserdomains`)

for x in "${domain[@]}"
 do
  for y in "${user[@]}"; do
   ln -s /home/$y /var/www/vhosts/$x
  done
done
