    #!/bin/bash
PS3='Select an option and press Enter: '
select tree_1 in Date Host Users Quit
do
    case $tree_1 in
        Date)
            select tree_1_1 in Date1 Date2
            do
                case $tree_1_1 in
                    Date1) date;;
                    Date2) date;;
            esac
    done
        Host)  hostname;;
        Users)  who;;
        Quit)  break;;
    esac
done
