#!/bin/bash

create(){

 directory=$1

 for((j=0; j < depth; j++)); do


  for x in ${directory[@]}; do

    temp=( )

    for((i=0; i < breadth; i++)); do

        mkdir $x/$i 

        temp=($temp $x/$i)

    done    

  done  

    directory=($temp)

 done



}

breadth=3
depth=2

direc=(tree)

create $direc
