    CMDLN_ARGS="$@" # Command line arguments for this script (if any)
    export CMDLN_ARGS
    func_check_for_sudo() {
        if [ ! $( id -u ) -eq 0 ]; then
                echo "You may be asked for your login password for [`whoami`]." ;sleep 1
                LAUNCH="`dirname \"${0}\"`"
                exec sudo -S su -c ${LAUNCH}/$(basename ${0}) ${CMDLN_ARGS}
        exit ${?}
    fi
    }

    exec sudo -S su -c ${0} ${CMDLN_ARGS}
