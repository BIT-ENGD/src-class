#!/bin/bash
#servertest.sh


username=user1
password=xyz


url="http://wwww.streamingaudioserver.com -passwd $password -user $username"

mplayer $url &
sleep 5

test=$(pgrep -c mplayer)

if [ $test = 0 ]; then
    #server is down!
    mailfile="downmail.txt"
    /usr/sbin/ssmtp test@maildomain.com < "/home/test/$mailfile"  
fi

killall mplayer
sleep 5

exit
