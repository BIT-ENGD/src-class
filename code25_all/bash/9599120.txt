if [ "$1" == "on" ]; then

        echo "Maintenance going on"
        echo "<maintenance/>" > $MFILE

else if [ "$1" == "off" ]; then

        echo "Maintenance going off"
        rm $MFILE

else if [ "$1" == "status" ]; then

        if [ -f $MFILE ]; then
                echo "Maintenance is ON"
        else
                echo "Maintenance is OFF"
        fi

else

        echo "ERROR: Unknown command $1"
        exit 1

fi
