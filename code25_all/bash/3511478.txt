count=1

while [ $# -ge $count ]

do
        head $"$count"//error here
    echo -e "Delete file $$count?? (y/n):"//error here

    read answer
    case $answer in
               Y|y)rm $"$count"//error here
                 echo -e "File $$count deleted\n" ;;//error here
               N|n)echo -e "File $$count NOT deleted\n" ;;//error here
               *)echo "Invalid Command";;
    esac            

    count=`expr $count + 1`
done
