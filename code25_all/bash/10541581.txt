#!/bin/bash

DOMAINS="a.com b.com c.com"
NODES="w2 w3"

for DOMAIN in $DOMAINS; do
        for NODE in $NODES; do
                COMMAND="rsync -arv --exclude 'logs' --exclude '.git' /var/www/vhosts/$DOMAIN/ root@$NODE:/var/www/vhosts/$DOMAIN/"
                echo "$COMMAND"
                "$COMMAND"
        done
done

rsync
