#!/bin/bash

then
 if [ -d $1 ]
 then
  rm liste.txt
  touch liste.txt
  ls -l $1 | grep '^-' | sed -e "s/^-/File /g" >> liste.txt
  ls -l $1 | grep '^d' | sed -e "s/^d/Directory /g" >> liste.txt
 fi

fi
