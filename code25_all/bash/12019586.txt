# Tab completion
if [ "$1" = "--complete" ]; then
    shift
    # Commands
    for line in $(compgen -ac "notes_$1" | sed 's/^notes_//'); do
        echo $line
    done
    # Notes
    for line in $(cd $NOTES_DIR; ls $1* 2>/dev/null); do
        echo $line
    done
    return

notes --complete
complete -C 'notes --complete "$COMP_LINE"' notes
