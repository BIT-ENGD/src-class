#!/bin/bash
ls | while read -r FILE
do
    mv -v "$FILE" `echo $FILE | tr ' ' '_' `
done
ls | while read -r FILE
do
    mv -v "$FILE" `echo $FILE | tr '\*.JPEG' '\*.jpg' `
done
    mogrify -resize 750 *.jpg
wait 
    jpegoptim *.jpg –max=70 --strip-all
exit

do
    mv -v "$FILE" `echo $FILE | tr '\*.JPEG' '\*.jpg' `

    jpegoptim $DIR/*.jpg –max=70 --strip-all
