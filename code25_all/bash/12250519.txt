num1=1
num2=255

until [ -z $1 ] 

  case in "$1" in 

               a) $2="$num1"
                  $3="$num2" ;;
   [0-9]*-[0-9]*) $num1=${1%-*} 
                  $num2=${1#*-}

  esac
done

for ip in $(seq $num1 $num2); do
  ping -c 1 -w 1 192.168.1.$ip>/dev/null; && eho "192.168.1.$ip is up" ||:;
done
