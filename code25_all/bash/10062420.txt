function params()
{
    for PARAM in $1; do
        echo "$PARAM"
    done

    echo .

    for ITEM in $2; do
        echo "$ITEM"
    done
}

PARAMS="$@"
ITEMS="x y 'z t'"
params "$PARAMS" "$ITEMS"

function params()
{
    for PARAM in "${BASH_ARGV[@]}"; do

        echo "$ITEM"
    done
}
