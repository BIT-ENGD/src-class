#!/bin/bash

STARTDIR=$1
MNTDIR=/tmp/test/mnt

find $STARTDIR -type l  |
    while read file;
    do
       echo Found symlink file: $file
       DIR=`sed 's|/\w*$||'`
       MKDIR=${MNTDIR}${DIR}
       mkdir -p $MKDIR
       cp -L $file $MKDIR
    done
