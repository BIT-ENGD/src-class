#!/bin/sh
# Startup script for xxx
#
# chkconfig: 2345 99 99
. /etc/rc.d/init.d/functions

prog="file"


start() {
    echo -n $"Starting : "
    echo
    daemon  /path/to/some/file
    let RETVAL=RETVAL+$?
    return $RETVAL
}

stop() {
    if test "x`pidof -x $prog`" != x; then
        echo -n $"Stopping $prog: "
        killproc  $prog
        echo
    fi
    RETVAL=$?
    return $RETVAL
}

case "$1" in
        start)
            start
            ;;

        stop)
            stop
            ;;

        status)
            echo $prog
            status $prog
            ;;

        restart)
            stop
            start
            ;;
        *)
            echo $"Usage: $0 {start|stop|restart|status}"
            exit 1

esac
