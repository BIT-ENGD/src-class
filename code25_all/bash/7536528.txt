EXT=`echo $1 | cut -f2 -d"."`

if [ "$EXT" == "c" ]; then
    NAME=`echo $1 | cut -f1 -d"."`
    gcc -Wall -o "$NAME" "$1"   
elif [ "$EXT" == "java" ]; then
    NAME=`echo $1 | cut -f1 -d"."`
    gcj -c -g -O $1 && gcj --main="$NAME" -o "$NAME" "${NAME}.o"
else    
    echo "hm... I don't seem to know what to do with that"
fi
