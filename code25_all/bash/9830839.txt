#!/bin/bash

echo "Please enter a number: "
read number

counter=0
answer=0
end=$(( ${#number} - 1 ))

echo -n "The sum of all digits of "
echo -n $number
echo -n " is "

while [ $counter -lt ${#number} ] ; do
num=${number:$counter:1}
if [ $counter -lt $end ] ; then
echo -n $num
echo -n "+"
else
echo -n $num
echo -n "="
fi
answer=$(( $answer + $num ))
counter=`expr $counter + 1`
done

echo $answer
