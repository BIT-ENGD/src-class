jobsel
(only "#! /bin/bash" and help message is omitted)

slots=$1
sel=$2

[[ $slots -gt 0 ]] || die_usage
[[ $sel -lt $slots ]] || die_usage

i=0
while read line
do
        (( i % slots == sel )) && eval $line
        i=$(( i + 1 ))
done

# in case the last line does not end with EOL
if [[ $line != "" ]]; then
        (( i % slots == sel )) && eval $line
        i=$(( i + 1 ))
fi

eval
$HOME/util/jobsel 22 0 < cmds
cmds
echo 0 >> out
0, 22, 44...
at
&
ps -AFH
echo 0 >> out
at
at
at/atd
