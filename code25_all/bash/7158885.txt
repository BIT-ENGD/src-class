function system_info
{


if ls /etc/*release 1>/dev/null 2>&1; then
    echo "<h2>System release info</h2>"
    echo "<pre>"
    for i in /etc/*release; do

        # Since we can't be sure of the
        # length of the file, only
        # display the first line.

        head -n 1 $i
    done
    uname -orp
    echo "</pre>"
fi

}   # end of system_info
