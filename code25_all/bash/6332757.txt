date=$(date "+%F")

backupall="$backdest/arch-full-$date.tgz"

                        --exclude=/proc/* --exclude=/sys/* --exclude=/dev/*   \
                        --exclude=/tmp/* --exclude=/lost+found/*

tar -czpvf $backuphome /home/jary_p

$ls /home/backup
