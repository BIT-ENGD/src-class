#== don't touch ==
 dirpath=$(dirname "$1")
 dirname=$(basename $(dirname "$1"))

#== config ==
 path="$dirpath"   #using the value of $dirpath straight in config would be ugly.
 name="$dirname" 

 echo "$path" + "$name"
