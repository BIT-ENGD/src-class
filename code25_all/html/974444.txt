function openModal(url) {
    if(window.showModalDialog)  showModalDialog(url);

        try {
            netscape.security.PrivilegeManager.enablePrivilege("UniversalBrowserWrite");
            open(url, "", "modal=yes");
        } catch (e) {
            alert("dialog windows unsupported by browser");
        }
    }
}
