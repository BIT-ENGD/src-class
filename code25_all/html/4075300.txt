 'TextBoxDiv' + counter);
     counter++;
 });

 $("#removeButton").click(function () {
  if(counter==1){
      alert("No more textbox to remove");
      return false;
   }   
   counter--;
     $("#TextBoxDiv" + counter).remove();
  });
 });
 </script>
