        $(".accordion .head").click(function () {

            if ($(this).next().is(":visible")) {
                $(this).next().hide('fast');
                $(this).find('img').attr('src', '<%=Url.Content("~/Content/images/Arrow-Right.png")%>');
            }
            else {
                $(this).next().show('fast');
                $(this).find('img').attr('src', '<%=Url.Content("~/Content/images/Arrow-Down.png")%>');
            }
            return false;
        });
