var inp = $("#text1");

if ("onpropertychange" in inp)
inp.attachEvent($.proxy(function () {
    if (event.propertyName == "value")
        $("div").text(this.value);
}, inp));
 else
  inp.addEventListener("input", function () { 
    $("#text2").text(this.value);
}, false);



<input type="text" id="text1" />
<input type="text" id="text2" />
