 $('#InviteMessage').keyup(function(event)
    {
        var enter = "";

        //if(event.keyCode = '13')
            //enter = 'br />';

        var text = $(this).val() + enter;

        //replace all the less than/greater than characters

        if(text == '')
            $('#message').html('[Your personal message]');
        else
            $('#message').html(text);
        }
    );
