function contentStringHtml(data, name) {
    return '<div id="gm-event-title">' +
            '<h2>' + name + '</h2>' +
            '</div>' +
            '<div class=\"gm-trafic-index\"><ul>' +
            $.each(data.DataTime, function(i, item) {
                return '<li class=\"gm-trafic-li"\>' + item.TimeAndTo + '</li>';
            });
            + '</ul></div>';
}
