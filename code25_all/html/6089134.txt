 $(document).ready(function(){
    $('button:not("#delete")').click(function(e){
      $('#clipboard').append($(this).prev('div.images').html()
                 .replace(/\</ig, '&lt;')
                 .replace(/\>/ig, '&gt;'));
    });

    $('#delete').click(function(e){
        $('#clipboard').html(' ');

    });
    });
