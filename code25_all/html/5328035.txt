$(document).ready(function() {
    var filepath= $("#first");
    $('#button').click(function() {
        $.ajax({
            type:       "post",
            url:        "imageservlet",
            data:       "user="+filepath.val(),
            success:    function(msg) {
                            $("#profile_img").attr('src',msg);
                            $("#test").html(msg).fadeIn("fast");
                        }
        });
    });
});
