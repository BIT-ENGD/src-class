$( "#search" ).catcomplete({
        delay: 0,
        source: 'complete_client.php',
        minLength: 2,
        select: function( event, ui ) {
            $( "#id" ).val( ui.item.id );
            $( "#title" ).val( ui.item.value );
            $( "#spec" ).val( ui.item.spec );
            document.forms["main_search"].submit();
    }
    });
