        $('#head_menu a').each(
            function()
            {
               $(this).addClass('menu_part'); 
            }
        )

.menu_part
{
    border-left: 1px solid black;
    background-image: url(../images/bg_menu.png);
    float:left;
    padding: 2px 10px;
}

        $('#head_menu a').click(
            function(e)
            {
                $(this).removeClass('menu_part').addClass('menu_chose');
            });

 .menu_chose
    {
        background-image: url(../images/bg_menu_hover.png);                            
        color: #FFF;
    }

.menu_part:hover
{
    background-image: url(../images/bg_menu_hover.png);
    color: #FFF;
}
