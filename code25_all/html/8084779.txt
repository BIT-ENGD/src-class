<div id=container>
<div id=box>
<table>
<tr><td><a href=link></a>
<div id=page class=hidden></div></td></tr>
</table>
</div>
</div>

$(document).ready(function() {
$("td a").click(function() {
  var myUrl = $(this).attr("href");
  $("#page").load(myUrl);
  $(this).toggle();
  return false;
});
$(".hidden").hide();
});
