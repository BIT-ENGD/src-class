$('.content-item .readmore').toggle(function() {
    parent = $(this).parent();
    $(parent).children('div.next').fadeIn();
    $(this).html("Read less &raquo;");
}, function() {
    parent = $(this).parent();
    $(parent).children('div.next').fadeOut();
    $(this).html("Read more &raquo;");
});
