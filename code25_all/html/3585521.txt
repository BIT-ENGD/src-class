 $(document).ready(function() {
   $('.file').hide();

   $('a.toggle').click(function() {
      $('.file').slideToggle(1000);
      $(this).text($(this).text() == 'show' ? 'hide' : 'show');
      return false;
   });
 });

<div class="file-wrapper">
 <h5>(<a href="#" class="toggle">show</a>)</h5>

</div><!-- .file-wrapper -->
