    var $plot = $("<a />")
   .css({
        'width': plot.width,
        'height': plot.height,
        'top': plot.top,
        'left': plot.left,
        'background': plot.color,
    })
    .hover(function() {
        $content.html(plot.content);
    });
    $map.append($plot);
});
