<ul id="test">
    <li>test</li>
</ul>



 $.ajax ({
        async: true,
        url: "http://json",
        dataType: "json",
        success: function (data) {
            var map_stop = "";
            for (info in data) {
                 var locationid = info.Location_Name;   
            }
            map_stop += '<li>test' + locationid + '</li>';

            $("#test").html(map_stop);

        },
        error: function() {
             alert('error');   
        }
    });
