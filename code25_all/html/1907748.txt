function expandAll(){
        $('#myTableID>tbody>tr>td:nth-child(2)>div:nth-child(2)').each ( function() {
            html = $(this).html();
// Is it possible to add this Row with animation
            $(this).parent().parent().after( "<tr><td colspan='2'>&nbsp;</td><td colspan='15'>" + html + "</td></tr>" ).slideDown('slow');          
        } );
    }
