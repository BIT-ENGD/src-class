$(window).scroll(function(){
    $('#fixedmore').hide();
    var distanceTop = $('#last').offset().top - $(window).height();
    if  ($(window).scrollTop() < distanceTop){
        $('#fixedmore').fadeIn(3000);
    }

    else{
        $('#fixedmore').hide();
    };          

});
