$(function(){
  $.get("images.xml",{},function(xml){
    var output = "<ul>";

    $(xml).each(function() {
      smallImage = $(this).find("small").text();
      //bigImage = $(this).find("big").text();

      output += "<li>" + smallImage + "</li>";
      output += "</ul>";
    });

    $("#wrapper").append(output);
  });
});
