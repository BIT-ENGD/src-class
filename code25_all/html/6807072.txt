$(".unlike").click(function(){
  var pid = $(this).attr("id");
  var email = $("#hidden").val();
  var data = "pid="+pid+"&email="+email;
  $("#flash").show();
  $("#flash").fadeIn(500).html('<img src="wamp.gif" /> Loading...');
  $.ajax({
    type: "POST",
    url: "unlike.php",
    data: data,
    cache: false,
    success: function(html) {
        $("#flash").fadeOut(100);
        $("#likes"+id).text(html).replaceWith('');
    }
});
