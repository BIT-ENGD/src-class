<script type="text/javascript" language="javascript">
$(document).ready(function() {

 //Default Action

 $(".tab_content:first").show(); 

 //On Click Event
 $("ul.tabs li").click(function() {
  $("ul.tabs li").removeClass("active");

  $(".tab_content").hide();  
  var activeTab = $(this).find("a").attr("href");  
  $(activeTab).fadeIn(); 
  return false;
 });
});
