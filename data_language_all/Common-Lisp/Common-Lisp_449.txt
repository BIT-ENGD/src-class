#!/usr/bin/sbcl --script

(load "../src/load")

(setf *print-pretty* t)
(rnd:set-rnd-state 1)


(defun main ()
  (let ((num 1000000)
        (rep 1000))

  (format t "rndspace:~%")
  (time
    (loop repeat rep
          do (rnd:rndspace num 0d0 1d0)))

  (format t "with-rndspace:~%")
  (time
    (loop repeat rep
          do (rnd:with-rndspace (num 0d0 1d0 v) v)))))

; (main))
(main)
