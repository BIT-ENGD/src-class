(defmethod resolution ((gpu-array gpu-array-t))
  (let ((dim (gpu-array-dimensions gpu-array)))
    (make-array (length dim) :element-type 'single-float
                :initial-contents (mapcar (lambda (i) (coerce i 'single-float))
                                          dim))))
