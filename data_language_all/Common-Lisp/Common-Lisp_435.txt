(in-package :cl-user)
(defpackage caveman.request
  (:use :cl
        :clack.request)
  (:shadow :<request>
           :make-request)
  (:export :request-method
           :script-name
           :path-info
           :server-name
           :server-port
           :server-protocol
           :request-uri
           :uri-scheme
           :remote-addr
           :remote-port
           :query-string
           :raw-body
           :content-length
           :content-type
           :clack-handler

           :referer
           :user-agent
           :uploads

           :securep
           :cookies
           :body-parameter
           :query-parameter
           :parameter))
(in-package :caveman.request)

(cl-syntax:use-syntax :annot)

@export
(defclass <request> (clack.request:<request>) ()
  (:documentation "Class for Caveman Request."))

@export
(defun make-request (req)
  "Construct a request instance."
  (apply #'make-instance '<request>
         :allow-other-keys t
         :raw-body (shared-raw-body req)
         req))
