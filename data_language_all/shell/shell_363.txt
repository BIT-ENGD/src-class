function apply_git_config {
	eval "declare -A configMap="${1#*=}
	
	gitcfg=$(git config --list)
	
	for configKey in "${!configMap[@]}"
	do
		if [[ $gitcfg != *$configKey* ]]; then
			configValue="${configMap[$configKey]}"
			git config --global "$configKey" "$configValue"
		fi
	done
}
