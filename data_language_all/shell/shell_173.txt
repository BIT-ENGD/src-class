if [[ ! -o interactive ]]; then
    return
fi

compctl -K _rbenv rbenv

_rbenv() {
  local words completions
  read -cA words

  if [ "${#words}" -eq 2 ]; then
    completions="$(rbenv commands)"
  else
    completions="$(rbenv completions ${words[2,-2]})"
  fi

  reply=("${(ps:\n:)completions}")
}
#!/usr/bin/env bash

branch=master
filter=ACM

if test $# -eq 1; then
  branch=$1
else
  if test $# -eq 2; then
    branch=$1
    filter=$2
  fi
fi

git diff --name-only --diff-filter=$filter $branch#!/usr/bin/env bash
