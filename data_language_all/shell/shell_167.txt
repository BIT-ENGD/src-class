#!/usr/bin/env bash

src=$1
dst=$2

test -z $src && echo "source branch required." 1>&2 && exit 1
test -z $dst && echo "destination branch required." 1>&2 && exit 1

git checkout $dst \
  && git merge --no-ff $src \
  && git branch -d $src
#!/usr/bin/env bash

if test "$1" = "--all"; then
  git shortlog -n -s | awk '{print substr($0,index($0,$2)) " (" $1 ")"}'
  echo
fi

echo total `git rev-list --count HEAD`
