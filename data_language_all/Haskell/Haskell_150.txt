-- | Internal utilities only.
--
module Haxl.Core.Util
  ( compose
  , textShow
  ) where

import Data.Text (Text)

import qualified Data.Text as Text

-- | Composes a list of endofunctions.
compose :: [a -> a] -> a -> a
compose = foldr (.) id

textShow :: (Show a) => a -> Text
textShow = Text.pack . show
