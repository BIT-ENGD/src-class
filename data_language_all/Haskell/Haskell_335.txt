data Expr where
    Lit :: Integer -> Expr
    Add :: Expr -> Expr -> Expr
    Sub :: Expr -> Expr -> Expr
    Mul :: Expr -> Expr -> Expr
    Var :: String -> Expr -- new constructor 
    deriving (Eq, Show, Read)

instance Num Expr where
 fromInteger n = Lit n
 e1 + e2 = Add e1 e2
 e1 - e2 = Sub e1 e2
 e1 * e2 = Mul e1 e2 
 
 
run :: Expr -> Integer
run (Lit n) = n
run (Add a b) = run a + run b
run (Sub a b) = run a - run b
run (Mul a b) = run a * run b 

f :: Expr -> Expr
f x = x + 1 

g :: Expr -> Expr -> Expr
g x y = x * x + y + 2
