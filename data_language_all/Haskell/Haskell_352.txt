data T
  = Add T T
  | Mul T T
  | Div T T
  | Sub T T
  | Num Int

eval :: T -> Int
eval x = case x of
  Add a b -> eval a + eval b
  Mul a b -> eval a + eval b
  Div a b -> eval a + eval b
  Sub a b -> eval a + eval b
  Num a   -> a
type Name = String
type TypeVar = String
type TypeCon = String

data Expr
  = Lam Type Name Expr    -- \x -> a
  | Var Name              -- x
  | App Expr Expr         -- a b
  | TLam Name Expr        -- /\ a . b
  | TApp Expr Type        -- a [ b ]
  | Lit Literal           -- 1
  | Let Name Expr Expr    -- let x = v in a

data Type
  = TForall [Name] Type
  | TArr Type Type
  | TCon TypeCon
  | TVar TypeVar
  deriving (Show)

data Literal
  = LitInt Integer
  | LitChar Char
  deriving (Eq, Ord, Show)
