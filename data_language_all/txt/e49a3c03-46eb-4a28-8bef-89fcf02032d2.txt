seajs.use(['tree'], function(Tree) {

  var fields = [{
    width: 200
  },{
    header: '编号',
    name: 'id',
    width: 150,
    align: 'center'
  }, {
    header: '名称',
    name: 'name',
    render: function(value) {
      return '<b>' + value + '</b>';
    }
  }];
  new Tree({
    url: './tree.json',
    model: {
      fields: fields,
      title: 'title',
      height: 190
    },
  }).render();
});
