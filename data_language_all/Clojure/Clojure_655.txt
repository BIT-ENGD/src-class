(defn version-deprecations [version]
  (->> model
       (map (fn [[k m]]
              [k (mapv key (filter (fn [[option doc]]
                                     (= version (:deprecated-version doc)))
                                   (:model m)))]))

       (remove (comp empty? second))
       (into {})))
