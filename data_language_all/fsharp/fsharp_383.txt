let toSeq (iter:Iterable) =
    let rec loop (x:Iterator) =
        seq {
            if x.hasNext() then
                yield x.next()
                yield! (loop x)
            }
    iter.iterator() |> loop |> Array.ofSeq |> Seq.readonly
