; require/edge
(define-syntax (require/edge stx)
  (syntax-case stx ()
    [(_ ${type})
     (let ([transform (string->symbol (string-trim (string-replace (syntax->string #'(type)) "." "/")))])
       (with-syntax ([type (format-id stx "fractalide/modules/rkt/rkt-fbp/edges/~a" transform)] )
         #'(require type)))]
    [(_ type)
     #'(require type)]))
