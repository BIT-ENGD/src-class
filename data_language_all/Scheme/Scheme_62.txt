  (export compile-harlan harlan->c++)

   (rnrs)
   (harlan compile-opts)
   (harlan front compile-front)
   (harlan middle compile-middle)
   (elegant-weapons insert-prototypes)

  (passes
   (compile-harlan-frontend)
   (compile-harlan-middle)

   (compile-harlan)
   (harlan-format-c)))
