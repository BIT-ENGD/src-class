(import (scheme base)
        (srfi 106))

(define echo-client-socket (make-client-socket "localhost" "5000"))

(socket-send echo-client-socket (string->utf8 "hello\r\n"))
(socket-recv echo-client-socket (string-length "hello\r\n"))
