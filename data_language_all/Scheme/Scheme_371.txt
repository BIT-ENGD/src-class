;; test for md5 module

(test-section "md5")

(use rfc.md5)
(test-module 'rfc.md5)

(for-each
 (^[args]
   (test* "md5-digest-string" (car args)
          (digest-hexify (apply md5-digest-string (cdr args))))
   (test* "digest-string" (car args)
          (digest-hexify (apply digest-string <md5> (cdr args)))))
