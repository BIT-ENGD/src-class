using System.Linq;
using GitHubSharp.Models;

    public class SelectIssueLabelsMessage
    {
        public LabelModel[] Labels { get; }

        public SelectIssueLabelsMessage(IEnumerable<LabelModel> labels)
        {
            Labels = labels.ToArray();
        }
    }
}
