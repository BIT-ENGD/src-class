-module(emqx_cm_sup).

-behaviour(supervisor).

-export([start_link/0]).

-export([init/1]).

start_link() ->
    supervisor:start_link({local, ?MODULE}, ?MODULE, []).

init([]) ->
    Banned = #{id       => banned,
               start    => {emqx_banned, start_link, []},
               restart  => permanent,
               shutdown => 5000,
               type     => worker,
               modules  => [emqx_banned]},
    Manager = #{id       => manager,
                start    => {emqx_cm, start_link, []},
                restart  => permanent,
                shutdown => 5000,
                type     => worker,
                modules  => [emqx_cm]},
    {ok, {{one_for_one, 10, 100}, [Banned, Manager]}}.
